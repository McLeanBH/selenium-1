14/**
15 * Class for a step in a path expression
16 * http://www.w3.org/TR/xpath20/#id-steps.
17 *
18 * @extends {wgxpath.Expr}
19 * @constructor
20 * @param {!wgxpath.Step.Axis} axis The axis for this Step.
21 * @param {!wgxpath.NodeTest} test The test for this Step.
22 * @param {!wgxpath.Predicates=} opt_predicates The predicates for this
23 *     Step.
24 * @param {boolean=} opt_descendants Whether descendants are to be included in
25 *     this step ('//' vs '/').
26 */
27wgxpath.Step = function(axis, test, opt_predicates, opt_descendants) {
28  var axisCast = /** @type {!wgxpath.Step.Axis_} */ (axis);
29  wgxpath.Expr.call(this, wgxpath.DataType.NODESET);
30
31  /**
32   * @type {!wgxpath.Step.Axis_}
33   * @private
34   */
35  this.axis_ = axisCast;
36
37
38  /**
39   * @type {!wgxpath.NodeTest}
40   * @private
41   */
42  this.test_ = test;
43
44  /**
45   * @type {!wgxpath.Predicates}
46   * @private
47   */
48  this.predicates_ = opt_predicates || new wgxpath.Predicates([]);
49
50
51  /**
52   * Whether decendants are included in this step
53   *
54   * @private
55   * @type {boolean}
56   */
57  this.descendants_ = !!opt_descendants;
58
59  var quickAttrInfo = this.predicates_.getQuickAttr();
60  if (axis.supportsQuickAttr_ && quickAttrInfo) {
61    var attrName = quickAttrInfo.name;
62    attrName = wgxpath.userAgent.IE_DOC_PRE_9 ?
63        attrName.toLowerCase() : attrName;
64    var attrValueExpr = quickAttrInfo.valueExpr;
65    this.setQuickAttr({
66      name: attrName,
67      valueExpr: attrValueExpr
68    });
69  }
70  this.setNeedContextPosition(this.predicates_.doesNeedContextPosition());
71};
72goog.inherits(wgxpath.Step, wgxpath.Expr);

75/**
76 * @override
77 * @return {!wgxpath.NodeSet} The nodeset result.
78 */
79wgxpath.Step.prototype.evaluate = function(ctx) {
80  var node = ctx.getNode();
81  var nodeset = null;
82  var quickAttr = this.getQuickAttr();
83  var attrName = null;
84  var attrValue = null;
85  var pstart = 0;
86  if (quickAttr) {
87    attrName = quickAttr.name;
88    attrValue = quickAttr.valueExpr ?
89        quickAttr.valueExpr.asString(ctx) : null;
90    pstart = 1;
91  }
92  if (this.descendants_) {
93    if (!this.doesNeedContextPosition() &&
94        this.axis_ == wgxpath.Step.Axis.CHILD) {
95      nodeset = wgxpath.Node.getDescendantNodes(this.test_, node,
96          attrName, attrValue);
97      nodeset = this.predicates_.evaluatePredicates(nodeset, pstart);
98    } else {
99      var step = new wgxpath.Step(wgxpath.Step.Axis.DESCENDANT_OR_SELF,
100          new wgxpath.KindTest('node'));
101      var iter = step.evaluate(ctx).iterator();
102      var n = iter.next();
103      if (!n) {
104        nodeset = new wgxpath.NodeSet();
105      } else {
106        nodeset = this.evaluate_(/** @type {!wgxpath.Node} */ (n),
107            attrName, attrValue, pstart);
108        while ((n = iter.next()) != null) {
109          nodeset = wgxpath.NodeSet.merge(nodeset,
110              this.evaluate_(/** @type {!wgxpath.Node} */ (n), attrName,
111              attrValue, pstart));
112        }
113      }
114    }
115  } else {
116    nodeset = this.evaluate_(ctx.getNode(), attrName, attrValue, pstart);
117  }
118  return nodeset;
119};

122/**
123 * Evaluates this step on the given context to a nodeset.
124 *     (assumes this.descendants_ = false)
125 *
126 * @private
127 * @param {!wgxpath.Node} node The context node.
128 * @param {?string} attrName The name of the attribute.
129 * @param {?string} attrValue The value of the attribute.
130 * @param {number} pstart The first predicate to evaluate.
131 * @return {!wgxpath.NodeSet} The nodeset from evaluating this Step.
132 */
133wgxpath.Step.prototype.evaluate_ = function(
134    node, attrName, attrValue, pstart) {
135  var nodeset = this.axis_.func_(this.test_, node, attrName, attrValue);
136  nodeset = this.predicates_.evaluatePredicates(nodeset, pstart);
137  return nodeset;
138};

171/**
172 * @override
173 */
174wgxpath.Step.prototype.toString = function() {
175  var text = 'Step:';
176  text += wgxpath.Expr.indent('Operator: ' + (this.descendants_ ? '//' : '/'));
177  if (this.axis_.name_) {
178    text += wgxpath.Expr.indent('Axis: ' + this.axis_);
179  }
180  text += wgxpath.Expr.indent(this.test_);
181  if (this.predicates_.getLength()) {
182    var predicates = goog.array.reduce(this.predicates_.getPredicates(),
183        function(prev, curr) {
184          return prev + wgxpath.Expr.indent(curr);
185        }, 'Predicates:');
186    text += wgxpath.Expr.indent(predicates);
187  }
188  return text;
189};

261/**
262 * Creates an axis and maps the axis's name to that axis.
263 *
264 * @param {string} name The axis name.
265 * @param {function(!wgxpath.NodeTest, wgxpath.Node, ?string, ?string):
266 *     !wgxpath.NodeSet} func The function for this axis.
267 * @param {boolean} reverse Whether to iterate over nodesets in reverse.
268 * @param {boolean=} opt_supportsQuickAttr Whether quickAttr can be enabled
269 *     for this axis.
270 * @return {!wgxpath.Step.Axis} The axis.
271 * @private
272 */
273wgxpath.Step.createAxis_ =
274    function(name, func, reverse, opt_supportsQuickAttr) {
275  if (wgxpath.Step.nameToAxisMap_.hasOwnProperty(name)) {
276    throw Error('Axis already created: ' + name);
277  }
278  // The upcast and then downcast for the JSCompiler.
279  var axis = /** @type {!Object} */ (new wgxpath.Step.Axis_(
280      name, func, reverse, !!opt_supportsQuickAttr));
281  axis = /** @type {!wgxpath.Step.Axis} */ (axis);
282  wgxpath.Step.nameToAxisMap_[name] = axis;
283  return axis;
284};
