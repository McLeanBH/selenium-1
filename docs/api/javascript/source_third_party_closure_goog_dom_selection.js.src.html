29/**
30 * Sets the place where the selection should start inside a textarea or a text
31 * input
32 * @param {Element} textfield A textarea or text input.
33 * @param {number} pos The position to set the start of the selection at.
34 */
35goog.dom.selection.setStart = function(textfield, pos) {
36  if (goog.dom.selection.useSelectionProperties_(textfield)) {
37    textfield.selectionStart = pos;
38  } else if (goog.userAgent.IE) {
39    // destructuring assignment would have been sweet
40    var tmp = goog.dom.selection.getRangeIe_(textfield);
41    var range = tmp[0];
42    var selectionRange = tmp[1];
43
44    if (range.inRange(selectionRange)) {
45      pos = goog.dom.selection.canonicalizePositionIe_(textfield, pos);
46
47      range.collapse(true);
48      range.move('character', pos);
49      range.select();
50    }
51  }
52};

69/**
70 * Returns the start and end points of the selection within a textarea in IE.
71 * IE treats newline characters as \r\n characters, and we need to check for
72 * these characters at the edge of our selection, to ensure that we return the
73 * right cursor position.
74 * @param {TextRange} range Complete range object, e.g., "Hello\r\n".
75 * @param {TextRange} selRange Selected range object.
76 * @param {boolean} getOnlyStart Value indicating if only start
77 *     cursor position is to be returned. In IE, obtaining the end position
78 *     involves extra work, hence we have this parameter for calls which need
79 *     only start position.
80 * @return {Array.<number>} An array with the start and end positions where the
81 *     selection starts and ends or [0,0] if it was unable to find the
82 *     positions or no selection exists. Note that we can't reliably tell the
83 *     difference between an element that has no selection and one where
84 *     it starts and ends at 0. If getOnlyStart was true, we return
85 *     -1 as end offset.
86 * @private
87 */
88goog.dom.selection.getEndPointsTextareaIe_ = function(
89    range, selRange, getOnlyStart) {
90  // Create a duplicate of the selected range object to perform our actions
91  // against. Example of selectionRange = "" (assuming that the cursor is
92  // just after the \r\n combination)
93  var selectionRange = selRange.duplicate();
94
95  // Text before the selection start, e.g.,"Hello" (notice how range.text
96  // excludes the \r\n sequence)
97  var beforeSelectionText = range.text;
98  // Text before the selection start, e.g., "Hello" (this will later include
99  // the \r\n sequences also)
100  var untrimmedBeforeSelectionText = beforeSelectionText;
101  // Text within the selection , e.g. "" assuming that the cursor is just after
102  // the \r\n combination.
103  var selectionText = selectionRange.text;
104  // Text within the selection, e.g.,  "" (this will later include the \r\n
105  // sequences also)
106  var untrimmedSelectionText = selectionText;
107
108  // Boolean indicating whether we are done dealing with the text before the
109  // selection's beginning.
110  var isRangeEndTrimmed = false;
111  // Go over the range until it becomes a 0-lengthed range or until the range
112  // text starts changing when we move the end back by one character.
113  // If after moving the end back by one character, the text remains the same,
114  // then we need to add a "\r\n" at the end to get the actual text.
115  while (!isRangeEndTrimmed) {
116    if (range.compareEndPoints('StartToEnd', range) == 0) {
117      isRangeEndTrimmed = true;
118    } else {
119      range.moveEnd('character', -1);
120      if (range.text == beforeSelectionText) {
121        // If the start position of the cursor was after a \r\n string,
122        // we would skip over it in one go with the moveEnd call, but
123        // range.text will still show "Hello" (because of the IE range.text
124        // bug) - this implies that we should add a \r\n to our
125        // untrimmedBeforeSelectionText string.
126        untrimmedBeforeSelectionText += '\r\n';
127      } else {
128        isRangeEndTrimmed = true;
129      }
130    }
131  }
132
133  if (getOnlyStart) {
134    // We return -1 as end, since the caller is only interested in the start
135    // value.
136    return [untrimmedBeforeSelectionText.length, -1];
137  }
138  // Boolean indicating whether we are done dealing with the text inside the
139  // selection.
140  var isSelectionRangeEndTrimmed = false;
141  // Go over the selected range until it becomes a 0-lengthed range or until
142  // the range text starts changing when we move the end back by one character.
143  // If after moving the end back by one character, the text remains the same,
144  // then we need to add a "\r\n" at the end to get the actual text.
145  while (!isSelectionRangeEndTrimmed) {
146    if (selectionRange.compareEndPoints('StartToEnd', selectionRange) == 0) {
147      isSelectionRangeEndTrimmed = true;
148    } else {
149      selectionRange.moveEnd('character', -1);
150      if (selectionRange.text == selectionText) {
151        // If the selection was not empty, and the end point of the selection
152        // was just after a \r\n, we would have skipped it in one go with the
153        // moveEnd call, and this implies that we should add a \r\n to the
154        // untrimmedSelectionText string.
155        untrimmedSelectionText += '\r\n';
156      } else {
157        isSelectionRangeEndTrimmed = true;
158      }
159    }
160  }
161  return [
162    untrimmedBeforeSelectionText.length,
163    untrimmedBeforeSelectionText.length + untrimmedSelectionText.length];
164};

182/**
183 * Returns the start and end points of the selection inside a textarea or a
184 * text input.
185 * @param {Element} textfield A textarea or text input.
186 * @param {boolean} getOnlyStart Value indicating if only start
187 *     cursor position is to be returned. In IE, obtaining the end position
188 *     involves extra work, hence we have this parameter. In FF, there is not
189 *     much extra effort involved.
190 * @return {Array.<number>} An array with the start and end positions where the
191 *     selection starts and ends or [0,0] if it was unable to find the
192 *     positions or no selection exists. Note that we can't reliably tell the
193 *     difference between an element that has no selection and one where
194 *     it starts and ends at 0. If getOnlyStart was true, we return
195 *     -1 as end offset.
196 * @private
197 */
198goog.dom.selection.getEndPoints_ = function(textfield, getOnlyStart) {
199  var startPos = 0;
200  var endPos = 0;
201  if (goog.dom.selection.useSelectionProperties_(textfield)) {
202    startPos = textfield.selectionStart;
203    endPos = getOnlyStart ? -1 : textfield.selectionEnd;
204  } else if (goog.userAgent.IE) {
205    var tmp = goog.dom.selection.getRangeIe_(textfield);
206    var range = tmp[0];
207    var selectionRange = tmp[1];
208
209    if (range.inRange(selectionRange)) {
210      range.setEndPoint('EndToStart', selectionRange);
211      if (textfield.type == 'textarea') {
212        return goog.dom.selection.getEndPointsTextareaIe_(
213            range, selectionRange, getOnlyStart);
214      }
215      startPos = range.text.length;
216      if (!getOnlyStart) {
217        endPos = range.text.length + selectionRange.text.length;
218      } else {
219        endPos = -1;  // caller did not ask for end position
220      }
221    }
222  }
223  return [startPos, endPos];
224};

227/**
228 * Sets the place where the selection should end inside a text area or a text
229 * input
230 * @param {Element} textfield A textarea or text input.
231 * @param {number} pos The position to end the selection at.
232 */
233goog.dom.selection.setEnd = function(textfield, pos) {
234  if (goog.dom.selection.useSelectionProperties_(textfield)) {
235    textfield.selectionEnd = pos;
236  } else if (goog.userAgent.IE) {
237    var tmp = goog.dom.selection.getRangeIe_(textfield);
238    var range = tmp[0];
239    var selectionRange = tmp[1];
240
241    if (range.inRange(selectionRange)) {
242      // Both the current position and the start cursor position need
243      // to be canonicalized to take care of possible \r\n miscounts.
244      pos = goog.dom.selection.canonicalizePositionIe_(textfield, pos);
245      var startCursorPos = goog.dom.selection.canonicalizePositionIe_(
246          textfield, goog.dom.selection.getStart(textfield));
247
248      selectionRange.collapse(true);
249      selectionRange.moveEnd('character', pos - startCursorPos);
250      selectionRange.select();
251    }
252  }
253};

267/**
268 * Sets the cursor position within a textfield.
269 * @param {Element} textfield A textarea or text input.
270 * @param {number} pos The position within the text field.
271 */
272goog.dom.selection.setCursorPosition = function(textfield, pos) {
273  if (goog.dom.selection.useSelectionProperties_(textfield)) {
274    // Mozilla directly supports this
275    textfield.selectionStart = pos;
276    textfield.selectionEnd = pos;
277
278  } else if (goog.userAgent.IE) {
279    pos = goog.dom.selection.canonicalizePositionIe_(textfield, pos);
280
281    // IE has textranges. A textfield's textrange encompasses the
282    // entire textfield's text by default
283    var sel = textfield.createTextRange();
284
285    sel.collapse(true);
286    sel.move('character', pos);
287    sel.select();
288  }
289};

292/**
293 * Sets the selected text inside a textarea or a text input
294 * @param {Element} textfield A textarea or text input.
295 * @param {string} text The text to change the selection to.
296 */
297goog.dom.selection.setText = function(textfield, text) {
298  if (goog.dom.selection.useSelectionProperties_(textfield)) {
299    var value = textfield.value;
300    var oldSelectionStart = textfield.selectionStart;
301    var before = value.substr(0, oldSelectionStart);
302    var after = value.substr(textfield.selectionEnd);
303    textfield.value = before + text + after;
304    textfield.selectionStart = oldSelectionStart;
305    textfield.selectionEnd = oldSelectionStart + text.length;
306  } else if (goog.userAgent.IE) {
307    var tmp = goog.dom.selection.getRangeIe_(textfield);
308    var range = tmp[0];
309    var selectionRange = tmp[1];
310
311    if (!range.inRange(selectionRange)) {
312      return;
313    }
314    // When we set the selection text the selection range is collapsed to the
315    // end. We therefore duplicate the current selection so we know where it
316    // started. Once we've set the selection text we move the start of the
317    // selection range to the old start
318    var range2 = selectionRange.duplicate();
319    selectionRange.text = text;
320    selectionRange.setEndPoint('StartToStart', range2);
321    selectionRange.select();
322  } else {
323    throw Error('Cannot set the selection end');
324  }
325};

328/**
329 * Returns the selected text inside a textarea or a text input
330 * @param {Element} textfield A textarea or text input.
331 * @return {string} The selected text.
332 */
333goog.dom.selection.getText = function(textfield) {
334  if (goog.dom.selection.useSelectionProperties_(textfield)) {
335    var s = textfield.value;
336    return s.substring(textfield.selectionStart, textfield.selectionEnd);
337  }
338
339  if (goog.userAgent.IE) {
340    var tmp = goog.dom.selection.getRangeIe_(textfield);
341    var range = tmp[0];
342    var selectionRange = tmp[1];
343
344    if (!range.inRange(selectionRange)) {
345      return '';
346    } else if (textfield.type == 'textarea') {
347      return goog.dom.selection.getSelectionRangeText_(selectionRange);
348    }
349    return selectionRange.text;
350  }
351
352  throw Error('Cannot get the selection text');
353};

356/**
357 * Returns the selected text within a textarea in IE.
358 * IE treats newline characters as \r\n characters, and we need to check for
359 * these characters at the edge of our selection, to ensure that we return the
360 * right string.
361 * @param {TextRange} selRange Selected range object.
362 * @return {string} Selected text in the textarea.
363 * @private
364 */
365goog.dom.selection.getSelectionRangeText_ = function(selRange) {
366  // Create a duplicate of the selected range object to perform our actions
367  // against. Suppose the text in the textarea is "Hello\r\nWorld" and the
368  // selection encompasses the "o\r\n" bit, initial selectionRange will be "o"
369  // (assuming that the cursor is just after the \r\n combination)
370  var selectionRange = selRange.duplicate();
371
372  // Text within the selection , e.g. "o" assuming that the cursor is just after
373  // the \r\n combination.
374  var selectionText = selectionRange.text;
375  // Text within the selection, e.g.,  "o" (this will later include the \r\n
376  // sequences also)
377  var untrimmedSelectionText = selectionText;
378
379  // Boolean indicating whether we are done dealing with the text inside the
380  // selection.
381  var isSelectionRangeEndTrimmed = false;
382  // Go over the selected range until it becomes a 0-lengthed range or until
383  // the range text starts changing when we move the end back by one character.
384  // If after moving the end back by one character, the text remains the same,
385  // then we need to add a "\r\n" at the end to get the actual text.
386  while (!isSelectionRangeEndTrimmed) {
387    if (selectionRange.compareEndPoints('StartToEnd', selectionRange) == 0) {
388      isSelectionRangeEndTrimmed = true;
389    } else {
390      selectionRange.moveEnd('character', -1);
391      if (selectionRange.text == selectionText) {
392        // If the selection was not empty, and the end point of the selection
393        // was just after a \r\n, we would have skipped it in one go with the
394        // moveEnd call, and this implies that we should add a \r\n to the
395        // untrimmedSelectionText string.
396        untrimmedSelectionText += '\r\n';
397      } else {
398        isSelectionRangeEndTrimmed = true;
399      }
400    }
401  }
402  return untrimmedSelectionText;
403};

406/**
407 * Helper function for returning the range for an object as well as the
408 * selection range
409 * @private
410 * @param {Element} el The element to get the range for.
411 * @return {Array.<TextRange>} Range of object and selection range in two
412 *     element array.
413 */
414goog.dom.selection.getRangeIe_ = function(el) {
415  var doc = el.ownerDocument || el.document;
416
417  var selectionRange = doc.selection.createRange();
418  // el.createTextRange() doesn't work on textareas
419  var range;
420
421  if (el.type == 'textarea') {
422    range = doc.body.createTextRange();
423    range.moveToElementText(el);
424  } else {
425    range = el.createTextRange();
426  }
427
428  return [range, selectionRange];
429};

432/**
433 * Helper function for canonicalizing a position inside a textfield in IE.
434 * Deals with the issue that \r\n counts as 2 characters, but
435 * move('character', n) passes over both characters in one move.
436 * @private
437 * @param {Element} textfield The text element.
438 * @param {number} pos The position desired in that element.
439 * @return {number} The canonicalized position that will work properly with
440 *     move('character', pos).
441 */
442goog.dom.selection.canonicalizePositionIe_ = function(textfield, pos) {
443  if (textfield.type == 'textarea') {
444    // We do this only for textarea because it is the only one which can
445    // have a \r\n (input cannot have this).
446    var value = textfield.value.substring(0, pos);
447    pos = goog.string.canonicalizeNewlines(value).length;
448  }
449  return pos;
450};

453/**
454 * Helper function to determine whether it's okay to use
455 * selectionStart/selectionEnd.
456 *
457 * @param {Element} el The element to check for.
458 * @return {boolean} Whether it's okay to use the selectionStart and
459 *     selectionEnd properties on {@code el}.
460 * @private
461 */
462goog.dom.selection.useSelectionProperties_ = function(el) {
463  try {
464    return typeof el.selectionStart == 'number';
465  } catch (e) {
466    // Firefox throws an exception if you try to access selectionStart
467    // on an element with display: none.
468    return false;
469  }
