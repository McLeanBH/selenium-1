147/**
148 * Evaluate and interpret the result as a number.
149 *
150 * @param {!wgxpath.Context} ctx The context to evaluate the expression in.
151 * @return {number} The evaluated number value.
152 */
153wgxpath.Expr.prototype.asNumber = function(ctx) {
154  var exrs = this.evaluate(ctx);
155  if (exrs instanceof wgxpath.NodeSet) {
156    return exrs.number();
157  }
158  return +exrs;
159};

162/**
163 * Evaluate and interpret the result as a string.
164 *
165 * @param {!wgxpath.Context} ctx The context to evaluate the expression in.
166 * @return {string} The evaluated string.
167 */
168wgxpath.Expr.prototype.asString = function(ctx) {
169  var exrs = this.evaluate(ctx);
170  if (exrs instanceof wgxpath.NodeSet) {
171    return exrs.string();
172  }
173  return '' + exrs;
174};

177/**
178 * Evaluate and interpret the result as a boolean value.
179 *
180 * @param {!wgxpath.Context} ctx The context to evaluate the expression in.
181 * @return {boolean} The evaluated boolean value.
182 */
183wgxpath.Expr.prototype.asBool = function(ctx) {
184  var exrs = this.evaluate(ctx);
185  if (exrs instanceof wgxpath.NodeSet) {
186    return !!exrs.getLength();
187  }
188  return !!exrs;
