43/**
44 * Known element location strategies. The returned objects have two
45 * methods on them, "single" and "many", for locating a single element
46 * or multiple elements, respectively.
47 *
48 * Note that the versions with spaces are synonyms for those without spaces,
49 * and are specified at:
50 * https://code.google.com/p/selenium/wiki/JsonWireProtocol
51 * @private {Object.<string,bot.locators.strategy>}
52 * @const
53 */
54bot.locators.STRATEGIES_ = {
55  'className': bot.locators.className,
56  'class name': bot.locators.className,

58  'css': bot.locators.css,
59  'css selector': bot.locators.css,
60
61  'id': bot.locators.id,
62
63  'linkText': bot.locators.linkText,
64  'link text': bot.locators.linkText,
65
66  'name': bot.locators.name,

78/**
79 * Add or override an existing strategy for locating elements.
80 *
81 * @param {string} name The name of the strategy.
82 * @param {!bot.locators.strategy} strategy The strategy to use.
83 */
84bot.locators.add = function(name, strategy) {
85  bot.locators.STRATEGIES_[name] = strategy;

89/**
90 * Returns one key from the object map that is not present in the
91 * Object.prototype, if any exists.
92 *
93 * @param {Object} target The object to pick a key from.
94 * @return {string?} The key or null if the object is empty.
95 */
96bot.locators.getOnlyKey = function(target) {
97  for (var k in target) {
98    if (target.hasOwnProperty(k)) {
99      return k;
100    }
101  }
102  return null;
103};

106/**
107 * Find the first element in the DOM matching the target. The target
108 * object should have a single key, the name of which determines the
109 * locator strategy and the value of which gives the value to be
110 * searched for. For example {id: 'foo'} indicates that the first
111 * element on the DOM with the ID 'foo' should be returned.
112 *
113 * @param {!Object} target The selector to search for.
114 * @param {(Document|Element)=} opt_root The node from which to start the
115 *     search. If not specified, will use {@code document} as the root.
116 * @return {Element} The first matching element found in the DOM, or null if no
117 *     such element could be found.
118 */
119bot.locators.findElement = function(target, opt_root) {
120  var key = bot.locators.getOnlyKey(target);
121
122  if (key) {
123    var strategy = bot.locators.STRATEGIES_[key];
124    if (strategy && goog.isFunction(strategy.single)) {
125      var root = opt_root || bot.getDocument();
126      return strategy.single(target[key], root);
127    }
128  }
129  throw Error('Unsupported locator strategy: ' + key);
130};

133/**
134 * Find all elements in the DOM matching the target. The target object
135 * should have a single key, the name of which determines the locator
136 * strategy and the value of which gives the value to be searched
137 * for. For example {name: 'foo'} indicates that all elements with the
138 * 'name' attribute equal to 'foo' should be returned.
139 *
140 * @param {!Object} target The selector to search for.
141 * @param {(Document|Element)=} opt_root The node from which to start the
142 *     search. If not specified, will use {@code document} as the root.
143 * @return {!goog.array.ArrayLike.<Element>} All matching elements found in the
144 *     DOM.
145 */
146bot.locators.findElements = function(target, opt_root) {
147  var key = bot.locators.getOnlyKey(target);
148
149  if (key) {
150    var strategy = bot.locators.STRATEGIES_[key];
151    if (strategy && goog.isFunction(strategy.many)) {
152      var root = opt_root || bot.getDocument();
153      return strategy.many(target[key], root);
154    }
155  }
156  throw Error('Unsupported locator strategy: ' + key);
157};
