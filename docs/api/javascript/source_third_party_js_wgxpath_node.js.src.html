20/**
21 * Returns whether two nodes are equal.
22 *
23 * @param {wgxpath.Node} a The first node.
24 * @param {wgxpath.Node} b The second node.
25 * @return {boolean} Whether the nodes are equal.
26 */
27wgxpath.Node.equal = function(a, b) {
28  return (a == b) || (a instanceof wgxpath.IEAttrWrapper &&
29      b instanceof wgxpath.IEAttrWrapper && a.getNode() ==
30      b.getNode());

34/**
35 * Returns the string-value of the required type from a node.
36 *
37 * @param {!wgxpath.Node} node The node to get value from.
38 * @return {string} The value required.
39 */
40wgxpath.Node.getValueAsString = function(node) {
41  var t = null, type = node.nodeType;
42  // Old IE title problem.
43  var needTitleFix = function(node) {
44    return wgxpath.userAgent.IE_DOC_PRE_9 &&
45        node.nodeName.toLowerCase() == 'title';
46  };
47  // goog.dom.getTextContent doesn't seem to work
48  if (type == goog.dom.NodeType.ELEMENT) {
49    t = node.textContent;
50    t = (t == undefined || t == null) ? node.innerText : t;
51    t = (t == undefined || t == null) ? '' : t;
52  }
53  if (typeof t != 'string') {
54    if (needTitleFix(node) && type == goog.dom.NodeType.ELEMENT) {
55      t = node.text;
56    } else if (type == goog.dom.NodeType.DOCUMENT ||
57        type == goog.dom.NodeType.ELEMENT) {
58      node = (type == goog.dom.NodeType.DOCUMENT) ?
59          node.documentElement : node.firstChild;
60      var i = 0, stack = [];
61      for (t = ''; node;) {
62        do {
63          if (node.nodeType != goog.dom.NodeType.ELEMENT) {
64            t += node.nodeValue;
65          }
66          if (needTitleFix(node)) {
67            t += node.text;
68          }
69          stack[i++] = node; // push
70        } while (node = node.firstChild);
71        while (i && !(node = stack[--i].nextSibling)) {}
72      }
73    } else {
74      t = node.nodeValue;
75    }
76  }
77  return '' + t;
78};

103/**
104 * Returns if the attribute matches the given value.
105 *
106 * @param {!wgxpath.Node} node The node to get value from.
107 * @param {?string} name The attribute name to match, if any.
108 * @param {?string} value The attribute value to match, if any.
109 * @return {boolean} Whether the node matches the attribute, if any.
110 */
111wgxpath.Node.attrMatches = function(node, name, value) {
112  // No attribute.
113  if (goog.isNull(name)) {
114    return true;
115  }
116  // TODO: If possible, figure out why this throws an exception in some
117  // cases on IE < 9.
118  try {
119    if (!node.getAttribute) {
120      return false;
121    }
122  } catch (e) {
123    return false;
124  }
125  if (wgxpath.userAgent.IE_DOC_PRE_8 && name == 'class') {
126    name = 'className';
127  }
128  return value == null ? !!node.getAttribute(name) :
129      (node.getAttribute(name, 2) == value);
130};

136 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.

144    opt_attrValue, opt_nodeset) {
145  var nodeset = opt_nodeset || new wgxpath.NodeSet();
146  var func = wgxpath.userAgent.IE_DOC_PRE_9 ?
147      wgxpath.Node.getDescendantNodesIEPre9_ :
148      wgxpath.Node.getDescendantNodesGeneric_;
149  var attrName = goog.isString(opt_attrName) ? opt_attrName : null;
150  var attrValue = goog.isString(opt_attrValue) ? opt_attrValue : null;
151  return func.call(null, test, node, attrName, attrValue, nodeset);

155/**
156 * Returns the descendants of a node for IE.
157 *
158 * @private
159 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.
160 * @param {!wgxpath.Node} node The node to get descendants from.
161 * @param {?string} attrName The attribute name to match, if any.
162 * @param {?string} attrValue The attribute value to match, if any.
163 * @param {!wgxpath.NodeSet} nodeset The node set to add descendants to.
164 * @return {!wgxpath.NodeSet} The nodeset with descendants.
165 */
166wgxpath.Node.getDescendantNodesIEPre9_ = function(test, node, attrName,
167    attrValue, nodeset) {
168  if (wgxpath.Node.doesNeedSpecialHandlingIEPre9_(test, attrName)) {
169    var descendants = node.all;
170    if (!descendants) {
171      return nodeset;
172    }
173    var name = wgxpath.Node.getNameFromTestIEPre9_(test);
174    // all.tags not working.
175    if (name != '*') {
176      descendants = node.getElementsByTagName(name);
177      if (!descendants) {
178        return nodeset;
179      }
180    }
181    if (attrName) {
182      /**
183       * The length property of the "all" collection is overwritten
184       * if there exists an element with id="length", therefore we
185       * have to iterate without knowing the length.
186       */
187      var result = [];
188      var i = 0;
189      while (node = descendants[i++]) {
190        if (wgxpath.Node.attrMatches(node, attrName, attrValue)) {
191          result.push(node);
192        }
193      }
194      descendants = result;
195    }
196    var i = 0;
197    while (node = descendants[i++]) {
198      if (name != '*' || node.tagName != '!') {
199        nodeset.add(node);
200      }
201    }
202    return nodeset;
203  }
204  wgxpath.Node.doRecursiveAttrMatch_(test, node, attrName,
205      attrValue, nodeset);
206  return nodeset;
207};

210/**
211 * Returns the descendants of a node for browsers other than IE.
212 *
213 * @private
214 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.
215 * @param {!wgxpath.Node} node The node to get descendants from.
216 * @param {?string} attrName The attribute name to match, if any.
217 * @param {?string} attrValue The attribute value to match, if any.
218 * @param {!wgxpath.NodeSet} nodeset The node set to add descendants to.
219 * @return {!wgxpath.NodeSet} The nodeset with descendants.
220 */
221wgxpath.Node.getDescendantNodesGeneric_ = function(test, node,
222    attrName, attrValue, nodeset) {
223  if (node.getElementsByName && attrValue && attrName == 'name' &&
224      !goog.userAgent.IE) {
225    var nodes = node.getElementsByName(attrValue);
226    goog.array.forEach(nodes, function(node) {
227      if (test.matches(node)) {
228        nodeset.add(node);
229      }
230    });
231  } else if (node.getElementsByClassName && attrValue && attrName == 'class') {
232    var nodes = node.getElementsByClassName(attrValue);
233    goog.array.forEach(nodes, function(node) {
234      if (node.className == attrValue && test.matches(node)) {
235        nodeset.add(node);
236      }
237    });
238  } else if (test instanceof wgxpath.KindTest) {
239    wgxpath.Node.doRecursiveAttrMatch_(test, node, attrName,
240        attrValue, nodeset);
241  } else if (node.getElementsByTagName) {
242    var nodes = node.getElementsByTagName(test.getName());
243    goog.array.forEach(nodes, function(node) {
244      if (wgxpath.Node.attrMatches(node, attrName, attrValue)) {
245        nodeset.add(node);
246      }
247    });
248  }
249  return nodeset;
250};

274/**
275 * Returns the child nodes of a node for IE browsers.
276 *
277 * @private
278 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.
279 * @param {!wgxpath.Node} node The node to get child nodes from.
280 * @param {?string} attrName The attribute name to match, if any.
281 * @param {?string} attrValue The attribute value to match, if any.
282 * @param {!wgxpath.NodeSet} nodeset The node set to add child nodes to.
283 * @return {!wgxpath.NodeSet} The nodeset with child nodes.
284 */
285wgxpath.Node.getChildNodesIEPre9_ = function(test, node,
286    attrName, attrValue, nodeset) {
287  var children;
288  if (wgxpath.Node.doesNeedSpecialHandlingIEPre9_(test, attrName) &&
289      (children = node.childNodes)) { // node.children seems buggy.
290    var name = wgxpath.Node.getNameFromTestIEPre9_(test);
291    if (name != '*') {
292      //children = children.tags(name); // children.tags seems buggy.
293      children = goog.array.filter(children, function(child) {
294        return child.tagName && child.tagName.toLowerCase() == name;
295      });
296      if (!children) {
297        return nodeset;
298      }
299    }
300    if (attrName) {
301      // TODO: See if an optimization is possible.
302      children = goog.array.filter(children, function(n) {
303        return wgxpath.Node.attrMatches(n, attrName, attrValue);
304      });
305    }
306    goog.array.forEach(children, function(node) {
307      if (name != '*' || node.tagName != '!' &&
308          !(name == '*' && node.nodeType != goog.dom.NodeType.ELEMENT)) {
309        nodeset.add(node);
310      }
311    });
312    return nodeset;
313  }
314  return wgxpath.Node.getChildNodesGeneric_(test, node, attrName,
315      attrValue, nodeset);
316};

319/**
320 * Returns the child nodes of a node genericly.
321 *
322 * @private
323 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.
324 * @param {!wgxpath.Node} node The node to get child nodes from.
325 * @param {?string} attrName The attribute name to match, if any.
326 * @param {?string} attrValue The attribute value to match, if any.
327 * @param {!wgxpath.NodeSet} nodeset The node set to add child nodes to.
328 * @return {!wgxpath.NodeSet} The nodeset with child nodes.
329 */
330wgxpath.Node.getChildNodesGeneric_ = function(test, node, attrName,
331    attrValue, nodeset) {
332  for (var current = node.firstChild; current; current = current.nextSibling) {
333    if (wgxpath.Node.attrMatches(current, attrName, attrValue)) {
334      if (test.matches(current)) {
335        nodeset.add(current);
336      }
337    }
338  }
339  return nodeset;
340};

343/**
344 * Returns whether a getting descendants/children call
345 * needs special handling on IE browsers.
346 *
347 * @private
348 * @param {!wgxpath.NodeTest} test A NodeTest for matching nodes.
349 * @param {!wgxpath.Node} node The root node to start the recursive call on.
350 * @param {?string} attrName The attribute name to match, if any.
351 * @param {?string} attrValue The attribute value to match, if any.
352 * @param {!wgxpath.NodeSet} nodeset The NodeSet to add nodes to.
353 */
354wgxpath.Node.doRecursiveAttrMatch_ = function(test, node,
355    attrName, attrValue, nodeset) {
356  for (var n = node.firstChild; n; n = n.nextSibling) {
357    if (wgxpath.Node.attrMatches(n, attrName, attrValue) &&
358        test.matches(n)) {
359      nodeset.add(n);
360    }
361    wgxpath.Node.doRecursiveAttrMatch_(test, n, attrName,
362        attrValue, nodeset);
363  }
364};

370 *
371 * @private

373 * @param {?string} attrName The attribute name to match, if any.

376wgxpath.Node.doesNeedSpecialHandlingIEPre9_ = function(test, attrName) {
377  return test instanceof wgxpath.NameTest ||
378      test.getType() == goog.dom.NodeType.COMMENT ||
379      (!!attrName && goog.isNull(test.getType()));

383/**
384 * Returns a fixed name of a NodeTest for IE browsers.
385 *
386 * @private
387 * @param {!wgxpath.NodeTest} test A NodeTest.
388 * @return {string} The name of the NodeTest.
389 */
390wgxpath.Node.getNameFromTestIEPre9_ = function(test) {
391  if (test instanceof wgxpath.KindTest) {
392    if (test.getType() == goog.dom.NodeType.COMMENT) {
393      return '!';
394    } else if (goog.isNull(test.getType())) {
395      return '*';
396    }
397  }
398  return test.getName();
