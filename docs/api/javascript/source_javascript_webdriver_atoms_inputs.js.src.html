31/**
32 * Send keyboard input to a particular element.
33 *
34 * @param {Element} element The element to send the keyboard input to, or
35 *     {@code null} to use the document's active element.
36 * @param {!Array.<string>} keys The keys to type on the element.
37 * @param {{currentPos: number, pressed: !Array.<!bot.Keyboard.Key>}=} opt_state
38 *     The predefined keyboard state to use.
39 * @param {boolean=} opt_persistModifiers Whether modifier keys should remain
40 *     pressed when this function ends.
41 * @return {{currentPos: number, pressed: !Array.<!bot.Keyboard.Key>}} The
42 *     keyboard state.
43 */
44webdriver.atoms.inputs.sendKeys = function(
45    element, keys, opt_state, opt_persistModifiers) {
46  var keyboard = new bot.Keyboard(opt_state);
47  if (!element) {
48    element = bot.dom.getActiveElement(document);
49  }
50  if (!element) {
51    throw Error('No element to send keys to');
52  }
53  webdriver.atoms.element.type(element, keys, keyboard, opt_persistModifiers);
54
55  return keyboard.getState();
56};
57goog.exportSymbol('webdriver.atoms.inputs.sendKeys',
58                  webdriver.atoms.inputs.sendKeys);

61/**
62 * Click on an element.
63 *
64 * @param {Element} element The element to click.
65 * @param {bot.Mouse.State=} opt_state The serialized state of the mouse.
66 * @return {!bot.Mouse.State} The mouse state.
67 */
68webdriver.atoms.inputs.click = function(element, opt_state) {
69  var mouse = new bot.Mouse(opt_state);
70  if (!element) {
71    element = mouse.getState().element;
72  }
73  if (!element) {
74    throw Error('No element to send keys to');
75  }
76  bot.action.click(element, null, mouse);
77  return mouse.getState();
78};
79goog.exportSymbol('webdriver.atoms.inputs.click',
80                  webdriver.atoms.inputs.click);

83/**
84 * Move the mouse to a specific element and/or coordinate location.
85 *
86 * @param {!Element} element The element to move the mouse to.
87 * @param {number} x_offset The x coordinate to use as an offset.
88 * @param {number} y_offset The y coordinate to use as an offset.
89 * @param {bot.Mouse.State=} opt_state The serialized state of the mouse.
90 * @return {!bot.Mouse.State} The mouse state.
91 */
92webdriver.atoms.inputs.mouseMove = function(element, x_offset, y_offset,
93    opt_state) {
94  var mouse = new bot.Mouse(opt_state);
95  var target = element || mouse.getState().element;
96
97  var offset_specified = (x_offset != null) && (y_offset != null);
98  x_offset = x_offset || 0;
99  y_offset = y_offset || 0;
100
101  // If we have specified an element and no offset, we should
102  // move the mouse to the center of the specified element.
103  if (element) {
104    if (!offset_specified) {
105      var source_element_size = bot.action.getInteractableSize(element);
106      x_offset = Math.floor(source_element_size.width / 2);
107      y_offset = Math.floor(source_element_size.height / 2);
108    }
109  } else {
110    // Moving to an absolute offset from the current target element,
111    // so we have to account for the existing offset of the current
112    // mouse position to the element origin (upper-left corner).
113    var pos = goog.style.getClientPosition(target);
114    x_offset += (mouse.getState().clientXY.x - pos.x);
115    y_offset += (mouse.getState().clientXY.y - pos.y);
116  }
117
118  var doc = goog.dom.getOwnerDocument(target);
119  var win = goog.dom.getWindow(doc);
120  var inViewAfterScroll = bot.action.scrollIntoView(
121      target,
122      new goog.math.Coordinate(x_offset, y_offset));
123
124  var coords = new goog.math.Coordinate(x_offset, y_offset);
125  mouse.move(target, coords);
126  return mouse.getState();
127};
128goog.exportSymbol('webdriver.atoms.inputs.mouseMove',
129                  webdriver.atoms.inputs.mouseMove);
