11/**
12 * Constructs a lexer.
13 *
14 * @param {!Array.<string>} tokens Tokens to iterate over.
15 * @constructor
16 */
17wgxpath.Lexer = function(tokens) {
18  /**
19   * @type {!Array.<string>}
20   * @private
21   */
22  this.tokens_ = tokens;
23
24  /**
25   * @type {number}
26   * @private
27   */
28  this.index_ = 0;
29};
30
31
32/**
33 * Tokenizes a source string into an array of tokens.
34 *
35 * @param {string} source Source string to tokenize.
36 * @return {!wgxpath.Lexer} Essentially an iterator over the tokens.
37 */
38wgxpath.Lexer.tokenize = function(source) {
39  var tokens = source.match(wgxpath.Lexer.TOKEN_);
40
41  // Removes tokens starting with whitespace from the array.
42  for (var i = 0; i < tokens.length; i++) {
43    if (wgxpath.Lexer.LEADING_WHITESPACE_.test(tokens[i])) {
44      tokens.splice(i, 1);
45    }
46  }
47  return new wgxpath.Lexer(tokens);
48};
49
50
51/**
52 * Regular expressions to match XPath productions.
53 *
54 * @const
55 * @type {!RegExp}
56 * @private
57 */
58wgxpath.Lexer.TOKEN_ = new RegExp(
59    '\\$?(?:(?![0-9-])[\\w-]+:)?(?![0-9-])[\\w-]+' +
60        // Nodename (possibly with namespace) or variable.
61    '|\\/\\/' + // Double slash.
62    '|\\.\\.' + // Double dot.
63    '|::' + // Double colon.
64    '|\\d+(?:\\.\\d*)?' + // Number starting with digit.
65    '|\\.\\d+' + // Number starting with decimal point.
66    '|"[^"]*"' + // Double quoted string.
67    '|\'[^\']*\'' + // Single quoted string.
68    '|[!<>]=' + // Operators
69    '|\\s+' + // Whitespaces.
70    '|.', // Any single character.
71    'g');
72
73
74/**
75 * Regex to check if a string starts with a whitespace character.
76 *
77 * @const
78 * @type {!RegExp}
79 * @private
80 */
81wgxpath.Lexer.LEADING_WHITESPACE_ = /^\s/;
82
83
84/**
85 * Peeks at the lexer. An optional index can be
86 * used to specify the token peek at.
87 *
88 * @param {number=} opt_i Index to peek at. Defaults to zero.
89 * @return {string} Token peeked.
90 */
91wgxpath.Lexer.prototype.peek = function(opt_i) {
92  return this.tokens_[this.index_ + (opt_i || 0)];
93};
94
95
96/**
97 * Returns the next token from the lexer and increments the index.
98 *
99 * @return {string} The next token.
100 */
101wgxpath.Lexer.prototype.next = function() {
102  return this.tokens_[this.index_++];
103};
104
105
106/**
107 * Decrements the index by one.
108 */
109wgxpath.Lexer.prototype.back = function() {
110  this.index_--;
111};
112
113
114/**
115 * Checks whether the lexer is empty.
116 *
117 * @return {boolean} Whether the lexer is empty.
118 */
119wgxpath.Lexer.prototype.empty = function() {
120  return this.tokens_.length <= this.index_;
121};
