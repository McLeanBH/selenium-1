45/**
46 * Apply recursive descent parsing on the input to construct an
47 * abstract syntax tree.
48 *
49 * @return {!wgxpath.Expr} The root of the constructed tree.
50 */
51wgxpath.Parser.prototype.parseExpr = function() {
52  var expr, stack = [];
53  while (true) {
54    this.checkNotEmpty_('Missing right hand side of binary expression.');
55    expr = this.parseUnaryExpr_(); // See if it's just a UnaryExpr.
56    var opString = this.lexer_.next();
57    if (!opString) {
58      break; // Done, we have only a UnaryExpr.
59    }
60
61    var op = wgxpath.BinaryExpr.getOp(opString);
62    var precedence = op && op.getPrecedence();
63    if (!precedence) {
64      this.lexer_.back();
65      break;
66    }
67    // Precedence climbing
68    while (stack.length &&
69        precedence <= stack[stack.length - 1].getPrecedence()) {
70      expr = new wgxpath.BinaryExpr(stack.pop(), stack.pop(), expr);
71    }
72    stack.push(expr, op);
73  }
74  while (stack.length) {
75    expr = new wgxpath.BinaryExpr(stack.pop(), stack.pop(),
76        /** @type {!wgxpath.Expr} */ (expr));
77  }
78  return /** @type {!wgxpath.Expr} */ (expr);
79};

124/**
125 * Attempts to parse the input as a FilterExpr.
126 *
127 * @private
128 * @return {wgxpath.Expr} The root of the constructed tree.
129 */
130wgxpath.Parser.prototype.parseFilterExpr_ = function() {
131  var expr;
132  var token = this.lexer_.peek();
133  var ch = token.charAt(0);
134  switch (ch) {
135    case '$':
136      throw Error('Variable reference not allowed in HTML XPath');
137    case '(':
138      this.lexer_.next();
139      expr = this.parseExpr();
140      this.checkNotEmpty_('unclosed "("');
141      this.checkNextEquals_(')');
142      break;
143    case '"':
144    case "'":
145      expr = this.parseLiteral_();
146      break;
147    default:
148      if (!isNaN(+token)) {
149        expr = this.parseNumber_();
150      } else if (wgxpath.KindTest.isValidType(token)) {
151        return null;
152      } else if (/(?![0-9])[\w]/.test(ch) && this.lexer_.peek(1) == '(') {
153        expr = this.parseFunctionCall_();
154      } else {
155        return null;
156      }
157  }
158  if (this.lexer_.peek() != '[') {
159    return expr;
160  }
161  var predicates = new wgxpath.Predicates(this.parsePredicates_());
162  return new wgxpath.FilterExpr(expr, predicates);
163};

166/**
167 * Parses FunctionCall.
168 *
169 * @private
170 * @return {!wgxpath.FunctionCall} The parsed expression.
171 */
172wgxpath.Parser.prototype.parseFunctionCall_ = function() {
173  var funcName = this.lexer_.next();
174  var func = wgxpath.FunctionCall.getFunc(funcName);
175  this.lexer_.next();
176
177  var args = [];
178  while (this.lexer_.peek() != ')') {
179    this.checkNotEmpty_('Missing function argument list.');
180    args.push(this.parseExpr());
181    if (this.lexer_.peek() != ',') {
182      break;
183    }
184    this.lexer_.next();
185  }
186  this.checkNotEmpty_('Unclosed function argument list.');
187  this.checkNextNotEquals_(')');
188
189  return new wgxpath.FunctionCall(func, args);
190};

193/**
194 * Parses the input to construct a KindTest.
195 *
196 * @private
197 * @return {!wgxpath.KindTest} The KindTest constructed.
198 */
199wgxpath.Parser.prototype.parseKindTest_ = function() {
200  var typeName = this.lexer_.next();
201  if (!wgxpath.KindTest.isValidType(typeName)) {
202    throw Error('Invalid type name: ' + typeName);
203  }
204  this.checkNextEquals_('(');
205  this.checkNotEmpty_('Bad nodetype');
206  var ch = this.lexer_.peek().charAt(0);
207
208  var literal = null;
209  if (ch == '"' || ch == "'") {
210    literal = this.parseLiteral_();
211  }
212  this.checkNotEmpty_('Bad nodetype');
213  this.checkNextNotEquals_(')');
214  return new wgxpath.KindTest(typeName, literal);
215};

233/**
234 * Parses the input to construct a NameTest.
235 *
236 * @private
237 * @return {!wgxpath.NameTest} The NameTest constructed.
238 */
239wgxpath.Parser.prototype.parseNameTest_ = function() {
240  var name = this.lexer_.next();
241
242  // Check whether there's a namespace prefix.
243  var colonIndex = name.indexOf(':');
244  if (colonIndex == -1) {
245    return new wgxpath.NameTest(name);
246  } else {
247    var namespacePrefix = name.substring(0, colonIndex);
248    var namespaceUri = this.nsResolver_(namespacePrefix);
249    if (!namespaceUri) {
250      throw Error('Namespace prefix not declared: ' + namespacePrefix);
251    }
252    name = name.substr(colonIndex + 1);
253    return new wgxpath.NameTest(name, namespaceUri);
254  }
255};

269/**
270 * Attempts to parse the input as a PathExpr.
271 *
272 * @private
273 * @return {!wgxpath.Expr} The root of the constructed tree.
274 */
275wgxpath.Parser.prototype.parsePathExpr_ = function() {
276  var op, expr;
277  var steps = [];
278  var filterExpr;
279  if (wgxpath.PathExpr.isValidOp(this.lexer_.peek())) {
280    op = this.lexer_.next();
281    var token = this.lexer_.peek();
282    if (op == '/' && (this.lexer_.empty() ||
283        (token != '.' && token != '..' && token != '@' && token != '*' &&
284        !/(?![0-9])[\w]/.test(token)))) {
285      return new wgxpath.PathExpr.RootHelperExpr();
286    }
287    filterExpr = new wgxpath.PathExpr.RootHelperExpr();
288
289    this.checkNotEmpty_('Missing next location step.');
290    expr = this.parseStep_(op);
291    steps.push(expr);
292  } else {
293    expr = this.parseFilterExpr_();
294    if (!expr) {
295      expr = this.parseStep_('/');
296      filterExpr = new wgxpath.PathExpr.ContextHelperExpr();
297      steps.push(expr);
298    } else if (!wgxpath.PathExpr.isValidOp(this.lexer_.peek())) {
299      return expr; // Done.
300    } else {
301      filterExpr = expr;
302    }
303  }
304  while (true) {
305    if (!wgxpath.PathExpr.isValidOp(this.lexer_.peek())) {
306      break;
307    }
308    op = this.lexer_.next();
309    this.checkNotEmpty_('Missing next location step.');
310    expr = this.parseStep_(op);
311    steps.push(expr);
312  }
313  return new wgxpath.PathExpr(filterExpr, steps);
314};

317/**
318 * Parses Step.
319 *
320 * @private
321 * @param {string} op The op for this step.
322 * @return {!wgxpath.Step} The parsed expression.
323 */
324wgxpath.Parser.prototype.parseStep_ = function(op) {
325  var test, step, token, predicates;
326  if (op != '/' && op != '//') {
327    throw Error('Step op should be "/" or "//"');
328  }
329  if (this.lexer_.peek() == '.') {
330    step = new wgxpath.Step(wgxpath.Step.Axis.SELF,
331        new wgxpath.KindTest('node'));
332    this.lexer_.next();
333    return step;
334  }
335  else if (this.lexer_.peek() == '..') {
336    step = new wgxpath.Step(wgxpath.Step.Axis.PARENT,
337        new wgxpath.KindTest('node'));
338    this.lexer_.next();
339    return step;
340  } else {
341    // Grab the axis.
342    var axis;
343    if (this.lexer_.peek() == '@') {
344      axis = wgxpath.Step.Axis.ATTRIBUTE;
345      this.lexer_.next();
346      this.checkNotEmpty_('Missing attribute name');
347    } else {
348      if (this.lexer_.peek(1) == '::') {
349        if (!/(?![0-9])[\w]/.test(this.lexer_.peek().charAt(0))) {
350          throw Error('Bad token: ' + this.lexer_.next());
351        }
352        var axisName = this.lexer_.next();
353        axis = wgxpath.Step.getAxis(axisName);
354        if (!axis) {
355          throw Error('No axis with name: ' + axisName);
356        }
357        this.lexer_.next();
358        this.checkNotEmpty_('Missing node name');
359      } else {
360        axis = wgxpath.Step.Axis.CHILD;
361      }
362    }
363
364    // Grab the test.
365    token = this.lexer_.peek();
366    if (!/(?![0-9])[\w]/.test(token.charAt(0))) {
367      if (token == '*') {
368        test = this.parseNameTest_();
369      } else {
370        throw Error('Bad token: ' + this.lexer_.next());
371      }
372    } else {
373      if (this.lexer_.peek(1) == '(') {
374        if (!wgxpath.KindTest.isValidType(token)) {
375          throw Error('Invalid node type: ' + token);
376        }
377        test = this.parseKindTest_();
378      } else {
379        test = this.parseNameTest_();
380      }
381    }
382    predicates = new wgxpath.Predicates(this.parsePredicates_(),
383        axis.isReverse());
384    return step || new wgxpath.Step(axis, test, predicates, op == '//');
385  }
386};

389/**
390 * Parses and returns the predicates from the this.lexer_.
391 *
392 * @private
393 * @return {!Array.<!wgxpath.Expr>} An array of the predicates.
394 */
395wgxpath.Parser.prototype.parsePredicates_ = function() {
396  var predicates = [];
397  while (this.lexer_.peek() == '[') {
398    this.lexer_.next();
399    this.checkNotEmpty_('Missing predicate expression.');
400    var predicate = this.parseExpr();
401    predicates.push(predicate);
402    this.checkNotEmpty_('Unclosed predicate expression.');
403    this.checkNextEquals_(']');
404  }
405  return predicates;
406};

409/**
410 * Attempts to parse the input as a unary expression with
411 * recursive descent parsing.
412 *
413 * @private
414 * @return {!wgxpath.Expr} The root of the constructed tree.
415 */
416wgxpath.Parser.prototype.parseUnaryExpr_ = function() {
417  if (this.lexer_.peek() == '-') {
418    this.lexer_.next();
419    return new wgxpath.UnaryExpr(this.parseUnaryExpr_());
420  } else {
421    return this.parseUnionExpr_();
422  }
423};

426/**
427 * Attempts to parse the input as a union expression with
428 * recursive descent parsing.
429 *
430 * @private
431 * @return {!wgxpath.Expr} The root of the constructed tree.
432 */
433wgxpath.Parser.prototype.parseUnionExpr_ = function() {
434  var expr = this.parsePathExpr_();
435  if (!(this.lexer_.peek() == '|')) {
436    return expr;  // Not a UnionExpr, returning as is.
437  }
438  var paths = [expr];
439  while (this.lexer_.next() == '|') {
440    this.checkNotEmpty_('Missing next union location path.');
441    paths.push(this.parsePathExpr_());
442  }
443  this.lexer_.back();
444  return new wgxpath.UnionExpr(paths);
