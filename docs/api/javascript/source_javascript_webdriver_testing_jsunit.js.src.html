60/** @override */
61webdriver.testing.jsunit.TestRunner.prototype.initialize = function(testCase) {
62  goog.base(this, 'initialize', testCase);
63  this.screenshotCacheEl_ = document.createElement('div');
64  document.body.appendChild(this.screenshotCacheEl_);
65  this.screenshotCacheEl_.style.display = 'none';
66};

69/** @override */
70webdriver.testing.jsunit.TestRunner.prototype.execute = function() {
71  if (!this.testCase) {
72    throw Error('The test runner must be initialized with a test case before ' +
73                'execute can be called.');
74  }
75  this.screenshotCacheEl_.innerHTML = '';
76  this.client_.sendInitEvent();
77  this.testCase.setCompletedCallback(goog.bind(this.onComplete_, this));
78  this.testCase.runTests();
79};

82/**
83 * Writes a nicely formatted log out to the document. Overrides
84 * {@link goog.testing.TestRunner#writeLog} to handle writing screenshots to the
85 *     log.
86 * @param {string} log The string to write.
87 * @override
88 */
89webdriver.testing.jsunit.TestRunner.prototype.writeLog = function(log) {
90  var lines = log.split('\n');
91  for (var i = 0; i < lines.length; i++) {
92    var line = lines[i];
93    var color;
94    var isFailOrError = /FAILED/.test(line) || /ERROR/.test(line);
95    var isScreenshot = / \[SCREENSHOT\] /.test(line);
96    if (/PASSED/.test(line)) {
97      color = 'darkgreen';
98    } else if (isFailOrError) {
99      color = 'darkred';
100    } else if (isScreenshot) {
101      color = 'darkblue';
102    } else {
103      color = '#333';
104    }
105
106    var div = document.createElement('div');
107    if (line.substr(0, 2) == '> ') {
108      // The stack trace may contain links so it has to be interpreted as HTML.
109      div.innerHTML = line;
110    } else {
111      div.appendChild(document.createTextNode(line));
112    }
113
114    if (isFailOrError) {
115      var testNameMatch = /(\S+) (\[[^\]]*] )?: (FAILED|ERROR)/.exec(line);
116      if (testNameMatch) {
117        // Build a URL to run the test individually.  If this test was already
118        // part of another subset test, we need to overwrite the old runTests
119        // query parameter.  We also need to do this without bringing in any
120        // extra dependencies, otherwise we could mask missing dependency bugs.
121        var newSearch = 'runTests=' + testNameMatch[1];
122        var search = window.location.search;
123        if (search) {
124          var oldTests = /runTests=([^&]*)/.exec(search);
125          if (oldTests) {
126            newSearch = search.substr(0, oldTests.index) +
127                        newSearch +
128                        search.substr(oldTests.index + oldTests[0].length);
129          } else {
130            newSearch = search + '&' + newSearch;
131          }
132        } else {
133          newSearch = '?' + newSearch;
134        }
135        var href = window.location.href;
136        var hash = window.location.hash;
137        if (hash && hash.charAt(0) != '#') {
138          hash = '#' + hash;
139        }
140        href = href.split('#')[0].split('?')[0] + newSearch + hash;
141
142        // Add the link.
143        var a = document.createElement('A');
144        a.innerHTML = '(run individually)';
145        a.style.fontSize = '0.8em';
146        a.href = href;
147        div.appendChild(document.createTextNode(' '));
148        div.appendChild(a);
149      }
150    }
151
152    if (isScreenshot && this.screenshotCacheEl_.childNodes.length) {
153      var nextScreenshot = this.screenshotCacheEl_.childNodes[0];
154      this.screenshotCacheEl_.removeChild(nextScreenshot);
155
156      a = document.createElement('A');
157      a.style.fontSize = '0.8em';
158      a.href = 'javascript:void(0);';
159      a.onclick = goog.partial(toggleVisibility, a, nextScreenshot);
160      toggleVisibility(a, nextScreenshot);
161      div.appendChild(document.createTextNode(' '));
162      div.appendChild(a);
163    }
164
165    div.style.color = color;
166    div.style.font = 'normal 100% monospace';
167
168    try {
169      div.style.whiteSpace = 'pre-wrap';
170    } catch (e) {
171      // NOTE(user): IE raises an exception when assigning to pre-wrap.
172      // Thankfully, it doesn't collapse whitespace when using monospace fonts,
173      // so it will display correctly if we ignore the exception.
174    }
175
176    if (i < 2) {
177      div.style.fontWeight = 'bold';
178    }
179    this.logEl_.appendChild(div);
180
181    if (nextScreenshot) {
182      a = document.createElement('A');
183      // Accessing the |src| property in IE sometimes results in an
184      // "Invalid pointer" error, which indicates it has been garbage
185      // collected. This does not occur when using getAttribute.
186      a.href = nextScreenshot.getAttribute('src');
187      a.target = '_blank';
188      a.appendChild(nextScreenshot);
189      this.logEl_.appendChild(a);
190      nextScreenshot = null;
191    }
192  }
193
194  function toggleVisibility(link, img) {
195    if (img.style.display === 'none') {
196      img.style.display = '';
197      link.innerHTML = '(hide screenshot)';
198    } else {
199      img.style.display = 'none';
200      link.innerHTML = '(view screenshot)';
201    }
202  }
203};

206/**
207 * Copied from goog.testing.TestRunner.prototype.onComplete_, which has private
208 * visibility.
209 * @private
210 */
211webdriver.testing.jsunit.TestRunner.prototype.onComplete_ = function() {
212  var log = this.testCase.getReport(true);
213  if (this.errors.length > 0) {
214    log += '\n' + this.errors.join('\n');
215  }
216
217  if (!this.logEl_) {
218    this.logEl_ = document.createElement('div');
219    document.body.appendChild(this.logEl_);
220  }
221
222  // Remove all children from the log element.
223  var logEl = this.logEl_;
224  while (logEl.firstChild) {
225    logEl.removeChild(logEl.firstChild);
226  }
227
228  this.writeLog(log);
229  this.client_.sendResultsEvent(this.isSuccess(), this.getReport(true));
230};

233/**
234 * Takes a screenshot. In addition to saving the screenshot for viewing in the
235 * HTML logs, the screenshot will also be saved using
236 * @param {!webdriver.WebDriver} driver The driver to take the screenshot with.
237 * @param {string=} opt_label An optional debug label to identify the screenshot
238 *     with.
239 * @return {!webdriver.promise.Promise} A promise that will be resolved to the
240 *     screenshot as a base-64 encoded PNG.
241 */
242webdriver.testing.jsunit.TestRunner.prototype.takeScreenshot = function(
243    driver, opt_label) {
244  if (!this.isInitialized()) {
245    throw Error(
246        'The test runner must be initialized before it may be used to' +
247        ' take screenshots');
248  }
249
250  var client = this.client_;
251  var testCase = this.testCase;
252  var screenshotCache = this.screenshotCacheEl_;
253  return driver.takeScreenshot().then(function(png) {
254    client.sendScreenshotEvent(png, opt_label);
255
256    var img = document.createElement('img');
257    img.src = 'data:image/png;base64,' + png;
258    img.style.border = '1px solid black';
259    img.style.maxWidth = '500px';
260    screenshotCache.appendChild(img);
261
262    if (testCase) {
263      testCase.saveMessage('[SCREENSHOT] ' + (opt_label || '<Not Labeled>'));
264    }
265    return png;
266  });
267};

270(function() {
271  var client = new webdriver.testing.Client();
272  var tr = new webdriver.testing.jsunit.TestRunner(client);
273
274  // Export our test runner so it can be accessed by Selenium/WebDriver. This
275  // will only work if webdriver.WebDriver is using a pure-JavaScript
276  // webdriver.CommandExecutor. Otherwise, the JS-client could change the
277  // driver's focus to another window or frame and the Java/Python-client
278  // wouldn't be able to access this object.
279  goog.exportSymbol('G_testRunner', tr);
280  goog.exportSymbol('G_testRunner.initialize', tr.initialize);
281  goog.exportSymbol('G_testRunner.isInitialized', tr.isInitialized);
282  goog.exportSymbol('G_testRunner.isFinished', tr.isFinished);
283  goog.exportSymbol('G_testRunner.isSuccess', tr.isSuccess);
284  goog.exportSymbol('G_testRunner.getReport', tr.getReport);
285  goog.exportSymbol('G_testRunner.getRunTime', tr.getRunTime);
286  goog.exportSymbol('G_testRunner.getNumFilesLoaded', tr.getNumFilesLoaded);
287  goog.exportSymbol('G_testRunner.setStrict', tr.setStrict);
288  goog.exportSymbol('G_testRunner.logTestFailure', tr.logTestFailure);
289
290  // Export debug as a global function for JSUnit compatibility.  This just
291  // calls log on the current test case.
292  if (!goog.global['debug']) {
293    goog.exportSymbol('debug', goog.bind(tr.log, tr));
294  }
295
296  // Add an error handler to report errors that may occur during
297  // initialization of the page.
298  var onerror = window.onerror;
299  window.onerror = function(error, url, line) {
300    // Call any existing onerror handlers.
301    if (onerror) {
302      onerror(error, url, line);
303    }
304    if (typeof error == 'object') {
305      // Webkit started passing an event object as the only argument to
306      // window.onerror.  It doesn't contain an error message, url or line
307      // number.  We therefore log as much info as we can.
308      if (error.target && error.target.tagName == 'SCRIPT') {
309        tr.logError('UNKNOWN ERROR: Script ' + error.target.src);
310      } else {
311        tr.logError('UNKNOWN ERROR: No error information available.');
312      }
313    } else {
314      tr.logError('JS ERROR: ' + error + '\nURL: ' + url + '\nLine: ' + line);
315    }
316  };
317
318  var onload = window.onload;
319  window.onload = function() {
320    // Call any existing onload handlers.
321    if (onload) {
322      onload();
323    }
324
325    var testCase = new webdriver.testing.TestCase(client, document.title);
326    testCase.autoDiscoverTests();
327
328    tr.initialize(testCase);
329    tr.execute();
330  };
331})();
