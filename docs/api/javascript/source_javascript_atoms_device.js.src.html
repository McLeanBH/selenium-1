36/**
37 * A Device class that provides common functionality for input devices.
38 * @param {bot.Device.ModifiersState=} opt_modifiersState state of modifier
39 * keys. The state is shared, not copied from this parameter.
40 * @param {bot.Device.EventEmitter=} opt_eventEmitter An object that should be
41 *     used to fire events.
42 * @constructor
43 */
44bot.Device = function(opt_modifiersState, opt_eventEmitter) {
45  /**
46   * Element being interacted with.
47   * @private {!Element}
48   */
49  this.element_ = bot.getDocument().documentElement;
50
51  /**
52   * If the element is an option, this is its parent select element.
53   * @private {Element}
54   */
55  this.select_ = null;
56
57  // If there is an active element, make that the current element instead.
58  var activeElement = bot.dom.getActiveElement(this.element_);
59  if (activeElement) {
60    this.setElement(activeElement);
61  }
62
63  /**
64   * State of modifier keys for this device.
65   * @protected {bot.Device.ModifiersState}
66   */
67  this.modifiersState = opt_modifiersState || new bot.Device.ModifiersState();
68
69  /** @protected {!bot.Device.EventEmitter} */
70  this.eventEmitter = opt_eventEmitter || new bot.Device.EventEmitter();
71};

85/**
86 * Sets the element with which the device is interacting.
87 *
88 * @param {!Element} element Element being interacted with.
89 * @protected
90 */
91bot.Device.prototype.setElement = function(element) {
92  this.element_ = element;
93  if (bot.dom.isElement(element, goog.dom.TagName.OPTION)) {
94    this.select_ = /** @type {Element} */ (goog.dom.getAncestor(element,
95        function(node) {
96          return bot.dom.isElement(node, goog.dom.TagName.SELECT);
97        }));
98  } else {
99    this.select_ = null;
100  }
101};

130/**
131 * Fires a mouse event given the state of the device and the given arguments.
132 * TODO: Populate the modifier keys in this method.
133 *
134 * @param {bot.events.EventType} type Mouse event type.
135 * @param {!goog.math.Coordinate} coord The coordinate where event will fire.
136 * @param {number} button The mouse button value for the event.
137 * @param {Element=} opt_related The related element of this event.
138 * @param {?number=} opt_wheelDelta The wheel delta value for the event.
139 * @param {boolean=} opt_force Whether the event should be fired even if the
140 *     element is not interactable, such as the case of a mousemove or
141 *     mouseover event that immediately follows a mouseout.
142 * @param {?number=} opt_pointerId The pointerId associated with the event.
143 * @return {boolean} Whether the event fired successfully; false if cancelled.
144 * @protected
145 */
146bot.Device.prototype.fireMouseEvent = function(type, coord, button,
147    opt_related, opt_wheelDelta, opt_force, opt_pointerId)  {
148  if (!opt_force && !bot.dom.isInteractable(this.element_)) {
149    return false;
150  }
151
152  if (opt_related &&
153      !(bot.events.EventType.MOUSEOVER == type ||
154        bot.events.EventType.MOUSEOUT == type)) {
155    throw new bot.Error(bot.ErrorCode.INVALID_ELEMENT_STATE,
156                        'Event type does not allow related target: ' + type);
157  }
158
159  var args = {
160    clientX: coord.x,
161    clientY: coord.y,
162    button: button,
163    altKey: this.modifiersState.isAltPressed(),
164    ctrlKey: this.modifiersState.isControlPressed(),
165    shiftKey: this.modifiersState.isShiftPressed(),
166    metaKey: this.modifiersState.isMetaPressed(),
167    wheelDelta: opt_wheelDelta || 0,
168    relatedTarget: opt_related || null
169  };
170
171  var pointerId = opt_pointerId || bot.Device.MOUSE_MS_POINTER_ID;
172
173  var target = this.element_;
174  // On click and mousedown events, captured pointers are ignored and the
175  // event always fires on the original element.
176  if (type != bot.events.EventType.CLICK &&
177      type != bot.events.EventType.MOUSEDOWN &&
178      pointerId in bot.Device.pointerElementMap_) {
179    target = bot.Device.pointerElementMap_[pointerId];
180  } else if (this.select_) {
181    target = this.getTargetOfOptionMouseEvent_(type);
182  }
183  return target ? this.eventEmitter.fireMouseEvent(target, type, args) : true;
184};

187/**
188 * Fires a touch event given the state of the deive and the given arguments.
189 *
190 * @param {bot.events.EventType} type Event type.
191 * @param {number} id The touch identifier.
192 * @param {!goog.math.Coordinate} coord The coordinate where event will fire.
193 * @param {number=} opt_id2 The touch identifier of the second finger.
194 * @param {!goog.math.Coordinate=} opt_coord2 The coordinate of the second
195 *    finger, if any.
196 * @return {boolean} Whether the event fired successfully or was cancelled.
197 * @protected
198 */
199bot.Device.prototype.fireTouchEvent = function(type, id, coord, opt_id2,
200                                               opt_coord2) {
201  var args = {
202    touches: [],
203    targetTouches: [],
204    changedTouches: [],
205    altKey: this.modifiersState.isAltPressed(),
206    ctrlKey: this.modifiersState.isControlPressed(),
207    shiftKey: this.modifiersState.isShiftPressed(),
208    metaKey: this.modifiersState.isMetaPressed(),
209    relatedTarget: null,
210    scale: 0,
211    rotation: 0
212  };
213
214  function addTouch(identifier, coords) {
215    // Android devices leave identifier to zero.
216    var id = goog.userAgent.product.ANDROID ? 0 : identifier;
217    var touch = {
218      identifier: identifier,
219      screenX: coords.x,
220      screenY: coords.y,
221      clientX: coords.x,
222      clientY: coords.y,
223      pageX: coords.x,
224      pageY: coords.y
225    };
226
227    args.changedTouches.push(touch);
228    if (type == bot.events.EventType.TOUCHSTART ||
229        type == bot.events.EventType.TOUCHMOVE) {
230      args.touches.push(touch);
231      args.targetTouches.push(touch);
232    }
233  }
234
235  addTouch(id, coord);
236  if (goog.isDef(opt_id2)) {
237    addTouch(opt_id2, opt_coord2);
238  }
239
240  return this.eventEmitter.fireTouchEvent(this.element_, type, args);
241};

244/**
245 * Fires a MSPointer event given the state of the device and the given
246 * arguments.
247 *
248 * @param {bot.events.EventType} type MSPointer event type.
249 * @param {!goog.math.Coordinate} coord The coordinate where event will fire.
250 * @param {number} button The mouse button value for the event.
251 * @param {number} pointerId The pointer id for this event.
252 * @param {number} device The device type used for this event.
253 * @param {boolean} isPrimary Whether the pointer represents the primary point
254 *     of contact.
255 * @param {Element=} opt_related The related element of this event.
256 * @param {boolean=} opt_force Whether the event should be fired even if the
257 *     element is not interactable, such as the case of a mousemove or
258 *     mouseover event that immediately follows a mouseout.
259 * @return {boolean} Whether the event fired successfully; false if cancelled.
260 * @protected
261 */
262bot.Device.prototype.fireMSPointerEvent = function(type, coord, button,
263    pointerId, device, isPrimary, opt_related, opt_force) {
264  if (!opt_force && !bot.dom.isInteractable(this.element_)) {
265    return false;
266  }
267
268  if (opt_related &&
269      !(bot.events.EventType.MSPOINTEROVER == type ||
270        bot.events.EventType.MSPOINTEROUT == type)) {
271    throw new bot.Error(bot.ErrorCode.INVALID_ELEMENT_STATE,
272                        'Event type does not allow related target: ' + type);
273  }
274
275  var args = {
276    clientX: coord.x,
277    clientY: coord.y,
278    button: button,
279    altKey: false,
280    ctrlKey: false,
281    shiftKey: false,
282    metaKey: false,
283    relatedTarget: opt_related || null,
284    width: 0,
285    height: 0,
286    pressure: 0, // Pressure is only given when a stylus is used.
287    rotation: 0,
288    pointerId: pointerId,
289    tiltX: 0,
290    tiltY: 0,
291    pointerType: device,
292    isPrimary: isPrimary
293  };
294
295  var target = this.select_ ?
296      this.getTargetOfOptionMouseEvent_(type) : this.element_;
297  if (bot.Device.pointerElementMap_[pointerId]) {
298    target = bot.Device.pointerElementMap_[pointerId];
299  }
300  var owner = goog.dom.getWindow(goog.dom.getOwnerDocument(this.element_));
301  var originalMsSetPointerCapture;
302  if (owner && type == bot.events.EventType.MSPOINTERDOWN) {
303    // Overwrite msSetPointerCapture on the Element's msSetPointerCapture
304    // because synthetic pointer events cause an access denied exception.
305    // The prototype is modified because the pointer event will bubble up and
306    // we do not know which element will handle the pointer event.
307    originalMsSetPointerCapture =
308        owner['Element'].prototype.msSetPointerCapture;
309    owner['Element'].prototype.msSetPointerCapture = function(id) {
310      bot.Device.pointerElementMap_[id] = this;
311    };
312  }
313  var result =
314      target ? this.eventEmitter.fireMSPointerEvent(target, type, args) : true;
315  if (originalMsSetPointerCapture) {
316    owner['Element'].prototype.msSetPointerCapture =
317        originalMsSetPointerCapture;
318  }
319  return result;
320};

323/**
324 * A mouse event fired "on" an option element, doesn't always fire on the
325 * option element itself. Sometimes it fires on the parent select element
326 * and sometimes not at all, depending on the browser and event type. This
327 * returns the true target element of the event, or null if none is fired.
328 *
329 * @param {bot.events.EventType} type Type of event.
330 * @return {Element} Element the event should be fired on, null if none.
331 * @private
332 */
333bot.Device.prototype.getTargetOfOptionMouseEvent_ = function(type) {
334  // IE either fires the event on the parent select or not at all.
335  if (goog.userAgent.IE) {
336    switch (type) {
337      case bot.events.EventType.MOUSEOVER:
338      case bot.events.EventType.MSPOINTEROVER:
339        return null;
340      case bot.events.EventType.CONTEXTMENU:
341      case bot.events.EventType.MOUSEMOVE:
342      case bot.events.EventType.MSPOINTERMOVE:
343        return this.select_.multiple ? this.select_ : null;
344      default:
345        return this.select_;
346    }
347  }
348
349  // Opera only skips mouseovers and contextmenus on single selects.
350  if (goog.userAgent.OPERA) {
351    switch (type) {
352      case bot.events.EventType.CONTEXTMENU:
353      case bot.events.EventType.MOUSEOVER:
354        return this.select_.multiple ? this.element_ : null;
355      default:
356        return this.element_;
357    }
358  }
359
360  // WebKit always fires on the option element of multi-selects.
361  // On single-selects, it either fires on the parent or not at all.
362  if (goog.userAgent.WEBKIT) {
363    switch (type) {
364      case bot.events.EventType.CLICK:
365      case bot.events.EventType.MOUSEUP:
366        return this.select_.multiple ? this.element_ : this.select_;
367      default:
368        return this.select_.multiple ? this.element_ : null;
369    }
370  }
371
372  // Firefox fires every event or the option element.
373  return this.element_;
374};

377/**
378 * A helper function to fire click events.  This method is shared between
379 * the mouse and touchscreen devices.
380 *
381 * @param {!goog.math.Coordinate} coord The coordinate where event will fire.
382 * @param {number} button The mouse button value for the event.
383 * @param {?number=} opt_pointerId The pointer id associated with the click.
384 * @protected
385 */
386bot.Device.prototype.clickElement = function(coord, button, opt_pointerId) {
387  if (!bot.dom.isInteractable(this.element_)) {
388    return;
389  }
390
391  // bot.events.fire(element, 'click') can trigger all onclick events, but may
392  // not follow links (FORM.action or A.href).
393  //     TAG      IE   GECKO  WebKit Opera
394  // A(href)      No    No     Yes    Yes
395  // FORM(action) No    Yes    Yes    Yes
396  var targetLink = null;
397  var targetButton = null;
398  if (!bot.Device.ALWAYS_FOLLOWS_LINKS_ON_CLICK_) {
399    for (var e = this.element_; e; e = e.parentNode) {
400      if (bot.dom.isElement(e, goog.dom.TagName.A)) {
401        targetLink = /**@type {!Element}*/ (e);
402        break;
403      } else if (bot.Device.isFormSubmitElement(e)) {
404        targetButton = e;
405        break;
406      }
407    }
408  }
409
410  // When an element is toggled as the result of a click, the toggling and the
411  // change event happens before the click event on some browsers. However, on
412  // radio buttons and checkboxes, the click handler can prevent the toggle from
413  // happening, so we must fire the click first to see if it is cancelled.
414  var isRadioOrCheckbox = !this.select_ && bot.dom.isSelectable(this.element_);
415  var wasChecked = isRadioOrCheckbox && bot.dom.isSelected(this.element_);
416
417  // NOTE: Clicking on a form submit button is a little broken:
418  // (1) When clicking a form submit button in IE, firing a click event or
419  // calling Form.submit() will not by itself submit the form, so we call
420  // Element.click() explicitly, but as a result, the coordinates of the click
421  // event are not provided. Also, when clicking on an <input type=image>, the
422  // coordinates click that are submitted with the form are always (0, 0).
423  // (2) When clicking a form submit button in GECKO, while the coordinates of
424  // the click event are correct, those submitted with the form are always (0,0)
425  // .
426  // TODO: See if either of these can be resolved, perhaps by adding
427  // hidden form elements with the coordinates before the form is submitted.
428  if (goog.userAgent.IE && targetButton) {
429    targetButton.click();
430    return;
431  }
432
433  var performDefault = this.fireMouseEvent(
434      bot.events.EventType.CLICK, coord, button, null, 0, false, opt_pointerId);
435  if (!performDefault) {
436    return;
437  }
438
439  if (targetLink && bot.Device.shouldFollowHref_(targetLink)) {
440    bot.Device.followHref_(targetLink);
441  } else if (isRadioOrCheckbox) {
442    this.toggleRadioButtonOrCheckbox_(wasChecked);
443  }
444};

447/**
448 * Focuses on the given element and returns true if it supports being focused
449 * and does not already have focus; otherwise, returns false. If another element
450 * has focus, that element will be blurred before focusing on the given element.
451 *
452 * @return {boolean} Whether the element was given focus.
453 * @protected
454 */
455bot.Device.prototype.focusOnElement = function() {
456  // Focusing on an <option> always focuses on the parent <select>.
457  var elementToFocus = this.select_ || this.element_;
458
459  var activeElement = bot.dom.getActiveElement(elementToFocus);
460  if (elementToFocus == activeElement) {
461    return false;
462  }
463
464  // If there is a currently active element, try to blur it.
465  if (activeElement && (goog.isFunction(activeElement.blur) ||
466      // IE reports native functions as being objects.
467      goog.userAgent.IE && goog.isObject(activeElement.blur))) {
468    // In IE, the focus() and blur() functions fire their respective events
469    // asynchronously, and as the result, the focus/blur events fired by the
470    // the atoms actions will often be in the wrong order on IE. Firing a blur
471    // out of order sometimes causes IE to throw an "Unspecified error", so we
472    // wrap it in a try-catch and catch and ignore the error in this case.
473    if (!bot.dom.isElement(activeElement, goog.dom.TagName.BODY)) {
474      try {
475        activeElement.blur();
476      } catch (e) {
477        if (!(goog.userAgent.IE && e.message == 'Unspecified error.')) {
478          throw e;
479        }
480      }
481    }
482
483    // Sometimes IE6 and IE7 will not fire an onblur event after blur()
484    // is called, unless window.focus() is called immediately afterward.
485    // Note that IE8 will hit this branch unless the page is forced into
486    // IE8-strict mode. This shouldn't hurt anything, we just use the
487    // useragent sniff so we can compile this out for proper browsers.
488    if (goog.userAgent.IE && !bot.userAgent.isEngineVersion(8)) {
489      goog.dom.getWindow(goog.dom.getOwnerDocument(elementToFocus)).focus();
490    }
491  }
492
493  // Try to focus on the element.
494  if (goog.isFunction(elementToFocus.focus) ||
495      goog.userAgent.IE && goog.isObject(elementToFocus.focus)) {
496    // Opera fires focus events on hidden elements (e.g. that are hidden after
497    // mousedown in a click sequence), but as of Opera 11 the focus() command
498    // does not, so we fire a focus event on the hidden element explicitly.
499    if (goog.userAgent.OPERA && bot.userAgent.isEngineVersion(11) &&
500        !bot.dom.isShown(elementToFocus)) {
501      bot.events.fire(elementToFocus, bot.events.EventType.FOCUS);
502    } else {
503      elementToFocus.focus();
504    }
505    return true;
506  }
507
508  return false;
509};

512/**
513 * Whether links must be manually followed when clicking (because firing click
514 * events doesn't follow them).
515 * @private {boolean}
516 * @const
517 */
518bot.Device.ALWAYS_FOLLOWS_LINKS_ON_CLICK_ =
519    goog.userAgent.WEBKIT || goog.userAgent.OPERA ||
520    (bot.userAgent.FIREFOX_EXTENSION && bot.userAgent.isProductVersion(3.6));
521

523/**
524 * @param {Node} element The element to check.
525 * @return {boolean} Whether the element is a submit element in form.
526 * @protected
527 */
528bot.Device.isFormSubmitElement = function(element) {
529  if (bot.dom.isElement(element, goog.dom.TagName.INPUT)) {
530    var type = element.type.toLowerCase();
531    if (type == 'submit' || type == 'image') {
532      return true;
533    }
534  }
535
536  if (bot.dom.isElement(element, goog.dom.TagName.BUTTON)) {
537    var type = element.type.toLowerCase();
538    if (type == 'submit') {
539      return true;
540    }
541  }
542  return false;
543};

546/**
547 * Indicates whether we should manually follow the href of the element we're
548 * clicking.
549 *
550 * Versions of firefox from 4+ will handle links properly when this is used in
551 * an extension. Versions of Firefox prior to this may or may not do the right
552 * thing depending on whether a target window is opened and whether the click
553 * has caused a change in just the hash part of the URL.
554 *
555 * @param {!Element} element The element to consider.
556 * @return {boolean} Whether following an href should be skipped.
557 * @private
558 */
559bot.Device.shouldFollowHref_ = function(element) {
560  if (bot.Device.ALWAYS_FOLLOWS_LINKS_ON_CLICK_ || !element.href) {
561    return false;
562  }
563
564  if (!bot.userAgent.FIREFOX_EXTENSION) {
565    return true;
566  }
567
568  if (element.target || element.href.toLowerCase().indexOf('javascript') == 0) {
569    return false;
570  }
571
572  var owner = goog.dom.getWindow(goog.dom.getOwnerDocument(element));
573  var sourceUrl = owner.location.href;
574  var destinationUrl = bot.Device.resolveUrl_(owner.location, element.href);
575  var isOnlyHashChange =
576      sourceUrl.split('#')[0] === destinationUrl.split('#')[0];
577
578  return !isOnlyHashChange;
579};

582/**
583 * Explicitly follows the href of an anchor.
584 *
585 * @param {!Element} anchorElement An anchor element.
586 * @private
587 */
588bot.Device.followHref_ = function(anchorElement) {
589  var targetHref = anchorElement.href;
590  var owner = goog.dom.getWindow(goog.dom.getOwnerDocument(anchorElement));
591
592  // IE7 and earlier incorrect resolve a relative href against the top window
593  // location instead of the window to which the href is assigned. As a result,
594  // we have to resolve the relative URL ourselves. We do not use Closure's
595  // goog.Uri to resolve, because it incorrectly fails to support empty but
596  // undefined query and fragment components and re-encodes the given url.
597  if (goog.userAgent.IE && !bot.userAgent.isEngineVersion(8)) {
598    targetHref = bot.Device.resolveUrl_(owner.location, targetHref);
599  }
600
601  if (anchorElement.target) {
602    owner.open(targetHref, anchorElement.target);
603  } else {
604    owner.location.href = targetHref;
605  }
606};

609/**
610 * Toggles the selected state of the current element if it is an option. This
611 * is a noop if the element is not an option, or if it is selected and belongs
612 * to a single-select, because it can't be toggled off.
613 *
614 * @protected
615 */
616bot.Device.prototype.maybeToggleOption = function() {
617  // If this is not an <option> or not interactable, exit.
618  if (!this.select_ || !bot.dom.isInteractable(this.element_)) {
619    return;
620  }
621  var select = /** @type {!Element} */ (this.select_);
622  var wasSelected = bot.dom.isSelected(this.element_);
623  // Cannot toggle off options in single-selects.
624  if (wasSelected && !select.multiple) {
625    return;
626  }
627
628  // TODO: in a multiselect, clicking an option without the shift key down
629  // should deselect all other selected options.
630
631  this.element_.selected = !wasSelected;
632  // Only WebKit fires the change event itself and only for multi-selects,
633  // except for Android versions >= 4.0 and Chrome >= 28.
634  if (!(goog.userAgent.WEBKIT && select.multiple) ||
635      (goog.userAgent.product.CHROME && bot.userAgent.isProductVersion(28)) ||
636      (goog.userAgent.product.ANDROID && bot.userAgent.isProductVersion(4))) {
637    bot.events.fire(select, bot.events.EventType.CHANGE);
638  }
639};

642/**
643 * Toggles the checked state of a radio button or checkbox. This is a noop if
644 * it is a radio button that is checked, because it can't be toggled off.
645 *
646 * @param {boolean} wasChecked Whether the element was originally checked.
647 * @private
648 */
649bot.Device.prototype.toggleRadioButtonOrCheckbox_ = function(wasChecked) {
650  // Gecko and WebKit toggle the element as a result of a click.
651  if (goog.userAgent.GECKO || goog.userAgent.WEBKIT) {
652    return;
653  }
654  // Cannot toggle off radio buttons.
655  if (wasChecked && this.element_.type.toLowerCase() == 'radio') {
656    return;
657  }
658  this.element_.checked = !wasChecked;
659  // Only Opera versions < 11 do not fire the change event themselves.
660  if (goog.userAgent.OPERA && !bot.userAgent.isEngineVersion(11)) {
661    bot.events.fire(this.element_, bot.events.EventType.CHANGE);

688/**
689 * Submits the specified form. Unlike the public function, it expects to be
690 * given a form element and fails if it is not.
691 * @param {!Element} form The form to submit.
692 * @protected
693 */
694bot.Device.prototype.submitForm = function(form) {
695  if (!bot.Device.isForm_(form)) {
696    throw new bot.Error(bot.ErrorCode.INVALID_ELEMENT_STATE,
697                        'Element is not a form, so could not submit.');
698  }
699  if (bot.events.fire(form, bot.events.EventType.SUBMIT)) {
700    // When a form has an element with an id or name exactly equal to "submit"
701    // (not uncommon) it masks the form.submit function. We  can avoid this by
702    // calling the prototype's submit function, except in IE < 8, where DOM id
703    // elements don't let you reference their prototypes. For IE < 8, can change
704    // the id and names of the elements and revert them back, but they must be
705    // reverted before the submit call, because the onsubmit handler might rely
706    // on their being correct, and the HTTP request might otherwise be left with
707    // incorrect value names. Fortunately, saving the submit function and
708    // calling it after reverting the ids and names works! Oh, and goog.typeOf
709    // (and thus goog.isFunction) doesn't work for form.submit in IE < 8.
710    if (!bot.dom.isElement(form.submit)) {
711      form.submit();
712    } else if (!goog.userAgent.IE || bot.userAgent.isEngineVersion(8)) {
713      /** @type {Function} */ (form.constructor.prototype['submit']).call(form);
714    } else {
715      var idMasks = bot.locators.findElements({'id': 'submit'}, form);
716      var nameMasks = bot.locators.findElements({'name': 'submit'}, form);
717      goog.array.forEach(idMasks, function(m) {
718        m.removeAttribute('id');
719      });
720      goog.array.forEach(nameMasks, function(m) {
721        m.removeAttribute('name');
722      });
723      var submitFunction = form.submit;
724      goog.array.forEach(idMasks, function(m) {
725        m.setAttribute('id', 'submit');
726      });
727      goog.array.forEach(nameMasks, function(m) {
728        m.setAttribute('name', 'submit');
729      });
730      submitFunction();
731    }
732  }
733};

736/**
737 * Regular expression for splitting up a URL into components.
738 * @private {!RegExp}
739 * @const
740 */
741bot.Device.URL_REGEXP_ = new RegExp(
742    '^' +
743    '([^:/?#.]+:)?' +   // protocol
744    '(?://([^/]*))?' +  // host
745    '([^?#]+)?' +       // pathname
746    '(\\?[^#]*)?' +     // search
747    '(#.*)?' +          // hash
748    '$');
749

751/**
752 * Resolves a potentially relative URL against a base location.
753 * @param {!Location} base Base location against which to resolve.
754 * @param {string} rel Url to resolve against the location.
755 * @return {string} Resolution of url against base location.
756 * @private
757 */
758bot.Device.resolveUrl_ = function(base, rel) {
759  var m = rel.match(bot.Device.URL_REGEXP_);
760  if (!m) {
761    return '';
762  }
763  var target = {
764    protocol: m[1] || '',
765    host: m[2] || '',
766    pathname: m[3] || '',
767    search: m[4] || '',
768    hash: m[5] || ''
769  };
770
771  if (!target.protocol) {
772    target.protocol = base.protocol;
773    if (!target.host) {
774      target.host = base.host;
775      if (!target.pathname) {
776        target.pathname = base.pathname;
777        target.search = target.search || base.search;
778      } else if (target.pathname.charAt(0) != '/') {
779        var lastSlashIndex = base.pathname.lastIndexOf('/');
780        if (lastSlashIndex != -1) {
781          var directory = base.pathname.substr(0, lastSlashIndex + 1);
782          target.pathname = directory + target.pathname;
783        }
784      }
785    }
786  }
787
788  return target.protocol + '//' + target.host + target.pathname +
789      target.search + target.hash;
790};
