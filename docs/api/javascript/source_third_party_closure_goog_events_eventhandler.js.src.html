15/**
16 * @fileoverview Class to create objects which want to handle multiple events
17 * and have their listeners easily cleaned up via a dispose method.
18 *
19 * Example:
20 * <pre>
21 * function Something() {
22 *   goog.base(this);
23 *
24 *   ... set up object ...
25 *
26 *   // Add event listeners
27 *   this.listen(this.starEl, goog.events.EventType.CLICK, this.handleStar);
28 *   this.listen(this.headerEl, goog.events.EventType.CLICK, this.expand);
29 *   this.listen(this.collapseEl, goog.events.EventType.CLICK, this.collapse);
30 *   this.listen(this.infoEl, goog.events.EventType.MOUSEOVER, this.showHover);
31 *   this.listen(this.infoEl, goog.events.EventType.MOUSEOUT, this.hideHover);
32 * }
33 * goog.inherits(Something, goog.events.EventHandler);
34 *
35 * Something.prototype.disposeInternal = function() {
36 *   goog.base(this, 'disposeInternal');
37 *   goog.dom.removeNode(this.container);
38 * };
39 *
40 *
41 * // Then elsewhere:
42 *
43 * var activeSomething = null;
44 * function openSomething() {
45 *   activeSomething = new Something();
46 * }
47 *
48 * function closeSomething() {
49 *   if (activeSomething) {
50 *     activeSomething.dispose();  // Remove event listeners
51 *     activeSomething = null;
52 *   }
53 * }
54 * </pre>
55 *
56 */
57

66/**
67 * Super class for objects that want to easily manage a number of event
68 * listeners.  It allows a short cut to listen and also provides a quick way
69 * to remove all events listeners belonging to this object.
70 * @param {Object=} opt_handler Object in whose scope to call the listeners.
71 * @constructor
72 * @extends {goog.Disposable}
73 */
74goog.events.EventHandler = function(opt_handler) {
75  goog.Disposable.call(this);
76  this.handler_ = opt_handler;
77
78  /**
79   * Keys for events that are being listened to.
80   * @type {!Object.<!goog.events.Key>}
81   * @private
82   */
83  this.keys_ = {};
84};
85goog.inherits(goog.events.EventHandler, goog.Disposable);

98/**
99 * Listen to an event on a Listenable.  If the function is omitted then the
100 * EventHandler's handleEvent method will be used.
101 * @param {goog.events.ListenableType} src Event source.
102 * @param {string|Array.<string>} type Event type to listen for or array of
103 *     event types.
104 * @param {Function|Object=} opt_fn Optional callback function to be used as the
105 *    listener or an object with handleEvent function.
106 * @param {boolean=} opt_capture Optional whether to use capture phase.
107 * @param {Object=} opt_handler Object in whose scope to call the listener.
108 * @return {goog.events.EventHandler} This object, allowing for chaining of
109 *     calls.
110 */
111goog.events.EventHandler.prototype.listen = function(src, type, opt_fn,
112                                                     opt_capture,
113                                                     opt_handler) {
114  if (!goog.isArray(type)) {
115    goog.events.EventHandler.typeArray_[0] = /** @type {string} */(type);
116    type = goog.events.EventHandler.typeArray_;
117  }
118  for (var i = 0; i < type.length; i++) {
119    var listenerObj = goog.events.listen(
120        src, type[i], opt_fn || this,
121        opt_capture || false,
122        opt_handler || this.handler_ || this);
123
124    if (goog.DEBUG && !listenerObj) {
125      // Some tests mock goog.events.listen, thus ensuring that
126      // they are never testing the real thing anyway, hence this is safe
127      // (except that #getListenerCount() will return the wrong value).
128      return this;
129    }
130
131    var key = listenerObj.key;
132    this.keys_[key] = listenerObj;
133  }
134
135  return this;
136};

139/**
140 * Listen to an event on a Listenable.  If the function is omitted, then the
141 * EventHandler's handleEvent method will be used. After the event has fired the
142 * event listener is removed from the target. If an array of event types is
143 * provided, each event type will be listened to once.
144 * @param {goog.events.ListenableType} src Event source.
145 * @param {string|Array.<string>} type Event type to listen for or array of
146 *     event types.
147 * @param {Function|Object=} opt_fn Optional callback function to be used as the
148 *    listener or an object with handleEvent function.
149 * @param {boolean=} opt_capture Optional whether to use capture phase.
150 * @param {Object=} opt_handler Object in whose scope to call the listener.
151 * @return {goog.events.EventHandler} This object, allowing for chaining of
152 *     calls.
153 */
154goog.events.EventHandler.prototype.listenOnce = function(src, type, opt_fn,
155                                                         opt_capture,
156                                                         opt_handler) {
157  if (goog.isArray(type)) {
158    for (var i = 0; i < type.length; i++) {
159      this.listenOnce(src, type[i], opt_fn, opt_capture, opt_handler);
160    }
161  } else {
162    var listenerObj = goog.events.listenOnce(
163        src, type, opt_fn || this, opt_capture,
164        opt_handler || this.handler_ || this);
165    var key = listenerObj.key;
166    this.keys_[key] = listenerObj;
167  }
168
169  return this;
170};

173/**
174 * Adds an event listener with a specific event wrapper on a DOM Node or an
175 * object that has implemented {@link goog.events.EventTarget}. A listener can
176 * only be added once to an object.
177 *
178 * @param {EventTarget|goog.events.EventTarget} src The node to listen to
179 *     events on.
180 * @param {goog.events.EventWrapper} wrapper Event wrapper to use.
181 * @param {Function|Object} listener Callback method, or an object with a
182 *     handleEvent function.
183 * @param {boolean=} opt_capt Whether to fire in capture phase (defaults to
184 *     false).
185 * @param {Object=} opt_handler Element in whose scope to call the listener.
186 * @return {goog.events.EventHandler} This object, allowing for chaining of
187 *     calls.
188 */
189goog.events.EventHandler.prototype.listenWithWrapper = function(src, wrapper,
190    listener, opt_capt, opt_handler) {
191  wrapper.listen(src, listener, opt_capt, opt_handler || this.handler_ || this,
192                 this);
193  return this;
194};

197/**
198 * @return {number} Number of listeners registered by this handler.
199 */
200goog.events.EventHandler.prototype.getListenerCount = function() {
201  var count = 0;
202  for (var key in this.keys_) {
203    if (Object.prototype.hasOwnProperty.call(this.keys_, key)) {
204      count++;
205    }
206  }
207  return count;
208};

211/**
212 * Unlistens on an event.
213 * @param {goog.events.ListenableType} src Event source.
214 * @param {string|Array.<string>} type Event type to listen for.
215 * @param {Function|Object=} opt_fn Optional callback function to be used as the
216 *    listener or an object with handleEvent function.
217 * @param {boolean=} opt_capture Optional whether to use capture phase.
218 * @param {Object=} opt_handler Object in whose scope to call the listener.
219 * @return {goog.events.EventHandler} This object, allowing for chaining of
220 *     calls.
221 */
222goog.events.EventHandler.prototype.unlisten = function(src, type, opt_fn,
223                                                       opt_capture,
224                                                       opt_handler) {
225  if (goog.isArray(type)) {
226    for (var i = 0; i < type.length; i++) {
227      this.unlisten(src, type[i], opt_fn, opt_capture, opt_handler);
228    }
229  } else {
230    var listener = goog.events.getListener(src, type, opt_fn || this,
231        opt_capture, opt_handler || this.handler_ || this);
232
233    if (listener) {
234      goog.events.unlistenByKey(listener);
235      delete this.keys_[listener.key];
236    }
237  }
238
239  return this;
240};

243/**
244 * Removes an event listener which was added with listenWithWrapper().
245 *
246 * @param {EventTarget|goog.events.EventTarget} src The target to stop
247 *     listening to events on.
248 * @param {goog.events.EventWrapper} wrapper Event wrapper to use.
249 * @param {Function|Object} listener The listener function to remove.
250 * @param {boolean=} opt_capt In DOM-compliant browsers, this determines
251 *     whether the listener is fired during the capture or bubble phase of the
252 *     event.
253 * @param {Object=} opt_handler Element in whose scope to call the listener.
254 * @return {goog.events.EventHandler} This object, allowing for chaining of
255 *     calls.
256 */
257goog.events.EventHandler.prototype.unlistenWithWrapper = function(src, wrapper,
258    listener, opt_capt, opt_handler) {
259  wrapper.unlisten(src, listener, opt_capt,
260                   opt_handler || this.handler_ || this, this);
261  return this;
262};

265/**
266 * Unlistens to all events.
267 */
268goog.events.EventHandler.prototype.removeAll = function() {
269  goog.object.forEach(this.keys_, goog.events.unlistenByKey);
270  this.keys_ = {};
271};

274/**
275 * Disposes of this EventHandler and removes all listeners that it registered.
276 * @override
277 * @protected
278 */
279goog.events.EventHandler.prototype.disposeInternal = function() {
280  goog.events.EventHandler.superClass_.disposeInternal.call(this);
281  this.removeAll();
282};
