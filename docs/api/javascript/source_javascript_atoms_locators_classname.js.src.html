24/**
25 * Tests whether the standardized W3C Selectors API are available on an
26 * element.
27 * @param {!(Document|Element)} root The document or element to test for CSS
28 *     selector support.
29 * @return {boolean} Whether or not the root supports query selector APIs.
30 * @see http://www.w3.org/TR/selectors-api/
31 * @private
32 */
33bot.locators.className.canUseQuerySelector_ = function(root) {
34  return !!(root.querySelectorAll && root.querySelector);
35};

38/**
39 * Find an element by its class name.
40 * @param {string} target The class name to search for.
41 * @param {!(Document|Element)} root The document or element to perform the
42 *     search under.
43 * @return {Element} The first matching element found in the DOM, or null if no
44 *     such element could be found.
45 */
46bot.locators.className.single = function(target, root) {
47  if (!target) {
48    throw Error('No class name specified');
49  }
50
51  target = goog.string.trim(target);
52  if (target.split(/\s+/).length > 1) {
53    throw Error('Compound class names not permitted');
54  }
55
56  // Closure will not properly escape class names that contain a '.' when using
57  // the native selectors API, so we have to handle this ourselves.
58  if (bot.locators.className.canUseQuerySelector_(root)) {
59    return root.querySelector('.' + target.replace(/\./g, '\\.')) || null;
60  }
61  var elements = goog.dom.getDomHelper(root).getElementsByTagNameAndClass(
62      /*tagName=*/'*', /*className=*/target, root);
63  return elements.length ? elements[0] : null;
64};

67/**
68 * Find an element by its class name.
69 * @param {string} target The class name to search for.
70 * @param {!(Document|Element)} root The document or element to perform the
71 *     search under.
72 * @return {!goog.array.ArrayLike} All matching elements, or an empty list.
73 */
74bot.locators.className.many = function(target, root) {
75  if (!target) {
76    throw Error('No class name specified');
77  }
78
79  target = goog.string.trim(target);
80  if (target.split(/\s+/).length > 1) {
81    throw Error('Compound class names not permitted');
82  }
83
84  // Closure will not properly escape class names that contain a '.' when using
85  // the native selectors API, so we have to handle this ourselves.
86  if (bot.locators.className.canUseQuerySelector_(root)) {
87    return root.querySelectorAll('.' + target.replace(/\./g, '\\.'));
88  }
89  return goog.dom.getDomHelper(root).getElementsByTagNameAndClass(
90      /*tagName=*/'*', /*className=*/target, root);
