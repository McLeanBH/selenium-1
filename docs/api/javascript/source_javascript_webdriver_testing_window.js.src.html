26/**
27 * Class for managing a window.
28 *
29 * <p>This class is implemented as a promise so consumers may register
30 * callbacks on it to handle situations where the window fails to open.
31 *
32 * For example:
33 * <pre><code>
34 *   var testWindow = webdriver.testing.Window.create(driver);
35 *   // Throw a custom error when the window fails to open.
36 *   testWindow.addErrback(function(e) {
37 *     throw Error('Failed to open test window: ' + e);
38 *   });
39 * </code></pre>
40 *
41 * @param {!webdriver.WebDriver} driver The driver to use.
42 * @param {(string|!webdriver.promise.Promise)} handle Either the managed
43 *     window's handle, or a promise that will resolve to it.
44 * @param {(Window|webdriver.promise.Promise)=} opt_window The raw window
45 *     object, if available.
46 * @constructor
47 * @extends {webdriver.promise.Promise}
48 */
49webdriver.testing.Window = function(driver, handle, opt_window) {
50  webdriver.promise.Promise.call(this);
51
52  /** @private {!webdriver.WebDriver} */
53  this.driver_ = driver;
54
55  /** @private {!webdriver.promise.Promise} */
56  this.handle_ = webdriver.promise.when(handle);
57
58  /** @private {!webdriver.promise.Promise} */
59  this.window_ = webdriver.promise.when(opt_window);
60};
61goog.inherits(webdriver.testing.Window, webdriver.promise.Promise);

80/**
81 * Opens and focuses on a new window.
82 *
83 * @param {!webdriver.WebDriver} driver The driver to use.
84 * @param {?{width: number, height: number}=} opt_size The desired size for
85 *     the new window.
86 * @param {number=} opt_timeout How long, in milliseconds, to wait for the new
87 *     window to open. Defaults to
88 *     {@link webdriver.testing.Window.DEFAULT_OPEN_TIMEOUT}.
89 * @return {!webdriver.testing.Window} The new window.
90 */
91webdriver.testing.Window.create = function(driver, opt_size, opt_timeout) {
92  var windowPromise = webdriver.promise.defer();
93  var handle = driver.call(function() {
94    var features = [
95      'location=yes',
96      'titlebar=yes'
97    ];
98
99    if (opt_size) {
100      features.push(
101          'width=' + opt_size.width,
102          'height=' + opt_size.height);
103    }
104
105    var name = goog.string.getRandomString();
106    windowPromise.fulfill(window.open('', name, features.join(',')));
107
108    driver.wait(function() {
109      return driver.switchTo().window(name).then(
110          function() { return true; },
111          function() { return false; });
112    }, opt_timeout || webdriver.testing.Window.DEFAULT_OPEN_TIMEOUT);
113    return driver.getWindowHandle();
114  });
115
116  return new webdriver.testing.Window(driver, handle, windowPromise);
117};

120/**
121 * Changes focus to the topmost window for the provided DOM window.
122 *
123 * @param {!webdriver.WebDriver} driver The driver to use.
124 * @param {Window=} opt_window The window to search for. Defaults to the window
125 *     running this script.
126 * @return {!webdriver.testing.Window} The located window.
127 */
128webdriver.testing.Window.focusOnWindow = function(driver, opt_window) {
129  var name = goog.string.getRandomString();
130  var win = opt_window ? opt_window.top : window;
131
132  var ret;
133  if (win === window && webdriver.testing.Window.currentWindow_) {
134    ret = webdriver.testing.Window.currentWindow_;
135    webdriver.testing.Window.currentWindow_.focus();
136  } else {
137    win.name = name;
138    ret = new webdriver.testing.Window(driver,
139        driver.switchTo().window(name).
140            addCallback(driver.getWindowHandle, driver),
141        win);
142    if (win === window) {
143      webdriver.testing.Window.currentWindow_ = ret;
144    }
145  }
146  return ret;
147};

162/**
163 * Focuses the wrapped driver on the window managed by this class.
164 * @return {!webdriver.promise.Promise} A promise that will be resolved
165 *     when this command has completed.
166 */
167webdriver.testing.Window.prototype.focus = function() {
168  return this.driver_.switchTo().window(this.handle_);
169};

172/**
173 * Focuses on and closes the managed window. The driver <em>must</em> be
174 * focused on another window before issuing any further commands.
175 * @return {!webdriver.promise.Promise} A promise that will be resolved
176 *     when this command has completed.
177 */
178webdriver.testing.Window.prototype.close = function() {
179  return this.window_.addCallback(function(win) {
180    if (win) {
181      win.close();
182    } else {
183      return this.focus().addCallback(this.driver_.close, this.driver_);
184    }
185  }, this);
186};

189/**
190 * Retrieves the current size of this window.
191 * @return {!webdriver.promise.Promise} A promise that resolves to the size of
192 *     this window as a {width:number, height:number} object.
193 */
194webdriver.testing.Window.prototype.getSize = function() {
195  var driver = this.driver_;
196  return this.focus().then(function() {
197    return driver.manage().window().getSize();
198  });
199};
