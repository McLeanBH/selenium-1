15goog.provide('webdriver.testing.Client');

17goog.require('goog.json');
18goog.require('goog.net.XmlHttp');

22/**
23 * The client responsible for publishing test events to the server. Each event
24 * will be published using a POST {@link goog.net.XmlHttp} request. The body of
25 * each request will be a JSON object with the following fields:
26 * <ul>
27 *   <li>id: An identifier for this client, derived from the window
28 *       locations' pathname.
29 *   <li>type: The type of event.
30 *   <li>data: A JSONObject whose contents will be specific to each event type.
31 * </ul>
32 *
33 * @param {Window=} opt_win The window to pull the path name from for this
34 *     client. Defaults to the current window.
35 * @param {string=} opt_url The URL to publish test events to. Defaults to
36 *     {@link webdriver.testing.Client.DEFAULT_URL}.
37 * @constructor
38 */
39webdriver.testing.Client = function(opt_win, opt_url) {
40
41  /** @private {string} */
42  this.id_ = (opt_win || window).location.pathname;
43
44  /** @private {string} */
45  this.url_ = opt_url || webdriver.testing.Client.DEFAULT_URL;
46};
47
48
49/**
50 * Default URL to publish test events to.
51 * @type {string}
52 * @const
53 */
54webdriver.testing.Client.DEFAULT_URL = '/testevent';

57/**
58 * The types of events that may be published by a TestClient to the server.
59 * @enum {string}
60 * @private
61 */
62webdriver.testing.Client.EventType_ = {
63
64  /** Sent to signal that a test suite has been fully initialized. */
65  INIT: 'INIT',
66
67  /**
68   * Sent when starting a new test.  The data object will have the following
69   * fields:
70   *  - name: The name of the test.
71   */
72  START_TEST: 'START_TEST',
73
74  /**
75   * Sent when an error has occurred. The data object will have the following
76   * fields:
77   *  - message: The error message.
78   */
79  ERROR: 'ERROR',
80
81  /**
82   * Sent when all tests have completed. The data object will have the following
83   * fields:
84   *  - isSuccess: Whether the tests succeeded.
85   *  - report: A flat log for the test suite.
86   */
87  RESULTS: 'RESULTS',
88
89  /**
90   * Sent when there is a screenshot for the server to record. The included data
91   * object will have two fields:
92   *  - name: A debug label for the screenshot.
93   *  - data: The PNG screenshot as a base64 string.
94   */
95  SCREENSHOT: 'SCREENSHOT'
96};

99/**
100 * Sends a simple message to the server, notifying it that the test runner has
101 * been initialized.
102 */
103webdriver.testing.Client.prototype.sendInitEvent = function() {
104  this.sendEvent_(webdriver.testing.Client.EventType_.INIT);
105};

108/**
109 * Sends an error event.
110 * @param {string} message The error message.

112webdriver.testing.Client.prototype.sendErrorEvent = function(message) {

130/**
131 * Sends an event to the server indicating that tests have completed.
132 * @param {boolean} isSuccess Whether the tests finished successfully.
133 * @param {string} report The test log.

135webdriver.testing.Client.prototype.sendResultsEvent = function(isSuccess,

144/**
145* Sends a screenshot to be recorded by the server.
146* @param {string} data The PNG screenshot as a base64 string.
147* @param {string=} opt_name A debug label for the screenshot; if omitted, the
148*     server will generate a random name.
149*/
150webdriver.testing.Client.prototype.sendScreenshotEvent = function(data,
151    opt_name) {
152  this.sendEvent_(webdriver.testing.Client.EventType_.SCREENSHOT, {
153    'name': opt_name,
154    'data': data
155  });
156};

159/**
160* Sends an event to the server.
161* @param {string} type The type of event to send.
162* @param {!Object=} opt_data JSON data object to send with the event, if any.
163* @private
164*/
165webdriver.testing.Client.prototype.sendEvent_ = function(type, opt_data) {
166  var payload = goog.json.serialize({
167    'id': this.id_,
168    'type': type,
169    'data': opt_data || {}
170  });
171
172  var xhr = new goog.net.XmlHttp;
173  xhr.open('POST', this.url_, true);
174  xhr.send(payload);
175  // TODO: Log if the event was not sent properly.
176};
