16goog.provide('bot.locators.id');

18goog.require('bot.dom');
19goog.require('goog.array');
20goog.require('goog.dom');

23/**
24 * Find an element by using the value of the ID attribute.
25 * @param {string} target The id to search for.
26 * @param {!(Document|Element)} root The document or element to perform the
27 *     search under.
28 * @return {Element} The first matching element found in the DOM, or null if no
29 *     such element could be found.
30 */
31bot.locators.id.single = function(target, root) {
32  var dom = goog.dom.getDomHelper(root);
33
34  var e = dom.getElement(target);
35  if (!e) {
36    return null;
37  }
38
39  // On IE getting by ID returns the first match by id _or_ name.
40  if (bot.dom.getAttribute(e, 'id') == target && goog.dom.contains(root, e)) {
41    return e;
42  }
43
44  var elements = dom.getElementsByTagNameAndClass('*');
45  var element = goog.array.find(elements, function(element) {
46    return bot.dom.getAttribute(element, 'id') == target &&
47        goog.dom.contains(root, element);
48  });
49  return /**@type{Element}*/ (element);
50};

53/**
54 * Find many elements by using the value of the ID attribute.
55 * @param {string} target The id to search for.
56 * @param {!(Document|Element)} root The document or element to perform the
57 *     search under.
58 * @return {!goog.array.ArrayLike} All matching elements, or an empty list.
59 */
60bot.locators.id.many = function(target, root) {
61  var dom = goog.dom.getDomHelper(root);
62  var elements = dom.getElementsByTagNameAndClass('*', null, root);
63  return goog.array.filter(elements, function(e) {
64    return bot.dom.getAttribute(e, 'id') == target;
65  });
66};
