13/**
14 * Constructor for FilterExpr.
15 *
16 * @param {!wgxpath.Expr} primary The primary expression.
17 * @param {!wgxpath.Predicates} predicates The predicates.
18 * @extends {wgxpath.Expr}
19 * @constructor
20 */
21wgxpath.FilterExpr = function(primary, predicates) {
22  if (predicates.getLength() && primary.getDataType() !=
23      wgxpath.DataType.NODESET) {
24    throw Error('Primary expression must evaluate to nodeset ' +
25        'if filter has predicate(s).');
26  }
27  wgxpath.Expr.call(this, primary.getDataType());
28
29  /**
30   * @type {!wgxpath.Expr}
31   * @private
32   */
33  this.primary_ = primary;
34
35
36  /**
37   * @type {!wgxpath.Predicates}
38   * @private
39   */
40  this.predicates_ = predicates;
41
42  this.setNeedContextPosition(primary.doesNeedContextPosition());
43  this.setNeedContextNode(primary.doesNeedContextNode());
44};
45goog.inherits(wgxpath.FilterExpr, wgxpath.Expr);
46
47
48/**
49 * @override
50 * @return {!wgxpath.NodeSet} The nodeset result.
51 */
52wgxpath.FilterExpr.prototype.evaluate = function(ctx) {
53  var result = this.primary_.evaluate(ctx);
54  return this.predicates_.evaluatePredicates(
55      /** @type {!wgxpath.NodeSet} */ (result));
56};
57
58
59/**
60 * @override
61 */
62wgxpath.FilterExpr.prototype.toString = function() {
63  var text = 'Filter:';
64  text += wgxpath.Expr.indent(this.primary_);
65  text += wgxpath.Expr.indent(this.predicates_);
66  return text;
67};
