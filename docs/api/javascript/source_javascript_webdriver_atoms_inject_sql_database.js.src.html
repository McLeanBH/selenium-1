28/**
29 * Executes the given query in the Web SQL database specified.
30 *
31 * @param {string} databaseName The name of the database.
32 * @param {string} query The SQL statement.
33 * @param {Array.<*>} args Arguments to pass to the query.
34 * @param {function(string)} onDone The callback to invoke when done. The
35 *     result, according to the wire protocol, will be passed to this callback.
36 */
37webdriver.atoms.inject.storage.database.executeSql =
38    function(databaseName, query, args, onDone) {
39  var onSuccessCallback = function(tx, result) {
40    onDone(webdriver.atoms.inject.executeScript(function(res) {
41      return result;
42    }, [result]));
43  };
44
45  var onErrorCallback = function(error) {
46    onDone(webdriver.atoms.inject.executeScript(function() {
47      throw new bot.Error(bot.ErrorCode.SQL_DATABASE_ERROR,
48          'SQL Error Code: ' + error.code + '. SQL Error Message: ' +
49          error.message);
50    }, []));
51  };
52
53  bot.storage.database.executeSql(
54      databaseName, query, args, onSuccessCallback, onErrorCallback);
55};
