19/**
20 * An abstract class for expressions with predicates.
21 *
22 * @constructor
23 * @param {!Array.<!wgxpath.Expr>} predicates The array of predicates.
24 * @param {boolean=} opt_reverse Whether to iterate over the nodeset in reverse.
25 */
26wgxpath.Predicates = function(predicates, opt_reverse) {
27
28  /**
29   * List of predicates
30   *
31   * @private
32   * @type {!Array.<!wgxpath.Expr>}
33   */
34  this.predicates_ = predicates;
35
36
37  /**
38   * Which direction to iterate over the predicates
39   *
40   * @private
41   * @type {boolean}
42   */
43  this.reverse_ = !!opt_reverse;
44};
45

47/**
48 * Evaluates the predicates against the given nodeset.
49 *
50 * @param {!wgxpath.NodeSet} nodeset The nodes against which to evaluate
51 *     the predicates.
52 * @param {number=} opt_start The index of the first predicate to evaluate,
53 *     defaults to 0.
54 * @return {!wgxpath.NodeSet} nodeset The filtered nodeset.
55 */
56wgxpath.Predicates.prototype.evaluatePredicates =
57    function(nodeset, opt_start) {
58  for (var i = opt_start || 0; i < this.predicates_.length; i++) {
59    var predicate = this.predicates_[i];
60    var iter = nodeset.iterator();
61    var l = nodeset.getLength();
62    var node;
63    for (var j = 0; node = iter.next(); j++) {
64      var position = this.reverse_ ? (l - j) : (j + 1);
65      var exrs = predicate.evaluate(new
66          wgxpath.Context(/** @type {wgxpath.Node} */ (node), position, l));
67      var keep;
68      if (typeof exrs == 'number') {
69        keep = (position == exrs);
70      } else if (typeof exrs == 'string' || typeof exrs == 'boolean') {
71        keep = !!exrs;
72      } else if (exrs instanceof wgxpath.NodeSet) {
73        keep = (exrs.getLength() > 0);
74      } else {
75        throw Error('Predicate.evaluate returned an unexpected type.');
76      }
77      if (!keep) {
78        iter.remove();
79      }
80    }
81  }
82  return nodeset;
83};

86/**
87 * Returns the quickAttr info.
88 *
89 * @return {?{name: string, valueExpr: wgxpath.Expr}}
90 */
91wgxpath.Predicates.prototype.getQuickAttr = function() {
92  return this.predicates_.length > 0 ?
93      this.predicates_[0].getQuickAttr() : null;

97/**
98 * Returns whether this set of predicates needs context position.
99 *
100 * @return {boolean} Whether something needs context position.
101 */
102wgxpath.Predicates.prototype.doesNeedContextPosition = function() {
103  for (var i = 0; i < this.predicates_.length; i++) {
104    var predicate = this.predicates_[i];
105    if (predicate.doesNeedContextPosition() ||
106        predicate.getDataType() == wgxpath.DataType.NUMBER ||
107        predicate.getDataType() == wgxpath.DataType.VOID) {
108      return true;
109    }
110  }
111  return false;
112};
