13/**
14 * Constructs a NameTest based on the xpath grammar:
15 * http://www.w3.org/TR/xpath/#NT-NameTest
16 *
17 * <p>If no namespace is provided, the default HTML namespace is used.
18 *
19 * @param {string} name Name to be tested.
20 * @param {string=} opt_namespaceUri Namespace URI; defaults to HTML namespace.
21 * @constructor
22 * @implements {wgxpath.NodeTest}
23 */
24wgxpath.NameTest = function(name, opt_namespaceUri) {
25  /**
26   * @type {string}
27   * @private
28   */
29  this.name_ = name.toLowerCase();
30
31  /**
32   * @type {string}
33   * @private
34   */
35  this.namespaceUri_ = opt_namespaceUri ? opt_namespaceUri.toLowerCase() :
36      wgxpath.NameTest.HTML_NAMESPACE_URI_;
37};
38
39
40/**
41 * The default namespace URI for XHTML nodes.
42 *
43 * @const
44 * @type {string}
45 * @private
46 */
47wgxpath.NameTest.HTML_NAMESPACE_URI_ = 'http://www.w3.org/1999/xhtml';
48
49
50/**
51 * @override
52 */
53wgxpath.NameTest.prototype.matches = function(node) {
54  var type = node.nodeType;
55  if (type != goog.dom.NodeType.ELEMENT &&
56      type != goog.dom.NodeType.ATTRIBUTE) {
57    return false;
58  }
59  if (this.name_ != '*' && this.name_ != node.nodeName.toLowerCase()) {
60    return false;
61  } else {
62    var namespaceUri = node.namespaceURI ? node.namespaceURI.toLowerCase() :
63        wgxpath.NameTest.HTML_NAMESPACE_URI_;
64    return this.namespaceUri_ == namespaceUri;
65  }
66};

70 * @override
71 */
72wgxpath.NameTest.prototype.getName = function() {
73  return this.name_;
74};

78 * Returns the namespace URI to be matched.
79 *
80 * @return {string} Namespace URI.
81 */
82wgxpath.NameTest.prototype.getNamespaceUri = function() {
83  return this.namespaceUri_;
84};

87/**
88 * @override
89 */
90wgxpath.NameTest.prototype.toString = function() {
91  var prefix = this.namespaceUri_ == wgxpath.NameTest.HTML_NAMESPACE_URI_ ?
92      '' : this.namespaceUri_ + ':';
93  return 'Name Test: ' + prefix + this.name_;
94};
