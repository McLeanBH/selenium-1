51/**
52 * The exported XPathExpression type.
53 *
54 * @constructor
55 * @extends {XPathExpression}
56 * @param {string} expr The expression string.
57 * @param {?(XPathNSResolver|function(string): ?string)} nsResolver
58 *     XPath namespace resolver.
59 * @private
60 */
61wgxpath.XPathExpression_ = function(expr, nsResolver) {
62  if (!expr.length) {
63    throw Error('Empty XPath expression.');
64  }
65  var lexer = wgxpath.Lexer.tokenize(expr);
66  if (lexer.empty()) {
67    throw Error('Invalid XPath expression.');
68  }
69
70  // nsResolver may either be an XPathNSResolver, which has a lookupNamespaceURI
71  // function, a custom function, or null. Standardize it to a function.
72  if (!nsResolver) {
73    nsResolver = function(string) {return null;};
74  } else if (!goog.isFunction(nsResolver)) {
75    nsResolver = goog.bind(nsResolver.lookupNamespaceURI, nsResolver);
76  }
77
78  var gexpr = new wgxpath.Parser(lexer, nsResolver).parseExpr();
79  if (!lexer.empty()) {
80    throw Error('Bad token: ' + lexer.next());
81  }
82  this['evaluate'] = function(node, type) {
83    var value = gexpr.evaluate(new wgxpath.Context(node));
84    return new wgxpath.XPathResult_(value, type);
85  };
86};

90/**
91 * The exported XPathResult type.
92 *
93 * @constructor
94 * @extends {XPathResult}
95 * @param {(!wgxpath.NodeSet|number|string|boolean)} value The result value.
96 * @param {number} type The result type.
97 * @private
98 */
99wgxpath.XPathResult_ = function(value, type) {
100  if (type == wgxpath.XPathResultType_.ANY_TYPE) {
101    if (value instanceof wgxpath.NodeSet) {
102      type = wgxpath.XPathResultType_.UNORDERED_NODE_ITERATOR_TYPE;
103    } else if (typeof value == 'string') {
104      type = wgxpath.XPathResultType_.STRING_TYPE;
105    } else if (typeof value == 'number') {
106      type = wgxpath.XPathResultType_.NUMBER_TYPE;
107    } else if (typeof value == 'boolean') {
108      type = wgxpath.XPathResultType_.BOOLEAN_TYPE;
109    } else {
110      throw Error('Unexpected evaluation result.');
111    }
112  }
113  if (type != wgxpath.XPathResultType_.STRING_TYPE &&
114      type != wgxpath.XPathResultType_.NUMBER_TYPE &&
115      type != wgxpath.XPathResultType_.BOOLEAN_TYPE &&
116      !(value instanceof wgxpath.NodeSet)) {
117    throw Error('value could not be converted to the specified type');
118  }
119  this['resultType'] = type;
120  var nodes;
121  switch (type) {
122    case wgxpath.XPathResultType_.STRING_TYPE:
123      this['stringValue'] = (value instanceof wgxpath.NodeSet) ?
124          value.string() : '' + value;
125      break;
126    case wgxpath.XPathResultType_.NUMBER_TYPE:
127      this['numberValue'] = (value instanceof wgxpath.NodeSet) ?
128          value.number() : +value;
129      break;
130    case wgxpath.XPathResultType_.BOOLEAN_TYPE:
131      this['booleanValue'] = (value instanceof wgxpath.NodeSet) ?
132          value.getLength() > 0 : !!value;
133      break;
134    case wgxpath.XPathResultType_.UNORDERED_NODE_ITERATOR_TYPE:
135    case wgxpath.XPathResultType_.ORDERED_NODE_ITERATOR_TYPE:
136    case wgxpath.XPathResultType_.UNORDERED_NODE_SNAPSHOT_TYPE:
137    case wgxpath.XPathResultType_.ORDERED_NODE_SNAPSHOT_TYPE:
138      var iter = value.iterator();
139      nodes = [];
140      for (var node = iter.next(); node; node = iter.next()) {
141        nodes.push(node instanceof wgxpath.IEAttrWrapper ?
142            node.getNode() : node);
143      }
144      this['snapshotLength'] = value.getLength();
145      this['invalidIteratorState'] = false;
146      break;
147    case wgxpath.XPathResultType_.ANY_UNORDERED_NODE_TYPE:
148    case wgxpath.XPathResultType_.FIRST_ORDERED_NODE_TYPE:
149      var firstNode = value.getFirst();
150      this['singleNodeValue'] =
151          firstNode instanceof wgxpath.IEAttrWrapper ?
152          firstNode.getNode() : firstNode;
153      break;
154    default:
155      throw Error('Unknown XPathResult type.');
156  }
157  var index = 0;
158  this['iterateNext'] = function() {
159    if (type != wgxpath.XPathResultType_.UNORDERED_NODE_ITERATOR_TYPE &&
160        type != wgxpath.XPathResultType_.ORDERED_NODE_ITERATOR_TYPE) {
161      throw Error('iterateNext called with wrong result type');
162    }
163    return (index >= nodes.length) ? null : nodes[index++];
164  };
165  this['snapshotItem'] = function(i) {
166    if (type != wgxpath.XPathResultType_.UNORDERED_NODE_SNAPSHOT_TYPE &&
167        type != wgxpath.XPathResultType_.ORDERED_NODE_SNAPSHOT_TYPE) {
168      throw Error('snapshotItem called with wrong result type');
169    }
170    return (i >= nodes.length || i < 0) ? null : nodes[i];
171  };
172};

205/**
206 * Installs the library. This is a noop if native XPath is available.
207 *
208 * @param {Window=} opt_win The window to install the library on.
209 */
210wgxpath.install = function(opt_win) {
211  var win = opt_win || goog.global;
212  var doc = win.document;
213
214  // Installation is a noop if native XPath is available.
215  if (doc['evaluate']) {
216    return;
217  }
218
219  win['XPathResult'] = wgxpath.XPathResult_;
220  doc['evaluate'] = function(expr, context, nsResolver, type, result) {
221    return new wgxpath.XPathExpression_(expr, nsResolver).
222        evaluate(context, type);
223  };
224  doc['createExpression'] = function(expr, nsResolver) {
225    return new wgxpath.XPathExpression_(expr, nsResolver);
226  };
227  doc['createNSResolver'] = function(node) {
228    return new wgxpath.XPathNSResolver_(node);
229  };
