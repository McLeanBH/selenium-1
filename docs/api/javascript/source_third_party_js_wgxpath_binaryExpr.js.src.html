16/**
17 * Constructor for BinaryExpr.
18 *
19 * @param {!wgxpath.BinaryExpr.Op} op A binary operator.
20 * @param {!wgxpath.Expr} left The left hand side of the expression.
21 * @param {!wgxpath.Expr} right The right hand side of the expression.
22 * @extends {wgxpath.Expr}
23 * @constructor
24 */
25wgxpath.BinaryExpr = function(op, left, right) {
26  var opCast = /** @type {!wgxpath.BinaryExpr.Op_} */ (op);
27  wgxpath.Expr.call(this, opCast.dataType_);
28
29  /**
30   * @private
31   * @type {!wgxpath.BinaryExpr.Op_}
32   */
33  this.op_ = opCast;
34
35  /**
36   * @private
37   * @type {!wgxpath.Expr}
38   */
39  this.left_ = left;
40
41  /**
42   * @private
43   * @type {!wgxpath.Expr}
44   */
45  this.right_ = right;
46
47  this.setNeedContextPosition(left.doesNeedContextPosition() ||
48      right.doesNeedContextPosition());
49  this.setNeedContextNode(left.doesNeedContextNode() ||
50      right.doesNeedContextNode());
51
52  // Optimize [@id="foo"] and [@name="bar"]
53  if (this.op_ == wgxpath.BinaryExpr.Op.EQUAL) {
54    if (!right.doesNeedContextNode() && !right.doesNeedContextPosition() &&
55        right.getDataType() != wgxpath.DataType.NODESET &&
56        right.getDataType() != wgxpath.DataType.VOID && left.getQuickAttr()) {
57      this.setQuickAttr({
58        name: left.getQuickAttr().name,
59        valueExpr: right});
60    } else if (!left.doesNeedContextNode() && !left.doesNeedContextPosition() &&
61        left.getDataType() != wgxpath.DataType.NODESET &&
62        left.getDataType() != wgxpath.DataType.VOID && right.getQuickAttr()) {
63      this.setQuickAttr({
64        name: right.getQuickAttr().name,
65        valueExpr: left});
66    }
67  }
68};
69goog.inherits(wgxpath.BinaryExpr, wgxpath.Expr);

72/**
73 * Performs comparison between the left hand side and the right hand side.
74 *
75 * @private
76 * @param {function((string|number|boolean), (string|number|boolean))}
77 *        comp A comparison function that takes two parameters.
78 * @param {!wgxpath.Expr} lhs The left hand side of the expression.
79 * @param {!wgxpath.Expr} rhs The right hand side of the expression.
80 * @param {!wgxpath.Context} ctx The context to perform the comparison in.
81 * @param {boolean=} opt_equChk Whether the comparison checks for equality.
82 * @return {boolean} True if comp returns true, false otherwise.
83 */
84wgxpath.BinaryExpr.compare_ = function(comp, lhs, rhs, ctx, opt_equChk) {
85  var left = lhs.evaluate(ctx);
86  var right = rhs.evaluate(ctx);
87  var lIter, rIter, lNode, rNode;
88  if (left instanceof wgxpath.NodeSet && right instanceof wgxpath.NodeSet) {
89    lIter = left.iterator();
90    for (lNode = lIter.next(); lNode; lNode = lIter.next()) {
91      rIter = right.iterator();
92      for (rNode = rIter.next(); rNode; rNode = rIter.next()) {
93        if (comp(wgxpath.Node.getValueAsString(lNode),
94            wgxpath.Node.getValueAsString(rNode))) {
95          return true;
96        }
97      }
98    }
99    return false;
100  }
101  if ((left instanceof wgxpath.NodeSet) ||
102      (right instanceof wgxpath.NodeSet)) {
103    var nodeset, primitive;
104    if ((left instanceof wgxpath.NodeSet)) {
105      nodeset = left, primitive = right;
106    } else {
107      nodeset = right, primitive = left;
108    }
109    var iter = nodeset.iterator();
110    var type = typeof primitive;
111    for (var node = iter.next(); node; node = iter.next()) {
112      var stringValue;
113      switch (type) {
114        case 'number':
115          stringValue = wgxpath.Node.getValueAsNumber(node);
116          break;
117        case 'boolean':
118          stringValue = wgxpath.Node.getValueAsBool(node);
119          break;
120        case 'string':
121          stringValue = wgxpath.Node.getValueAsString(node);
122          break;
123        default:
124          throw Error('Illegal primitive type for comparison.');
125      }
126      if (comp(stringValue,
127          /** @type {(string|number|boolean)} */ (primitive))) {
128        return true;
129      }
130    }
131    return false;
132  }
133  if (opt_equChk) {
134    if (typeof left == 'boolean' || typeof right == 'boolean') {
135      return comp(!!left, !!right);
136    }
137    if (typeof left == 'number' || typeof right == 'number') {
138      return comp(+left, +right);
139    }
140    return comp(left, right);
141  }
142  return comp(+left, +right);
143};

233/**
234 * Creates a binary operator.
235 *
236 * @param {string} opString The operator string.
237 * @param {number} precedence The precedence when evaluated.
238 * @param {!wgxpath.DataType} dataType The dataType to return when evaluated.
239 * @param {function(!wgxpath.Expr, !wgxpath.Expr, !wgxpath.Context)}
240 *         evaluate An evaluation function.
241 * @return {!wgxpath.BinaryExpr.Op} A binary expression operator.
242 * @private
243 */
244wgxpath.BinaryExpr.createOp_ = function(opString, precedence, dataType,
245    evaluate) {
246  if (wgxpath.BinaryExpr.stringToOpMap_.hasOwnProperty(opString)) {
247    throw new Error('Binary operator already created: ' + opString);
248  }
249  // The upcast and then downcast for the JSCompiler.
250  var op = /** @type {!Object} */ (new wgxpath.BinaryExpr.Op_(
251      opString, precedence, dataType, evaluate));
252  op = /** @type {!wgxpath.BinaryExpr.Op} */ (op);
253  wgxpath.BinaryExpr.stringToOpMap_[op.toString()] = op;
254  return op;
255};
