16/**
17 * @fileoverview Functions to locate elements by XPath.
18 *
19 * <p>The locator implementations below differ from the Closure functions
20 * goog.dom.xml.{selectSingleNode,selectNodes} in three important ways:
21 * <ol>
22 * <li>they do not refer to "document" which is undefined in the context of a
23 * Firefox extension;
24 * <li> they use a default NsResolver for browsers that do not provide
25 * document.createNSResolver (e.g. Android); and
26 * <li> they prefer document.evaluate to node.{selectSingleNode,selectNodes}
27 * because the latter silently return nothing when the xpath resolves to a
28 * non-Node type, limiting the error-checking the implementation can provide.
29 * </ol>
30 */
31

46 * XPathResult enum values. These are defined separately since
47 * the context running this script may not support the XPathResult
48 * type.

66/**
67 * Default XPath namespace resolver.
68 * @private
69 */
70bot.locators.xpath.DEFAULT_RESOLVER_ = (function() {
71  var namespaces = {svg: 'http://www.w3.org/2000/svg'};
72  return function(prefix) {
73    return namespaces[prefix] || null;
74  };
75})();

78/**
79 * Evaluates an XPath expression using a W3 XPathEvaluator.
80 * @param {!(Document|Element)} node The document or element to perform the
81 *     search under.
82 * @param {string} path The xpath to search for.
83 * @param {!bot.locators.XPathResult_} resultType The desired result type.
84 * @return {XPathResult} The XPathResult or null if the root's ownerDocument
85 *     does not support XPathEvaluators.
86 * @private
87 * @see http://www.w3.org/TR/DOM-Level-3-XPath/xpath.html#XPathEvaluator-evaluate
88 */
89bot.locators.xpath.evaluate_ = function(node, path, resultType) {
90  var doc = goog.dom.getOwnerDocument(node);
91
92  // Let the wgxpath library be compiled away unless we are on IE or Android.
93  // TODO: Restrict this to just IE when we drop support for Froyo.
94  if (goog.userAgent.IE || goog.userAgent.product.ANDROID) {
95    wgxpath.install(goog.dom.getWindow(doc));
96  }
97
98  try {
99    var resolver = doc.createNSResolver ?
100        doc.createNSResolver(doc.documentElement) :
101        bot.locators.xpath.DEFAULT_RESOLVER_;
102    if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher(7)) {
103      // IE6, and only IE6, has an issue where calling a custom function
104      // directly attached to the document object does not correctly propagate
105      // thrown errors. So in that case *only* we will use apply().
106      return doc.evaluate.call(doc, path, node, resolver, resultType, null);
107    } else {
108      return doc.evaluate(path, node, resolver, resultType, null);
109    }
110  } catch (ex) {
111    // The Firefox XPath evaluator can throw an exception if the document is
112    // queried while it's in the midst of reloading, so we ignore it. In all
113    // other cases, we assume an invalid xpath has caused the exception.
114    if (!(goog.userAgent.GECKO && ex.name == 'NS_ERROR_ILLEGAL_VALUE')) {
115      throw new bot.Error(bot.ErrorCode.INVALID_SELECTOR_ERROR,
116          'Unable to locate an element with the xpath expression ' + path +
117          ' because of the following error:\n' + ex);
118    }
119  }
120};

123/**
124 * @param {Node|undefined} node Node to check whether it is an Element.
125 * @param {string} path XPath expression to include in the error message.
126 * @private
127 */
128bot.locators.xpath.checkElement_ = function(node, path) {
129  if (!node || node.nodeType != goog.dom.NodeType.ELEMENT) {
130    throw new bot.Error(bot.ErrorCode.INVALID_SELECTOR_ERROR,
131        'The result of the xpath expression "' + path +
132        '" is: ' + node + '. It should be an element.');
133  }
134};

137/**
138 * Find an element by using an xpath expression
139 * @param {string} target The xpath to search for.
140 * @param {!(Document|Element)} root The document or element to perform the
141 *     search under.
142 * @return {Element} The first matching element found in the DOM, or null if no
143 *     such element could be found.
144 */
145bot.locators.xpath.single = function(target, root) {
146
147  function selectSingleNode() {
148    var result = bot.locators.xpath.evaluate_(root, target,
149        bot.locators.XPathResult_.FIRST_ORDERED_NODE_TYPE);
150    if (result) {
151      var node = result.singleNodeValue;
152      // On Opera, a singleNodeValue of undefined indicates a type error, while
153      // other browsers may use it to indicate something has not been found.
154      return goog.userAgent.OPERA ? node : (node || null);
155    } else if (root.selectSingleNode) {
156      var doc = goog.dom.getOwnerDocument(root);
157      if (doc.setProperty) {
158        doc.setProperty('SelectionLanguage', 'XPath');
159      }
160      return root.selectSingleNode(target);
161    }
162    return null;
163  }
164
165  var node = selectSingleNode();
166  if (!goog.isNull(node)) {
167    bot.locators.xpath.checkElement_(node, target);
168  }
169  return /** @type {Element} */ (node);
170};

173/**
174 * Find elements by using an xpath expression
175 * @param {string} target The xpath to search for.
176 * @param {!(Document|Element)} root The document or element to perform the
177 *     search under.
178 * @return {!goog.array.ArrayLike} All matching elements, or an empty list.
179 */
180bot.locators.xpath.many = function(target, root) {
181
182  function selectNodes() {
183    var result = bot.locators.xpath.evaluate_(root, target,
184        bot.locators.XPathResult_.ORDERED_NODE_SNAPSHOT_TYPE);
185    if (result) {
186      var count = result.snapshotLength;
187      // On Opera, if the XPath evaluates to a non-Node value, snapshotLength
188      // will be undefined and the result empty, so fail immediately.
189      if (goog.userAgent.OPERA && !goog.isDef(count)) {
190        bot.locators.xpath.checkElement_(null, target);
191      }
192      var results = [];
193      for (var i = 0; i < count; ++i) {
194        results.push(result.snapshotItem(i));
195      }
196      return results;
197    } else if (root.selectNodes) {
198      var doc = goog.dom.getOwnerDocument(root);
199      if (doc.setProperty) {
200        doc.setProperty('SelectionLanguage', 'XPath');
201      }
202      return root.selectNodes(target);
203    }
204    return [];
205  }
206
207  var nodes = selectNodes();
208  goog.array.forEach(nodes, function(n) {
209    bot.locators.xpath.checkElement_(n, target);
210  });
211  return /** @type {!goog.array.ArrayLike} */ (nodes);
212};
