18/**
19 * Constructor for PathExpr.
20 *
21 * @param {!wgxpath.Expr} filter A filter expression.
22 * @param {!Array.<!wgxpath.Step>} steps The steps in the location path.
23 * @extends {wgxpath.Expr}
24 * @constructor
25 */
26wgxpath.PathExpr = function(filter, steps) {
27  wgxpath.Expr.call(this, filter.getDataType());
28
29  /**
30   * @type {!wgxpath.Expr}
31   * @private
32   */
33  this.filter_ = filter;
34
35  /**
36   * @type {!Array.<!wgxpath.Step>}
37   * @private
38   */
39  this.steps_ = steps;
40
41  this.setNeedContextPosition(filter.doesNeedContextPosition());
42  this.setNeedContextNode(filter.doesNeedContextNode());
43  if (this.steps_.length == 1) {
44    var firstStep = this.steps_[0];
45    if (!firstStep.doesIncludeDescendants() &&
46        firstStep.getAxis() == wgxpath.Step.Axis.ATTRIBUTE) {
47      var test = firstStep.getTest();
48      if (test.getName() != '*') {
49        this.setQuickAttr({
50          name: test.getName(),
51          valueExpr: null
52        });
53      }
54    }
55  }
56};
57goog.inherits(wgxpath.PathExpr, wgxpath.Expr);

74 * Evaluates the root-node helper expression.
75 *
76 * @param {!wgxpath.Context} ctx The context to evaluate the expression in.
77 * @return {!wgxpath.NodeSet} The evaluation result.
78 */
79wgxpath.PathExpr.RootHelperExpr.prototype.evaluate = function(ctx) {
80  var nodeset = new wgxpath.NodeSet();
81  var node = ctx.getNode();
82  if (node.nodeType == goog.dom.NodeType.DOCUMENT) {
83    nodeset.add(node);
84  } else {
85    nodeset.add(/** @type {!Node} */ (node.ownerDocument));
86  }
87  return nodeset;

144/**
145 * @override
146 * @return {!wgxpath.NodeSet} The nodeset result.
147 */
148wgxpath.PathExpr.prototype.evaluate = function(ctx) {
149  var nodeset = this.filter_.evaluate(ctx);
150  if (!(nodeset instanceof wgxpath.NodeSet)) {
151    throw Error('Filter expression must evaluate to nodeset.');
152  }
153  var steps = this.steps_;
154  for (var i = 0, l0 = steps.length; i < l0 && nodeset.getLength(); i++) {
155    var step = steps[i];
156    var reverse = step.getAxis().isReverse();
157    var iter = nodeset.iterator(reverse);
158    nodeset = null;
159    var node, next;
160    if (!step.doesNeedContextPosition() &&
161        step.getAxis() == wgxpath.Step.Axis.FOLLOWING) {
162      for (node = iter.next(); next = iter.next(); node = next) {
163        if (node.contains && !node.contains(next)) {
164          break;
165        } else {
166          if (!(next.compareDocumentPosition(/** @type {!Node} */ (node)) &
167              8)) {
168            break;
169          }
170        }
171      }
172      nodeset = step.evaluate(new
173          wgxpath.Context(/** @type {wgxpath.Node} */ (node)));
174    } else if (!step.doesNeedContextPosition() &&
175        step.getAxis() == wgxpath.Step.Axis.PRECEDING) {
176      node = iter.next();
177      nodeset = step.evaluate(new
178          wgxpath.Context(/** @type {wgxpath.Node} */ (node)));
179    } else {
180      node = iter.next();
181      nodeset = step.evaluate(new
182          wgxpath.Context(/** @type {wgxpath.Node} */ (node)));
183      while ((node = iter.next()) != null) {
184        var result = step.evaluate(new
185            wgxpath.Context(/** @type {wgxpath.Node} */ (node)));
186        nodeset = wgxpath.NodeSet.merge(nodeset, result);
187      }
188    }
189  }
190  return /** @type {!wgxpath.NodeSet} */ (nodeset);
191};

194/**
195 * @override
196 */
197wgxpath.PathExpr.prototype.toString = function() {
198  var text = 'Path Expression:';
199  text += wgxpath.Expr.indent(this.filter_);
200  if (this.steps_.length) {
201    var steps = goog.array.reduce(this.steps_, function(prev, curr) {
202      return prev + wgxpath.Expr.indent(curr);
203    }, 'Steps:');
204    text += wgxpath.Expr.indent(steps);
205  }
206  return text;
