33/**
34 * @param {!Element} element The element to use.
35 * @return {boolean} Whether the element is checked or selected.
36 */
37webdriver.atoms.element.isSelected = function(element) {
38  // Although this method looks unloved, its compiled form is used by
39  // the Chrome and OperaDrivers.
40  if (!bot.dom.isSelectable(element)) {
41    return false;
42  }
43
44  return bot.dom.isSelected(element);
45};

117/**
118 * Get the value of the given property or attribute. If the "attribute" is for
119 * a boolean property, we return null in the case where the value is false. If
120 * the attribute name is "style" an attempt to convert that style into a string
121 * is done.
122 *
123 * @param {!Element} element The element to use.
124 * @param {string} attribute The name of the attribute to look up.
125 * @return {?string} The string value of the attribute or property, or null.
126 */
127webdriver.atoms.element.getAttribute = function(element, attribute) {
128  var value = null;
129  var name = attribute.toLowerCase();
130
131  if ('style' == name) {
132    value = element.style;
133
134    if (value && !goog.isString(value)) {
135      value = value.cssText;
136    }
137
138    return /** @type {?string} */ (value);
139  }
140
141  if (('selected' == name || 'checked' == name) &&
142      bot.dom.isSelectable(element)) {
143    return bot.dom.isSelected(element) ? 'true' : null;
144  }
145
146  // Our tests suggest that returning the attribute is desirable for
147  // the href attribute of <a> tags and the src attribute of <img> tags,
148  // but we normally attempt to get the property value before the attribute.
149  var isLink = bot.dom.isElement(element, goog.dom.TagName.A);
150  var isImg = bot.dom.isElement(element, goog.dom.TagName.IMG);
151
152  // Although the attribute matters, the property is consistent. Return that in
153  // preference to the attribute for links and images.
154  if ((isImg && name == 'src') || (isLink && name == 'href')) {
155    value = bot.dom.getAttribute(element, name);
156    if (value) {
157      // We want the full URL if present
158      value = bot.dom.getProperty(element, name);
159    }
160    return /** @type {?string} */ (value);
161  }
162
163  var propName = webdriver.atoms.element.PROPERTY_ALIASES_[attribute] ||
164      attribute;
165  if (goog.array.contains(webdriver.atoms.element.BOOLEAN_PROPERTIES_, name)) {
166    value = !goog.isNull(bot.dom.getAttribute(element, attribute)) ||
167        bot.dom.getProperty(element, propName);
168    return value ? 'true' : null;
169  }
170
171  var property;
172  try {
173    property = bot.dom.getProperty(element, propName);
174  } catch (e) {
175    // Leaves property undefined or null
176  }
177
178  // 1- Call getAttribute if getProperty fails,
179  // i.e. property is null or undefined.
180  // This happens for event handlers in Firefox.
181  // For example, calling getProperty for 'onclick' would
182  // fail while getAttribute for 'onclick' will succeed and
183  // return the JS code of the handler.
184  //
185  // 2- When property is an object we fall back to the
186  // actual attribute instead.
187  // See issue http://code.google.com/p/selenium/issues/detail?id=966
188  if (!goog.isDefAndNotNull(property) || goog.isObject(property)) {
189    value = bot.dom.getAttribute(element, attribute);
190  } else {
191    value = property;
192  }
193
194  // The empty string is a valid return value.
195  return goog.isDefAndNotNull(value) ? value.toString() : null;
196};

199/**
200 * Get the location of the element in page space, if it's displayed.
201 *
202 * @param {!Element} element The element to get the location for.
203 * @return {goog.math.Rect} The bounding rectangle of the element.
204 */
205webdriver.atoms.element.getLocation = function(element) {
206  if (!bot.dom.isShown(element)) {
207    return null;
208  }
209  return goog.style.getBounds(element);
210};

213/**
214 * Scrolls the element into the client's view and returns its position
215 * relative to the client viewport. If the element or region is too
216 * large to fit in the view, it will be aligned to the top-left of the
217 * container.
218 *
219 * The element should be attached to the current document.
220 *
221 * @param {!Element} elem The element to use.
222 * @param {!goog.math.Rect=} opt_elemRegion The region relative to the element
223 *     to be scrolled into view.
224 * @return {!goog.math.Coordinate} The coordinate of the element in client
225 *     space.
226 */
227webdriver.atoms.element.getLocationInView = function(elem, opt_elemRegion) {
228  bot.action.scrollIntoView(elem, opt_elemRegion);
229  var region = bot.dom.getClientRegion(elem, opt_elemRegion);
230  return new goog.math.Coordinate(region.left, region.top);
231};

234/**
235 * @param {Node} element The element to use.
236 * @return {boolean} Whether the element is in the HEAD tag.
237 * @private
238 */
239webdriver.atoms.element.isInHead_ = function(element) {
240  while (element) {
241    if (element.tagName && element.tagName.toLowerCase() == 'head') {
242      return true;
243    }
244    try {
245      element = element.parentNode;
246    } catch (e) {
247      // Fine. the DOM has dispeared from underneath us
248      return false;
249    }
250  }
251
252  return false;
253};

265/**
266 * Types keys on the given {@code element} with a virtual keyboard. Converts
267 * special characters from the WebDriver JSON wire protocol to the appropriate
268 * {@link bot.Keyboard.Key} value.
269 *
270 * @param {!Element} element The element to type upon.
271 * @param {!Array.<string>} keys The keys to type on the element.
272 * @param {bot.Keyboard=} opt_keyboard Keyboard to use; if not provided,
273 *    constructs one.
274 * @param {boolean=} opt_persistModifiers Whether modifier keys should remain
275 *     pressed when this function ends.
276 * @see bot.action.type
277 * @see http://code.google.com/p/selenium/wiki/JsonWireProtocol
278 */
279webdriver.atoms.element.type = function(
280    element, keys, opt_keyboard, opt_persistModifiers) {
281  var persistModifierKeys = !!opt_persistModifiers;
282  function createSequenceRecord() {
283    return {persist: persistModifierKeys, keys: []};
284  }
285
286  /**
287   * @type {!Array.<{persist: boolean,
288   *                 keys: !Array.<(string|!bot.Keyboard.Key)>}>}
289   */
290  var convertedSequences = [];
291
292  /**
293   * @type {{persist: boolean,
294   *         keys: !Array.<(string|!bot.Keyboard.Key)>}}
295   */
296  var current = createSequenceRecord();
297  convertedSequences.push(current);
298
299  goog.array.forEach(keys, function(sequence) {
300    goog.array.forEach(sequence.split(''), function(key) {
301      if (isWebDriverKey(key)) {
302        var webdriverKey = webdriver.atoms.element.type.JSON_TO_KEY_MAP_[key];
303        // goog.isNull uses ==, which accepts undefined.
304        if (webdriverKey === null) {
305          // bot.action.type does not support a "null" key, so we have to
306          // terminate the entire sequence to release modifier keys. If
307          // we currently allow modifier key state to persist across key
308          // sequences, we need to inject a dummy sequence that does not
309          // persist state so every modifier key gets released.
310          convertedSequences.push(current = createSequenceRecord());
311          if (persistModifierKeys) {
312            current.persist = false;
313            convertedSequences.push(current = createSequenceRecord());
314          }
315        } else if (goog.isDef(webdriverKey)) {
316          current.keys.push(webdriverKey);
317        } else {
318          throw Error('Unsupported WebDriver key: \\u' +
319              key.charCodeAt(0).toString(16));
320        }
321      } else {
322        // Handle common aliases.
323        switch (key) {
324          case '\n':
325            current.keys.push(bot.Keyboard.Keys.ENTER);
326            break;
327          case '\t':
328            current.keys.push(bot.Keyboard.Keys.TAB);
329            break;
330          case '\b':
331            current.keys.push(bot.Keyboard.Keys.BACKSPACE);
332            break;
333          default:
334            current.keys.push(key);
335            break;
336        }
337      }
338    });
339  });
340
341  goog.array.forEach(convertedSequences, function(sequence) {
342    bot.action.type(element, sequence.keys, opt_keyboard,
343        sequence.persist);
344  });
345
346  function isWebDriverKey(c) {
347    return '\uE000' <= c && c <= '\uE03D';
348  }
349};
