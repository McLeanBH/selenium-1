28/**
29 * A pure JavaScript command executor suitable for use in a browser.
30 * @param {!Window} win The window to direct commands to.
31 * @constructor
32 * @extends {goog.Disposable}
33 * @implements {webdriver.CommandExecutor}
34 */
35webdriver.browser.CommandExecutor = function(win) {
36  goog.base(this);
37
38  /** @private {!Window} */
39  this.window_ = win;
40
41  /** @private {goog.log.Logger} */
42  this.log_ = goog.log.getLogger('webdriver.browser.CommandExecutor');
43
44  /** @private {!goog.events.EventHandler} */
45  this.handler_ = new goog.events.EventHandler(this);

47  this.listenForUnload_();

76  this.handler_.dispose();
77  if (this.loadingDeferred_) {
78    this.loadingDeferred_.reject(Error('Executor was disposed'));
79    this.loadingDeferred_ = null;
80  }

84/** @override */
85webdriver.browser.CommandExecutor.prototype.execute = function(
86    command, callback) {
87  if (this.isDisposed()) {
88    callback(Error('Executor was disposed'));
89    return;
90  }
91
92  if (this.window_.closed) {
93    callback(new bot.Error(
94        bot.ErrorCode.NO_SUCH_WINDOW, 'The window has closed'));
95    return;
96  }
97
98  var handler = webdriver.browser.CommandExecutor.COMMAND_MAP_[
99      command.getName()];
100  if (!handler) {
101    callback(new bot.Error(
102        bot.ErrorCode.UNKNOWN_COMMAND,
103        'Unsupported command: ' + command.getName()));
104    return;
105  }
106
107  var executeCommand = goog.bind(function() {
108    goog.log.info(this.log_, 'Executing ' + command.getName());
109    try {
110      var result = handler.call(this, command);
111    } catch (ex) {
112      callback(ex);
113      return;
114    }
115
116    // TODO: abort on dispose?
117    webdriver.promise.when(
118        result,
119        bot.response.createResponse,
120        bot.response.createErrorResponse).
121        then(goog.partial(callback, null));
122  }, this);
123
124  if (this.loadingDeferred_) {
125    goog.log.info(this.log_, 'Waiting for window to load');
126    this.loadingDeferred_.then(executeCommand, callback);
127  } else {
128    executeCommand();
129  }
130};

140/**
141 * Registers an unload listener on the window targeted by this instance.
142 * This must be called each time the window loads a new document.
143 * @private
144 */
145webdriver.browser.CommandExecutor.prototype.listenForUnload_ = function() {
146  if (!this.listening_ && !this.window_.closed) {
147    this.handler_.listenOnce(
148        this.window_, goog.events.EventType.UNLOAD, this.onUnload_);
149    this.listening_ = true;
150  }
151};

154/** @private */
155webdriver.browser.CommandExecutor.prototype.onUnload_ = function() {
156  this.listening_ = false;
157  goog.log.info(this.log_, 'The window has unloaded');
158  if (!this.loadingDeferred_) {
159    this.loadingDeferred_ = webdriver.promise.defer(function() {
160      throw Error('This Deferred may not be canceled');
161    });
162  }
163  this.checkWindowState_();
164};

167/** @private */
168webdriver.browser.CommandExecutor.prototype.checkWindowState_ = function() {
169  setTimeout(goog.bind(check, this),
170      webdriver.browser.CommandExecutor.STATE_CHECK_TIMEOUT_);
171
172  /** @this {webdriver.browser.CommandExecutor} */
173  function check() {
174    if (this.isDisposed()) {
175      return;  // Do nothing.
176    }
177
178    goog.log.info(this.log_, 'Checking window state');
179    this.listenForUnload_();
180
181    // Do not check the readyState if this was called because the window
182    // dispatched an unload event.
183    if (this.window_.closed ||
184        (this.window_.document &&
185         this.window_.document.readyState === 'complete')) {
186      goog.log.info(this.log_, 'Window is ready');
187      if (this.loadingDeferred_) {
188        this.loadingDeferred_.fulfill();
189        this.loadingDeferred_ = null;
190      }
191    } else {
192      this.checkWindowState_();
193    }
194  }
195};

206  var url = command.getParameter('url');

208  var win = this.window_;
209  // TODO: Only wait for page load if URL points offsite.
210  return this.executeAndWaitForLoad_(function() {

272/**
273 * Executes a function, then waits for the window to load.
274 * @param {function()} fn The function to execute.
275 * @return {!webdriver.promise.Promise} A promise that will resolve when the
276 *     window has finished loading.
277 * @private
278 */
279webdriver.browser.CommandExecutor.prototype.executeAndWaitForLoad_ =
280    function(fn) {
281  // TODO: A new page will not actually load if only the hash is changing.
282  this.loadingDeferred_ = webdriver.promise.defer(function() {
283    throw Error('This Deferred may not be canceled');
284  });
285  // Allow the command to be cancelled, but not the load wait.
286  var d = webdriver.promise.defer();
287  this.loadingDeferred_.then(d.fulfill, d.reject);
288  fn();
289  return d.promise;
290};

293/**
294 * Locates an element on the page.
295 * @param {!webdriver.Command} command The command.
296 * @return {bot.response.ResponseObject} A response object with the
297 *     encoded element.
298 * @throws {bot.Error} If the element cannot be found.
299 * @private
300 */
301webdriver.browser.CommandExecutor.prototype.findElement_ = function(
302    command) {
303  var using = command.getParameter('using');
304  var value = command.getParameter('value');
305  var locator = {};
306  locator[using] = value;
307
308  var root = this.window_.document;
309  var id = command.getParameter('id');
310  if (id) {
311    root = bot.inject.cache.getElement(
312        id[bot.inject.ELEMENT_KEY], this.window_.document);
313  }
314
315  var el = bot.locators.findElement(locator, root);
316  if (el) {
317    return bot.response.createResponse(bot.inject.wrapValue(el));
318  } else {
319    throw new bot.Error(
320        bot.ErrorCode.NO_SUCH_ELEMENT,
321        'Unable to locate element by ' + using + ': ' + value);
322  }
323};
