16/**
17 * @fileoverview Wrapping execute script to use a serialized window object.
18 *
19 */
20
21goog.provide('webdriver.atoms.inject');
22
23goog.require('bot.inject');
24goog.require('bot.inject.cache');
25
26
27/**
28 * Wrapper to allow passing a seliazed window object to executeScript.
29 *
30 * @param {!(string|Function)} fn The function to execute.
31 * @param {Array.<*>} args Array of arguments to pass to fn.
32 * @param {{WINDOW: string}=} opt_window The serialized window object to be
33 *     read from the cache.
34 * @return {string} The response object, serialized and returned in string
35 *     format.
36 */
37webdriver.atoms.inject.executeScript = function(fn, args, opt_window) {
38  return /**@type {string}*/(bot.inject.executeScript(fn, args, true,
39      webdriver.atoms.inject.getWindow(opt_window)));
40};
41
42
43/**
44 *
45 * @param {!(string|Function)} fn The function to execute.
46 * @param {Array.<*>} args Array of arguments to pass to fn.
47 * @param {number} timeout The timeout to wait up to in millis.
48 * @param {function(string)|function(!bot.response.ResponseObject)} onDone
49 *     The function to call when the given {@code fn} invokes its callback,
50 *     or when an exception or timeout occurs. This will always be called.
51 * @param {{WINDOW: string}=} opt_window The serialized window
52 *     object to be read from the cache.
53 */
54webdriver.atoms.inject.executeAsyncScript =
55    function(fn, args, timeout, onDone, opt_window) {
56  bot.inject.executeAsyncScript(
57      fn, args, timeout, onDone, true,
58      webdriver.atoms.inject.getWindow(opt_window));
59};
60
61
62/**
63 * Decodes a serialized {WINDOW: string} object using the current document's
64 * cache.
65 *
66 * @param {{WINDOW: string}=} opt_window The serialized window object to be
67 *     read from the cache. If undefined, this function will trivially return
68 *     the current window.
69 * @return {!Window} A reference to a window.
70 * @throws {bot.Error} If the serialized window cannot be found in the current
71 *     document's cache.
72 */
73webdriver.atoms.inject.getWindow = function(opt_window) {
74  var win;
75  if (opt_window) {
76    win = bot.inject.cache.getElement(opt_window[bot.inject.WINDOW_KEY]);
77  } else {
78    win = window;
79  }
80  return /**@type {!Window}*/(win);
81};
