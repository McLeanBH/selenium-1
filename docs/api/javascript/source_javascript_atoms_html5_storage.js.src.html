16/**
17 * @fileoverview Atoms for accessing HTML5 web storage maps (localStorage,
18 * sessionStorage). These storage objects store each item as a key-value
19 * mapping pair.
20 *
21 */
22

24goog.provide('bot.storage.Storage');

26goog.require('bot');
27goog.require('bot.Error');
28goog.require('bot.ErrorCode');
29goog.require('bot.html5');

32/**
33 * A factory method to create a wrapper to access the HTML5 localStorage
34 * object.
35 * Note: We are not using Closure from goog.storage,
36 * Closure uses "window" object directly, which may not always be
37 * defined (for example in firefox extensions).
38 * We use bot.window() from bot.js instead to keep track of the window or frame
39 * is currently being used for command execution. The implementation is
40 * otherwise similar to the implementation in the Closure library
41 * (goog.storage.mechansim.HTML5LocalStorage).
42 *
43 * @param {Window=} opt_window The window whose storage to access;
44 *     defaults to the main window.
45 * @return {!bot.storage.Storage} The wrapper Storage object.
46 */
47bot.storage.getLocalStorage = function(opt_window) {
48  var win = opt_window || bot.getWindow();
49
50  if (!bot.html5.isSupported(bot.html5.API.LOCAL_STORAGE, win)) {
51    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR, 'Local storage undefined');
52  }
53  var storageMap = win.localStorage;
54  return new bot.storage.Storage(storageMap);
55};

58/**
59 * A factory method to create a wrapper to access the HTML5 sessionStorage
60 * object.
61 *
62 * @param {Window=} opt_window The window whose storage to access;
63 *     defaults to the main window.
64 * @return {!bot.storage.Storage} The wrapper Storage object.
65 */
66bot.storage.getSessionStorage = function(opt_window) {
67  var win = opt_window || bot.getWindow();
68
69  if (bot.html5.isSupported(bot.html5.API.SESSION_STORAGE, win)) {
70    var storageMap = win.sessionStorage;
71    return new bot.storage.Storage(storageMap);
72  }
73  throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
74      'Session storage undefined');
75};

79/**
80 * Provides a wrapper object to the HTML5 web storage object.
81 * @constructor
82 *
83 * @param {Storage} storageMap HTML5 storage object e.g. localStorage,
84 *     sessionStorage.
85 */
86bot.storage.Storage = function(storageMap) {
87  /**
88   * Member variable to access the assigned HTML5 storage object.
89   * @private {Storage}
90   */
91  this.storageMap_ = storageMap;
92};
93
94
95/**
96 * Sets the value item of a key/value pair in the Storage object.
97 * If the value given is null, the string 'null' will be inserted
98 * instead.
99 *
100 * @param {string} key The key of the item.
101 * @param {*} value The value of the item.
102 */
103bot.storage.Storage.prototype.setItem = function(key, value) {
104  try {
105    // Note: Ideally, browsers should set a null value. But the browsers
106    // report arbitrarily. Firefox returns <null>, while Chrome reports
107    // the string "null". We are setting the value to the string "null".
108    this.storageMap_.setItem(key, value + '');
109  } catch (e) {
110    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR, e.message);
111  }
112};
113
114
115/**
116 * Returns the value item of a key in the Storage object.
117 *
118 * @param {string} key The key of the returned value.
119 * @return {?string} The mapped value if present in the storage object,
120 *     otherwise null. If a null value  was inserted for a given
121 *     key, then the string 'null' is returned.
122 */
123bot.storage.Storage.prototype.getItem = function(key) {
124  var value = this.storageMap_.getItem(key);
125  return /** @type {?string} */ (value);
126};

129/**
130 * Returns an array of keys of all keys of the Storage object.
131 *
132 * @return {Array.<string>} The array of stored keys..
133 */
134bot.storage.Storage.prototype.keySet = function() {
135  var keys = [];
136  var length = this.size();
137  for (var i = 0; i < length; i++) {
138    keys[i] = this.storageMap_.key(i);
139  }
140  return keys;
141};

144/**
145 * Removes an item with a given key.
146 *
147 * @param {string} key The key item of the key/value pair.
148 * @return {?string} The removed value if present, otherwise null.
149 */
150bot.storage.Storage.prototype.removeItem = function(key) {
151  var value = this.getItem(key);
152  this.storageMap_.removeItem(key);
153  return value;
154};
155
156
157/**
158 * Removes all items.
159 */
160bot.storage.Storage.prototype.clear = function() {
161  this.storageMap_.clear();
162};
163
164
165/**
166 * Returns the number of items in the Storage object.
167 *
168 * @return {number} The number of the key/value pairs.
169 */
170bot.storage.Storage.prototype.size = function() {
171  return this.storageMap_.length;
172};

175/**
176 * Returns the key item of the key/value pairs in the Storage object
177 * of a given index.
178 *
179 * @param {number} index The index of the key/value pair list.
180 * @return {?string} The key item of a given index.
181 */
182bot.storage.Storage.prototype.key = function(index) {
183  return this.storageMap_.key(index);
184};
