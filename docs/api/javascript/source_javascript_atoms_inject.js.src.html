59/**
60 * Converts an element to a JSON friendly value so that it can be
61 * stringified for transmission to the injector. Values are modified as
62 * follows:
63 * <ul>
64 * <li>booleans, numbers, strings, and null are returned as is</li>
65 * <li>undefined values are returned as null</li>
66 * <li>functions are returned as a string</li>
67 * <li>each element in an array is recursively processed</li>
68 * <li>DOM Elements are wrapped in object-literals as dictated by the
69 *     WebDriver wire protocol</li>
70 * <li>all other objects will be treated as hash-maps, and will be
71 *     recursively processed for any string and number key types (all
72 *     other key types are discarded as they cannot be converted to JSON).
73 * </ul>
74 *
75 * @param {*} value The value to make JSON friendly.
76 * @return {*} The JSON friendly value.
77 * @see http://code.google.com/p/selenium/wiki/JsonWireProtocol
78 */
79bot.inject.wrapValue = function(value) {
80  switch (goog.typeOf(value)) {
81    case 'string':
82    case 'number':
83    case 'boolean':
84      return value;
85
86    case 'function':
87      return value.toString();
88
89    case 'array':
90      return goog.array.map(/**@type {goog.array.ArrayLike}*/ (value),
91          bot.inject.wrapValue);
92
93    case 'object':
94      // Since {*} expands to {Object|boolean|number|string|undefined}, the
95      // JSCompiler complains that it is too broad a type for the remainder of
96      // this block where {!Object} is expected. Downcast to prevent generating
97      // a ton of compiler warnings.
98      value = /**@type {!Object}*/ (value);
99
100      // Sniff out DOM elements. We're using duck-typing instead of an
101      // instanceof check since the instanceof might not always work
102      // (e.g. if the value originated from another Firefox component)
103      if (goog.object.containsKey(value, 'nodeType') &&
104          (value['nodeType'] == goog.dom.NodeType.ELEMENT ||
105           value['nodeType'] == goog.dom.NodeType.DOCUMENT)) {
106        var ret = {};
107        ret[bot.inject.ELEMENT_KEY] =
108            bot.inject.cache.addElement(/**@type {!Element}*/ (value));
109        return ret;
110      }
111
112      // Check if this is a Window
113      if (goog.object.containsKey(value, 'document')) {
114        var ret = {};
115        ret[bot.inject.WINDOW_KEY] =
116            bot.inject.cache.addElement(/**@type{!Window}*/ (value));
117        return ret;
118      }
119
120      if (goog.isArrayLike(value)) {
121        return goog.array.map(/**@type {goog.array.ArrayLike}*/ (value),
122            bot.inject.wrapValue);
123      }
124
125      var filtered = goog.object.filter(value, function(val, key) {
126        return goog.isNumber(key) || goog.isString(key);
127      });
128      return goog.object.map(filtered, bot.inject.wrapValue);
129
130    default:  // goog.typeOf(value) == 'undefined' || 'null'
131      return null;
132  }
133};

136/**
137 * Unwraps any DOM element's encoded in the given {@code value}.
138 * @param {*} value The value to unwrap.
139 * @param {Document=} opt_doc The document whose cache to retrieve wrapped
140 *     elements from. Defaults to the current document.
141 * @return {*} The unwrapped value.
142 */
143bot.inject.unwrapValue = function(value, opt_doc) {
144  if (goog.isArray(value)) {
145    return goog.array.map(/**@type {goog.array.ArrayLike}*/ (value),
146        function(v) { return bot.inject.unwrapValue(v, opt_doc); });
147  } else if (goog.isObject(value)) {
148    if (typeof value == 'function') {
149      return value;
150    }
151
152    if (goog.object.containsKey(value, bot.inject.ELEMENT_KEY)) {
153      return bot.inject.cache.getElement(value[bot.inject.ELEMENT_KEY],
154          opt_doc);
155    }
156
157    if (goog.object.containsKey(value, bot.inject.WINDOW_KEY)) {
158      return bot.inject.cache.getElement(value[bot.inject.WINDOW_KEY],
159          opt_doc);
160    }
161
162    return goog.object.map(value, function(val) {
163      return bot.inject.unwrapValue(val, opt_doc);
164    });
165  }
166  return value;
167};

170/**
171 * Recompiles {@code fn} in the context of another window so that the
172 * correct symbol table is used when the function is executed. This
173 * function assumes the {@code fn} can be decompiled to its source using
174 * {@code Function.prototype.toString} and that it only refers to symbols
175 * defined in the target window's context.
176 *
177 * @param {!(Function|string)} fn Either the function that shold be
178 *     recompiled, or a string defining the body of an anonymous function
179 *     that should be compiled in the target window's context.
180 * @param {!Window} theWindow The window to recompile the function in.
181 * @return {!Function} The recompiled function.
182 * @private
183 */
184bot.inject.recompileFunction_ = function(fn, theWindow) {
185  if (goog.isString(fn)) {
186    return new theWindow['Function'](fn);
187  }
188  return theWindow == window ? fn : new theWindow['Function'](
189      'return (' + fn + ').apply(null,arguments);');
190};

193/**
194 * Executes an injected script. This function should never be called from
195 * within JavaScript itself. Instead, it is used from an external source that
196 * is injecting a script for execution.
197 *
198 * <p/>For example, in a WebDriver Java test, one might have:
199 * <pre><code>
200 * Object result = ((JavascriptExecutor) driver).executeScript(
201 *     "return arguments[0] + arguments[1];", 1, 2);
202 * </code></pre>
203 *
204 * <p/>Once transmitted to the driver, this command would be injected into the
205 * page for evaluation as:
206 * <pre><code>
207 * bot.inject.executeScript(
208 *     function() {return arguments[0] + arguments[1];},
209 *     [1, 2]);
210 * </code></pre>
211 *
212 * <p/>The details of how this actually gets injected for evaluation is left
213 * as an implementation detail for clients of this library.
214 *
215 * @param {!(Function|string)} fn Either the function to execute, or a string
216 *     defining the body of an anonymous function that should be executed. This
217 *     function should only contain references to symbols defined in the context
218 *     of the target window ({@code opt_window}). Any references to symbols
219 *     defined in this context will likely generate a ReferenceError.
220 * @param {Array.<*>} args An array of wrapped script arguments, as defined by
221 *     the WebDriver wire protocol.
222 * @param {boolean=} opt_stringify Whether the result should be returned as a
223 *     serialized JSON string.
224 * @param {!Window=} opt_window The window in whose context the function should
225 *     be invoked; defaults to the current window.
226 * @return {!(string|bot.response.ResponseObject)} The response object. If
227 *     opt_stringify is true, the result will be serialized and returned in
228 *     string format.
229 */
230bot.inject.executeScript = function(fn, args, opt_stringify, opt_window) {
231  var win = opt_window || bot.getWindow();
232  var ret;
233  try {
234    fn = bot.inject.recompileFunction_(fn, win);
235    var unwrappedArgs = /**@type {Object}*/ (bot.inject.unwrapValue(args,
236        win.document));
237    ret = bot.inject.wrapResponse(fn.apply(null, unwrappedArgs));
238  } catch (ex) {
239    ret = bot.inject.wrapError(ex);
240  }
241  return opt_stringify ? bot.json.stringify(ret) : ret;
242};

245/**
246 * Executes an injected script, which is expected to finish asynchronously
247 * before the given {@code timeout}. When the script finishes or an error
248 * occurs, the given {@code onDone} callback will be invoked. This callback
249 * will have a single argument, a {@link bot.response.ResponseObject} object.
250 *
251 * The script signals its completion by invoking a supplied callback given
252 * as its last argument. The callback may be invoked with a single value.
253 *
254 * The script timeout event will be scheduled with the provided window,
255 * ensuring the timeout is synchronized with that window's event queue.
256 * Furthermore, asynchronous scripts do not work across new page loads; if an
257 * "unload" event is fired on the window while an asynchronous script is
258 * pending, the script will be aborted and an error will be returned.
259 *
260 * Like {@code bot.inject.executeScript}, this function should only be called
261 * from an external source. It handles wrapping and unwrapping of input/output
262 * values.
263 *
264 * @param {(!Function|string)} fn Either the function to execute, or a string
265 *     defining the body of an anonymous function that should be executed. This
266 *     function should only contain references to symbols defined in the context
267 *     of the target window ({@code opt_window}). Any references to symbols
268 *     defined in this context will likely generate a ReferenceError.
269 * @param {Array.<*>} args An array of wrapped script arguments, as defined by
270 *     the WebDriver wire protocol.
271 * @param {number} timeout The amount of time, in milliseconds, the script
272 *     should be permitted to run; must be non-negative.
273 * @param {function(string)|function(!bot.response.ResponseObject)} onDone
274 *     The function to call when the given {@code fn} invokes its callback,
275 *     or when an exception or timeout occurs. This will always be called.
276 * @param {boolean=} opt_stringify Whether the result should be returned as a
277 *     serialized JSON string.
278 * @param {!Window=} opt_window The window to synchronize the script with;
279 *     defaults to the current window.
280 */
281bot.inject.executeAsyncScript = function(fn, args, timeout, onDone,
282                                         opt_stringify, opt_window) {
283  var win = opt_window || window;
284  var timeoutId;
285  var responseSent = false;
286
287  function sendResponse(status, value) {
288    if (!responseSent) {
289      if (win.removeEventListener) {
290        win.removeEventListener('unload', onunload, true);
291      } else {
292        win.detachEvent('onunload', onunload);
293      }
294
295      win.clearTimeout(timeoutId);
296      if (status != bot.ErrorCode.SUCCESS) {
297        var err = new bot.Error(status, value.message || value + '');
298        err.stack = value.stack;
299        value = bot.inject.wrapError(err);
300      } else {
301        value = bot.inject.wrapResponse(value);
302      }
303      onDone(opt_stringify ? bot.json.stringify(value) : value);
304      responseSent = true;
305    }
306  }
307  var sendError = goog.partial(sendResponse, bot.ErrorCode.UNKNOWN_ERROR);
308
309  if (win.closed) {
310    sendError('Unable to execute script; the target window is closed.');
311    return;
312  }
313
314  fn = bot.inject.recompileFunction_(fn, win);
315
316  args = /** @type {Array.<*>} */ (bot.inject.unwrapValue(args, win.document));
317  args.push(goog.partial(sendResponse, bot.ErrorCode.SUCCESS));
318
319  if (win.addEventListener) {
320    win.addEventListener('unload', onunload, true);
321  } else {
322    win.attachEvent('onunload', onunload);
323  }
324
325  var startTime = goog.now();
326  try {
327    fn.apply(win, args);
328
329    // Register our timeout *after* the function has been invoked. This will
330    // ensure we don't timeout on a function that invokes its callback after
331    // a 0-based timeout.
332    timeoutId = win.setTimeout(function() {
333      sendResponse(bot.ErrorCode.SCRIPT_TIMEOUT,
334                   Error('Timed out waiting for asyncrhonous script result ' +
335                         'after ' + (goog.now() - startTime) + ' ms'));
336    }, Math.max(0, timeout));
337  } catch (ex) {
338    sendResponse(ex.code || bot.ErrorCode.UNKNOWN_ERROR, ex);
339  }
340
341  function onunload() {
342    sendResponse(bot.ErrorCode.UNKNOWN_ERROR,
343        Error('Detected a page unload event; asynchronous script ' +
344              'execution does not work across page loads.'));
345  }
346};

385/**
386 * The property key used to store the element cache on the DOCUMENT node
387 * when it is injected into the page. Since compiling each browser atom results
388 * in a different symbol table, we must use this known key to access the cache.
389 * This ensures the same object is used between injections of different atoms.
390 * @private {string}
391 * @const
392 */
393bot.inject.cache.CACHE_KEY_ = '$wdc_';
394
395
396/**
397 * The prefix for each key stored in an cache.
398 * @type {string}
399 * @const
400 */
401bot.inject.cache.ELEMENT_KEY_PREFIX = ':wdc:';
402
403
404/**
405 * Retrieves the cache object for the given window. Will initialize the cache
406 * if it does not yet exist.
407 * @param {Document=} opt_doc The document whose cache to retrieve. Defaults to
408 *     the current document.
409 * @return {Object.<string, (Element|Window)>} The cache object.
410 * @private
411 */
412bot.inject.cache.getCache_ = function(opt_doc) {
413  var doc = opt_doc || document;
414  var cache = doc[bot.inject.cache.CACHE_KEY_];
415  if (!cache) {
416    cache = doc[bot.inject.cache.CACHE_KEY_] = {};
417    // Store the counter used for generated IDs in the cache so that it gets
418    // reset whenever the cache does.
419    cache.nextId = goog.now();
420  }
421  // Sometimes the nextId does not get initialized and returns NaN
422  // TODO: Generate UID on the fly instead.
423  if (!cache.nextId) {
424    cache.nextId = goog.now();
425  }
426  return cache;
427};

430/**
431 * Adds an element to its ownerDocument's cache.
432 * @param {(Element|Window)} el The element or Window object to add.
433 * @return {string} The key generated for the cached element.
434 */
435bot.inject.cache.addElement = function(el) {
436  // Check if the element already exists in the cache.
437  var cache = bot.inject.cache.getCache_(el.ownerDocument);
438  var id = goog.object.findKey(cache, function(value) {
439    return value == el;
440  });
441  if (!id) {
442    id = bot.inject.cache.ELEMENT_KEY_PREFIX + cache.nextId++;
443    cache[id] = el;
444  }
445  return id;
446};

449/**
450 * Retrieves an element from the cache. Will verify that the element is
451 * still attached to the DOM before returning.
452 * @param {string} key The element's key in the cache.
453 * @param {Document=} opt_doc The document whose cache to retrieve the element
454 *     from. Defaults to the current document.
455 * @return {Element|Window} The cached element.
456 */
457bot.inject.cache.getElement = function(key, opt_doc) {
458  key = decodeURIComponent(key);
459  var doc = opt_doc || document;
460  var cache = bot.inject.cache.getCache_(doc);
461  if (!goog.object.containsKey(cache, key)) {
462    // Throw STALE_ELEMENT_REFERENCE instead of NO_SUCH_ELEMENT since the
463    // key may have been defined by a prior document's cache.
464    throw new bot.Error(bot.ErrorCode.STALE_ELEMENT_REFERENCE,
465        'Element does not exist in cache');
466  }
467
468  var el = cache[key];
469
470  // If this is a Window check if it's closed
471  if (goog.object.containsKey(el, 'setInterval')) {
472    if (el.closed) {
473      delete cache[key];
474      throw new bot.Error(bot.ErrorCode.NO_SUCH_WINDOW,
475          'Window has been closed.');
476    }
477    return el;
478  }
479
480  // Make sure the element is still attached to the DOM before returning.
481  var node = el;
482  while (node) {
483    if (node == doc.documentElement) {
484      return el;
485    }
486    node = node.parentNode;
487  }
488  delete cache[key];
489  throw new bot.Error(bot.ErrorCode.STALE_ELEMENT_REFERENCE,
490      'Element is no longer attached to the DOM');
491};
