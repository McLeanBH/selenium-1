20/**
21 * A function call expression.
22 *
23 * @constructor
24 * @extends {wgxpath.Expr}
25 * @param {!wgxpath.FunctionCall.Func} func Function.
26 * @param {!Array.<!wgxpath.Expr>} args Arguments to the function.
27 */
28wgxpath.FunctionCall = function(func, args) {
29  // Check the provided arguments match the function parameters.
30  if (args.length < func.minArgs_) {
31    throw new Error('Function ' + func.name_ + ' expects at least' +
32        func.minArgs_ + ' arguments, ' + args.length + ' given');
33  }
34  if (!goog.isNull(func.maxArgs_) && args.length > func.maxArgs_) {
35    throw new Error('Function ' + func.name_ + ' expects at most ' +
36        func.maxArgs_ + ' arguments, ' + args.length + ' given');
37  }
38  if (func.nodesetsRequired_) {
39    goog.array.forEach(args, function(arg, i) {
40      if (arg.getDataType() != wgxpath.DataType.NODESET) {
41        throw new Error('Argument ' + i + ' to function ' + func.name_ +
42            ' is not of type Nodeset: ' + arg);
43      }
44    });
45  }
46  wgxpath.Expr.call(this, func.dataType_);
47
48  /**
49   * @type {!wgxpath.FunctionCall.Func}
50   * @private
51   */
52  this.func_ = func;
53
54  /**
55   * @type {!Array.<!wgxpath.Expr>}
56   * @private
57   */
58  this.args_ = args;
59
60  this.setNeedContextPosition(func.needContextPosition_ ||
61      goog.array.some(args, function(arg) {
62        return arg.doesNeedContextPosition();
63      }));
64  this.setNeedContextNode(
65      (func.needContextNodeWithoutArgs_ && !args.length) ||
66      (func.needContextNodeWithArgs_ && !!args.length) ||
67      goog.array.some(args, function(arg) {
68        return arg.doesNeedContextNode();
69      }));
70};
71goog.inherits(wgxpath.FunctionCall, wgxpath.Expr);

74/**
75 * @override
76 */
77wgxpath.FunctionCall.prototype.evaluate = function(ctx) {
78  var result = this.func_.evaluate_.apply(null,
79      goog.array.concat(ctx, this.args_));
80  return /** @type {!(string|boolean|number|wgxpath.NodeSet)} */ (result);
81};

84/**
85 * @override
86 */
87wgxpath.FunctionCall.prototype.toString = function() {
88  var text = 'Function: ' + this.func_;
89  if (this.args_.length) {
90    var args = goog.array.reduce(this.args_, function(prev, curr) {
91      return prev + wgxpath.Expr.indent(curr);
92    }, 'Arguments:');
93    text += wgxpath.Expr.indent(args);
94  }
95  return text;
96};

198/**
199 * Constructs a Func and maps its name to it.
200 *
201 * @param {string} name Name of the function.
202 * @param {wgxpath.DataType} dataType Datatype of the function return value.
203 * @param {boolean} needContextPosition Whether the function needs a context
204 *     position.
205 * @param {boolean} needContextNodeWithoutArgs Whether the function needs a
206 *     context node when not given arguments.
207 * @param {boolean} needContextNodeWithArgs Whether the function needs a context
208 *     node when the function is given arguments.
209 * @param {function(!wgxpath.Context, ...[!wgxpath.Expr]):*} evaluate
210 *     Evaluates the function in a context with any number of expression
211 *     arguments.
212 * @param {number} minArgs Minimum number of arguments accepted by the function.
213 * @param {?number=} opt_maxArgs Maximum number of arguments accepted by the
214 *     function; null means there is no max; defaults to minArgs.
215 * @param {boolean=} opt_nodesetsRequired Whether the args must be nodesets.
216 * @return {!wgxpath.FunctionCall.Func} The function created.
217 * @private
218 */
219wgxpath.FunctionCall.createFunc_ = function(name, dataType,
220    needContextPosition, needContextNodeWithoutArgs, needContextNodeWithArgs,
221    evaluate, minArgs, opt_maxArgs, opt_nodesetsRequired) {
222  if (wgxpath.FunctionCall.nameToFuncMap_.hasOwnProperty(name)) {
223    throw new Error('Function already created: ' + name + '.');
224  }
225  var func = new wgxpath.FunctionCall.Func_(name, dataType,
226      needContextPosition, needContextNodeWithoutArgs, needContextNodeWithArgs,
227      evaluate, minArgs, opt_maxArgs, opt_nodesetsRequired);
228  func = /** @type {!wgxpath.FunctionCall.Func} */ (func);
229  wgxpath.FunctionCall.nameToFuncMap_[name] = func;
230  return func;
231};
