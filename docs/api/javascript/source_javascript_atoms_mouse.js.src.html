39/**
40 * A mouse that provides atomic mouse actions. This mouse currently only
41 * supports having one button pressed at a time.
42 * @param {bot.Mouse.State=} opt_state The mouse's initial state.
43 * @param {bot.Device.ModifiersState=} opt_modifiersState State of the keyboard.
44 * @param {bot.Device.EventEmitter=} opt_eventEmitter An object that should be
45 *     used to fire events.
46 * @constructor
47 * @extends {bot.Device}
48 */
49bot.Mouse = function(opt_state, opt_modifiersState, opt_eventEmitter) {
50  goog.base(this, opt_modifiersState, opt_eventEmitter);
51
52  /** @private {?bot.Mouse.Button} */
53  this.buttonPressed_ = null;
54
55  /** @private {Element} */
56  this.elementPressed_ = null;
57
58  /** @private {!goog.math.Coordinate} */
59  this.clientXY_ = new goog.math.Coordinate(0, 0);
60
61  /** @private {boolean} */
62  this.nextClickIsDoubleClick_ = false;
63
64  /**
65   * Whether this Mouse has ever explicitly interacted with any element.
66   * @private {boolean}
67   */
68  this.hasEverInteracted_ = false;
69
70  if (opt_state) {
71    this.buttonPressed_ = opt_state.buttonPressed;
72
73    try {
74      if (bot.dom.isElement(opt_state.elementPressed)) {
75        this.elementPressed_ = opt_state.elementPressed;
76      }
77    } catch (ignored) {
78      this.buttonPressed_ = null;
79    }
80
81    this.clientXY_ = opt_state.clientXY;
82    this.nextClickIsDoubleClick_ = opt_state.nextClickIsDoubleClick;
83    this.hasEverInteracted_ = opt_state.hasEverInteracted;
84
85    try {
86      if (bot.dom.isElement(opt_state.element)) {
87        this.setElement(/** @type {!Element} */ (opt_state.element));
88      }
89    } catch (ignored) {
90      this.buttonPressed_ = null;
91    }
92  }
93};
94goog.inherits(bot.Mouse, bot.Device);

128/**
129 * Maps mouse events to an array of button argument value for each mouse button.
130 * The array is indexed by the bot.Mouse.Button values. It encodes this table,
131 * where each cell contains the (left/middle/right/none) button values.
132 * <pre>
133 *               click/    mouseup/   mouseout/  mousemove  contextmenu
134 *               dblclick  mousedown  mouseover
135 * IE_DOC_PRE9   0 0 0 X   1 4 2 X    0 0 0 0    1 4 2 0    X X 0 X
136 * WEBKIT/IE9    0 1 2 X   0 1 2 X    0 1 2 0    0 1 2 0    X X 2 X
137 * GECKO/OPERA   0 1 2 X   0 1 2 X    0 0 0 0    0 0 0 0    X X 2 X
138 * </pre>
139 * @private {!Object.<bot.events.EventType, !Array.<?number>>}
140 * @const
141 */
142bot.Mouse.MOUSE_BUTTON_VALUE_MAP_ = (function() {
143  // EventTypes can safely be used as keys without collisions in a JS Object,
144  // because its toString method returns a unique string (the event type name).
145  var buttonValueMap = {};
146  if (bot.userAgent.IE_DOC_PRE9) {
147    buttonValueMap[bot.events.EventType.CLICK] = [0, 0, 0, null];
148    buttonValueMap[bot.events.EventType.CONTEXTMENU] = [null, null, 0, null];
149    buttonValueMap[bot.events.EventType.MOUSEUP] = [1, 4, 2, null];
150    buttonValueMap[bot.events.EventType.MOUSEOUT] = [0, 0, 0, 0];
151    buttonValueMap[bot.events.EventType.MOUSEMOVE] = [1, 4, 2, 0];
152  } else if (goog.userAgent.WEBKIT || bot.userAgent.IE_DOC_9) {
153    buttonValueMap[bot.events.EventType.CLICK] = [0, 1, 2, null];
154    buttonValueMap[bot.events.EventType.CONTEXTMENU] = [null, null, 2, null];
155    buttonValueMap[bot.events.EventType.MOUSEUP] = [0, 1, 2, null];
156    buttonValueMap[bot.events.EventType.MOUSEOUT] = [0, 1, 2, 0];
157    buttonValueMap[bot.events.EventType.MOUSEMOVE] = [0, 1, 2, 0];
158  } else {
159    buttonValueMap[bot.events.EventType.CLICK] = [0, 1, 2, null];
160    buttonValueMap[bot.events.EventType.CONTEXTMENU] = [null, null, 2, null];
161    buttonValueMap[bot.events.EventType.MOUSEUP] = [0, 1, 2, null];
162    buttonValueMap[bot.events.EventType.MOUSEOUT] = [0, 0, 0, 0];
163    buttonValueMap[bot.events.EventType.MOUSEMOVE] = [0, 0, 0, 0];
164  }
165
166  if (bot.userAgent.IE_DOC_10) {
167    buttonValueMap[bot.events.EventType.MSPOINTERDOWN] =
168        buttonValueMap[bot.events.EventType.MOUSEUP];
169    buttonValueMap[bot.events.EventType.MSPOINTERUP] =
170        buttonValueMap[bot.events.EventType.MOUSEUP];
171    buttonValueMap[bot.events.EventType.MSPOINTERMOVE] = [-1, -1, -1, -1];
172    buttonValueMap[bot.events.EventType.MSPOINTEROUT] =
173        buttonValueMap[bot.events.EventType.MSPOINTERMOVE];
174    buttonValueMap[bot.events.EventType.MSPOINTEROVER] =
175        buttonValueMap[bot.events.EventType.MSPOINTERMOVE];
176  }
177
178  buttonValueMap[bot.events.EventType.DBLCLICK] =
179      buttonValueMap[bot.events.EventType.CLICK];
180  buttonValueMap[bot.events.EventType.MOUSEDOWN] =
181      buttonValueMap[bot.events.EventType.MOUSEUP];
182  buttonValueMap[bot.events.EventType.MOUSEOVER] =
183      buttonValueMap[bot.events.EventType.MOUSEOUT];
184  return buttonValueMap;
185})();

203/**
204 * Attempts to fire a mousedown event and then returns whether or not the
205 * element should receive focus as a result of the mousedown.
206 *
207 * @return {boolean} Whether to focus on the element after the mousedown.
208 * @private
209 */
210bot.Mouse.prototype.fireMousedown_ = function() {
211  // On some browsers, a mouse down event on an OPTION or SELECT element cause
212  // the SELECT to open, blocking further JS execution. This is undesirable,
213  // and so needs to be detected. We always focus in this case.
214  // TODO: This is a nasty way to avoid locking the browser
215  var isFirefox3 = goog.userAgent.GECKO && !bot.userAgent.isProductVersion(4);
216  var blocksOnMousedown = (goog.userAgent.WEBKIT || isFirefox3) &&
217      (bot.dom.isElement(this.getElement(), goog.dom.TagName.OPTION) ||
218       bot.dom.isElement(this.getElement(), goog.dom.TagName.SELECT));
219  if (blocksOnMousedown) {
220    return true;
221  }
222
223  // On some browsers, if the mousedown event handler makes a focus() call to
224  // change the active element, this preempts the focus that would happen by
225  // default on the mousedown, so we should not explicitly focus in this case.
226  var beforeActiveElement;
227  var mousedownCanPreemptFocus = goog.userAgent.GECKO || goog.userAgent.IE;
228  if (mousedownCanPreemptFocus) {
229    beforeActiveElement = bot.dom.getActiveElement(this.getElement());
230  }
231  var performFocus = this.fireMouseEvent_(bot.events.EventType.MOUSEDOWN);
232  if (performFocus && mousedownCanPreemptFocus &&
233      beforeActiveElement != bot.dom.getActiveElement(this.getElement())) {
234    return false;
235  }
236  return performFocus;
237};

240/**
241 * Press a mouse button on an element that the mouse is interacting with.
242 *
243 * @param {!bot.Mouse.Button} button Button.
244*/
245bot.Mouse.prototype.pressButton = function(button) {
246  if (!goog.isNull(this.buttonPressed_)) {
247    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
248        'Cannot press more then one button or an already pressed button.');
249  }
250  this.buttonPressed_ = button;
251  this.elementPressed_ = this.getElement();
252
253  var performFocus = this.fireMousedown_();
254  if (performFocus) {
255    if (bot.userAgent.IE_DOC_10 &&
256        this.buttonPressed_ == bot.Mouse.Button.LEFT &&
257        bot.dom.isElement(this.elementPressed_, goog.dom.TagName.OPTION)) {
258      this.fireMSPointerEvent(bot.events.EventType.MSGOTPOINTERCAPTURE,
259          this.clientXY_, 0, bot.Device.MOUSE_MS_POINTER_ID,
260          MSPointerEvent.MSPOINTER_TYPE_MOUSE, true);
261    }
262    this.focusOnElement();
263  }
264};

267/**
268 * Releases the pressed mouse button. Throws exception if no button pressed.
269 *
270 */
271bot.Mouse.prototype.releaseButton = function() {
272  if (goog.isNull(this.buttonPressed_)) {
273    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
274        'Cannot release a button when no button is pressed.');
275  }
276
277  this.maybeToggleOption();
278  this.fireMouseEvent_(bot.events.EventType.MOUSEUP);
279
280  // TODO: Middle button can also trigger click.
281  if (this.buttonPressed_ == bot.Mouse.Button.LEFT &&
282      this.getElement() == this.elementPressed_) {
283    if (!(bot.userAgent.WINDOWS_PHONE &&
284        bot.dom.isElement(this.elementPressed_, goog.dom.TagName.OPTION))) {
285      this.clickElement(this.clientXY_,
286          this.getButtonValue_(bot.events.EventType.CLICK));
287    }
288    this.maybeDoubleClickElement_();
289    if (bot.userAgent.IE_DOC_10 &&
290        this.buttonPressed_ == bot.Mouse.Button.LEFT &&
291        bot.dom.isElement(this.elementPressed_, goog.dom.TagName.OPTION)) {
292      this.fireMSPointerEvent(bot.events.EventType.MSLOSTPOINTERCAPTURE,
293          new goog.math.Coordinate(0, 0), 0, bot.Device.MOUSE_MS_POINTER_ID,
294          MSPointerEvent.MSPOINTER_TYPE_MOUSE, false);
295    }
296  // TODO: In Linux, this fires after mousedown event.
297  } else if (this.buttonPressed_ == bot.Mouse.Button.RIGHT) {
298    this.fireMouseEvent_(bot.events.EventType.CONTEXTMENU);
299  }
300  bot.Device.clearPointerMap();
301  this.buttonPressed_ = null;
302  this.elementPressed_ = null;
303};

306/**
307 * A helper function to fire mouse double click events.
308 *
309 * @private
310 */
311bot.Mouse.prototype.maybeDoubleClickElement_ = function() {
312  // Trigger an additional double click event if it is the second click.
313  if (this.nextClickIsDoubleClick_) {
314    this.fireMouseEvent_(bot.events.EventType.DBLCLICK);
315  }
316  this.nextClickIsDoubleClick_ = !this.nextClickIsDoubleClick_;

320/**
321 * Given a coordinates (x,y) related to an element, move mouse to (x,y) of the
322 * element. The top-left point of the element is (0,0).
323 *
324 * @param {!Element} element The destination element.
325 * @param {!goog.math.Coordinate} coords Mouse position related to the target.
326 */
327bot.Mouse.prototype.move = function(element, coords) {
328  // If the element is interactable at the start of the move, it receives the
329  // full event sequence, even if hidden by an element mid sequence.
330  var toElemWasInteractable = bot.dom.isInteractable(element);
331
332  var rect = bot.dom.getClientRect(element);
333  this.clientXY_.x = coords.x + rect.left;
334  this.clientXY_.y = coords.y + rect.top;
335  var fromElement = this.getElement();
336
337  if (element != fromElement) {
338    // If the window of fromElement is closed, set fromElement to null as a flag
339    // to skip the mouseout event and so relatedTarget of the mouseover is null.
340    try {
341      if (goog.dom.getWindow(goog.dom.getOwnerDocument(fromElement)).closed) {
342        fromElement = null;
343      }
344    } catch (ignore) {
345      // Sometimes accessing a window that no longer exists causes an error.
346      fromElement = null;
347    }
348
349    if (fromElement) {
350      // For the first mouse interaction on a page, if the mouse was over the
351      // browser window, the browser will pass null as the relatedTarget for the
352      // mouseover event. For subsequent interactions, it will pass the
353      // last-focused element. Unfortunately, we don't have anywhere to keep the
354      // state of which elements have been focused across Mouse instances, so we
355      // treat every Mouse initially positioned over the documentElement or body
356      // as if it's on a new page. Accordingly, for complex actions (e.g.
357      // drag-and-drop), a single Mouse instance should be used for the whole
358      // action, to ensure the correct relatedTargets are fired for any events.
359      var isRoot = fromElement === bot.getDocument().documentElement ||
360                   fromElement === bot.getDocument().body;
361      fromElement = (!this.hasEverInteracted_ && isRoot) ? null : fromElement;
362      this.fireMouseEvent_(bot.events.EventType.MOUSEOUT, element);
363    }
364    this.setElement(element);
365
366    // All browsers except IE fire the mouseover before the mousemove.
367    if (!goog.userAgent.IE) {
368      this.fireMouseEvent_(bot.events.EventType.MOUSEOVER, fromElement, null,
369          toElemWasInteractable);
370    }
371  }
372
373  this.fireMouseEvent_(bot.events.EventType.MOUSEMOVE, null, null,
374      toElemWasInteractable);
375
376  // IE fires the mouseover event after the mousemove.
377  if (goog.userAgent.IE && element != fromElement) {
378    this.fireMouseEvent_(bot.events.EventType.MOUSEOVER, fromElement, null,
379        toElemWasInteractable);
380  }
381
382  this.nextClickIsDoubleClick_ = false;
383};

386/**
387 * Scrolls the wheel of the mouse by the given number of ticks, where a positive
388 * number indicates a downward scroll and a negative is upward scroll.
389 *
390 * @param {number} ticks Number of ticks to scroll the mouse wheel.
391 */
392bot.Mouse.prototype.scroll = function(ticks) {
393  if (ticks == 0) {
394    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
395        'Must scroll a non-zero number of ticks.');
396  }
397
398  // The wheelDelta value for a single up-tick of the mouse wheel is 120, and
399  // a single down-tick is -120. The deltas in pixels (which is only relevant
400  // for Firefox) appears to be -57 and 57, respectively.
401  var wheelDelta = ticks > 0 ? -120 : 120;
402  var pixelDelta = ticks > 0 ? 57 : -57;
403
404  // Browsers fire a separate event (or pair of events in Gecko) for each tick.
405  for (var i = 0; i < Math.abs(ticks); i++) {
406    this.fireMouseEvent_(bot.events.EventType.MOUSEWHEEL, null, wheelDelta);
407    if (goog.userAgent.GECKO) {
408      this.fireMouseEvent_(bot.events.EventType.MOUSEPIXELSCROLL, null,
409                           pixelDelta);
410    }
411  }
412};

415/**
416 * A helper function to fire mouse events.
417 *
418 * @param {bot.events.EventType} type Event type.
419 * @param {Element=} opt_related The related element of this event.
420 * @param {?number=} opt_wheelDelta The wheel delta value for the event.
421 * @param {boolean=} opt_force Whether the event should be fired even if the
422 *     element is not interactable.
423 * @return {boolean} Whether the event fired successfully or was cancelled.
424 * @private
425 */
426bot.Mouse.prototype.fireMouseEvent_ = function(type, opt_related,
427                                               opt_wheelDelta, opt_force) {
428  this.hasEverInteracted_ = true;
429  if (bot.userAgent.IE_DOC_10) {
430    var msPointerEvent = bot.Mouse.MOUSE_EVENT_MAP_[type];
431    if (msPointerEvent) {
432      // The pointerId for mouse events is always 1 and the mouse event is never
433      // fired if the MSPointer event fails.
434      if (!this.fireMSPointerEvent(msPointerEvent, this.clientXY_,
435          this.getButtonValue_(msPointerEvent), bot.Device.MOUSE_MS_POINTER_ID,
436          MSPointerEvent.MSPOINTER_TYPE_MOUSE, /* isPrimary */ true,
437          opt_related, opt_force)) {
438        return false;
439      }
440    }
441  }
442  return this.fireMouseEvent(type, this.clientXY_,
443      this.getButtonValue_(type), opt_related, opt_wheelDelta, opt_force);
444};

447/**
448 * Given an event type and a mouse button, sets the mouse button value used
449 * for that event on the current browser. The mouse button value is 0 for any
450 * event not covered by bot.Mouse.MOUSE_BUTTON_VALUE_MAP_.
451 *
452 * @param {bot.events.EventType} eventType Type of mouse event.
453 * @return {number} The mouse button ID value to the current browser.
454 * @private
455*/
456bot.Mouse.prototype.getButtonValue_ = function(eventType) {
457  if (!(eventType in bot.Mouse.MOUSE_BUTTON_VALUE_MAP_)) {
458    return 0;
459  }
460
461  var buttonIndex = goog.isNull(this.buttonPressed_) ?
462      bot.Mouse.NO_BUTTON_VALUE_INDEX_ : this.buttonPressed_;
463  var buttonValue = bot.Mouse.MOUSE_BUTTON_VALUE_MAP_[eventType][buttonIndex];
464  if (goog.isNull(buttonValue)) {
465    throw new bot.Error(bot.ErrorCode.UNKNOWN_ERROR,
466        'Event does not permit the specified mouse button.');
467  }
468  return buttonValue;
469};

472/**
473 * Serialize the current state of the mouse.
474 * @return {!bot.Mouse.State} The current mouse state.
475 */
476bot.Mouse.prototype.getState = function() {
477  var state = {};
478  state.buttonPressed = this.buttonPressed_;
479  state.elementPressed = this.elementPressed_;
480  state.clientXY = this.clientXY_;
481  state.nextClickIsDoubleClick = this.nextClickIsDoubleClick_;
482  state.hasEverInteracted = this.hasEverInteracted_;
