27/**
28 * Returns a property, with a standardized color if it contains a
29 * convertible color.
30 * @param {string} propertyName Name of the CSS property in selector-case.
31 * @param {string} propertyValue The value of the CSS property.
32 * @return {string} The value, in a standardized format
33 *    if it is a color property.
34 */
35bot.color.standardizeColor = function(propertyName, propertyValue) {
36  if (bot.color.isColorProperty(propertyName) &&
37      bot.color.isConvertibleColor(propertyValue)) {
38    return bot.color.standardizeToRgba_(propertyValue);
39  }
40  return propertyValue;
41};

44/**
45 * Returns a color in RGBA format - rgba(r, g, b, a).
46 * @param {string} propertyValue The value of the CSS property.
47 * @return {string} The value, in RGBA format.
48 * @private
49 */
50bot.color.standardizeToRgba_ = function(propertyValue) {
51  var rgba = bot.color.parseRgbaColor(propertyValue);
52  if (!rgba.length) {
53    rgba = bot.color.convertToRgba_(propertyValue);
54    bot.color.addAlphaIfNecessary_(rgba);
55  }
56  if (rgba.length != 4) {
57    return propertyValue;
58  }
59  return bot.color.toRgbaStyle_(rgba);
60};

63/**
64 * Coverts a color to RGBA.
65 * @param {string} propertyValue The value of the CSS property.
66 * @return {!Array.<number>} array containing [r, g, b, a]
67 *  with r, g, b as ints in [0, 255] and a as a float in [0, 1].
68 * @private
69 */
70bot.color.convertToRgba_ = function(propertyValue) {
71  var rgba = bot.color.parseRgbColor_(propertyValue);
72  if (rgba.length) {
73    return rgba;
74  }
75  var hex = goog.color.names[propertyValue.toLowerCase()];
76  hex = (!hex) ? bot.color.prependHashIfNecessary_(propertyValue) : hex;
77  if (bot.color.isValidHexColor_(hex)) {
78    rgba = bot.color.hexToRgb(bot.color.normalizeHex(hex));
79    if (rgba.length) {
80      return rgba;
81    }
82  }
83  return [];
84};

87/**
88 * Determines if the given string is a color that can be converted to RGBA.
89 * Browsers can return colors in the following formats:
90 * RGB, RGBA, Hex, NamedColor
91 * So only those are supported by this module and therefore considered
92 * convertible.
93 *
94 * @param {string} str Potential color string.
95 * @return {boolean} True if str is in a format that can be converted to RGBA.
96 */
97bot.color.isConvertibleColor = function(str) {
98  return !!(bot.color.isValidHexColor_(
99      bot.color.prependHashIfNecessary_(str)) ||
100      bot.color.parseRgbColor_(str).length ||
101      goog.color.names && goog.color.names[str.toLowerCase()] ||
102      bot.color.parseRgbaColor(str).length
103  );
104};

139/**
140 * Regular expression for extracting the digits in a hex color triplet.
141 * @private {!RegExp}
142 * @const
143 */
144bot.color.HEX_TRIPLET_RE_ = /#([0-9a-fA-F])([0-9a-fA-F])([0-9a-fA-F])/;

147/**
148 * Normalize an hex representation of a color
149 * @param {string} hexColor an hex color string.
150 * @return {string} hex color in the format '#rrggbb' with all lowercase
151 *     literals.
152 */
153bot.color.normalizeHex = function(hexColor) {
154  if (!bot.color.isValidHexColor_(hexColor)) {
155    throw Error("'" + hexColor + "' is not a valid hex color");
156  }
157  if (hexColor.length == 4) { // of the form #RGB
158    hexColor = hexColor.replace(bot.color.HEX_TRIPLET_RE_, '#$1$1$2$2$3$3');
159  }
160  return hexColor.toLowerCase();
161};

164/**
165 * Converts a hex representation of a color to RGB.
166 * @param {string} hexColor Color to convert.
167 * @return {!Array} array containing [r, g, b] as ints in [0, 255].
168 */
169bot.color.hexToRgb = function(hexColor) {
170  hexColor = bot.color.normalizeHex(hexColor);
171  var r = parseInt(hexColor.substr(1, 2), 16);
172  var g = parseInt(hexColor.substr(3, 2), 16);
173  var b = parseInt(hexColor.substr(5, 2), 16);
174
175  return [r, g, b];
176};

230/**
231 * Attempts to parse a string as an rgba color.  We expect strings of the
232 * format '(r, g, b, a)', or 'rgba(r, g, b, a)', where r, g, b are ints in
233 * [0, 255] and a is a float in [0, 1].
234 * @param {string} str String to check.
235 * @return {!Array.<number>} the integers [r, g, b, a] for valid colors or the
236 *     empty array for invalid colors.
237 */
238bot.color.parseRgbaColor = function(str) {
239  // Each component is separate (rather than using a repeater) so we can
240  // capture the match. Also, we explicitly set each component to be either 0,
241  // or start with a non-zero, to prevent octal numbers from slipping through.
242  var regExpResultArray = str.match(bot.color.RGBA_COLOR_RE_);
243  if (regExpResultArray) {
244    var r = Number(regExpResultArray[1]);
245    var g = Number(regExpResultArray[2]);
246    var b = Number(regExpResultArray[3]);
247    var a = Number(regExpResultArray[4]);
248    if (r >= 0 && r <= 255 &&
249        g >= 0 && g <= 255 &&
250        b >= 0 && b <= 255 &&
251        a >= 0 && a <= 1) {
252      return [r, g, b, a];
253    }
254  }
255  return [];
256};

269/**
270 * Attempts to parse a string as an rgb color.  We expect strings of the format
271 * '(r, g, b)', or 'rgb(r, g, b)', where each color component is an int in
272 * [0, 255].
273 * @param {string} str String to check.
274 * @return {!Array.<number>} the integers [r, g, b] for valid colors or the
275 *     empty array for invalid colors.
276 * @private
277 */
278bot.color.parseRgbColor_ = function(str) {
279  // Each component is separate (rather than using a repeater) so we can
280  // capture the match. Also, we explicitly set each component to be either 0,
281  // or start with a non-zero, to prevent octal numbers from slipping through.
282  var regExpResultArray = str.match(bot.color.RGB_COLOR_RE_);
283  if (regExpResultArray) {
284    var r = Number(regExpResultArray[1]);
285    var g = Number(regExpResultArray[2]);
286    var b = Number(regExpResultArray[3]);
287    if (r >= 0 && r <= 255 &&
288        g >= 0 && g <= 255 &&
289        b >= 0 && b <= 255) {
290      return [r, g, b];
291    }
292  }
293  return [];
294};

297/**
298 * Takes a string a prepends a '#' sign if one doesn't exist.
299 * Small helper method for use by bot.color and friends.
300 * @param {string} str String to check.
301 * @return {string} The value passed in, prepended with a '#' if it didn't
302 *     already have one.
303 * @private
304 */
305bot.color.prependHashIfNecessary_ = function(str) {
306  return str.charAt(0) == '#' ? str : '#' + str;
307};

310/**
311 * Takes an array and appends a 1 to it if the array only contains 3 elements.
312 * @param {!Array.<number>} arr The array to check.
313 * @return {!Array.<number>} The same array with a 1 appended
314 *  if it only contained 3 elements.
315 * @private
316 */
317bot.color.addAlphaIfNecessary_ = function(arr) {
318  if (arr.length == 3) {
319    arr.push(1);
320  }
321  return arr;
322};
