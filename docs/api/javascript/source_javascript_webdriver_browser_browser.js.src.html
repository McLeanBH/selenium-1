19/**
20 * The common name for the current user-agent. This will be the empty if the
21 * user-agent is not recognized.
22 * @private {string}
23 * @const
24 */
25webdriver.browser.BROWSER_NAME_ = (function() {
26  var map = {};
27  map[goog.userAgent.product.ANDROID] = webdriver.Browser.ANDROID;
28  map[goog.userAgent.product.CHROME] = webdriver.Browser.CHROME;
29  map[goog.userAgent.product.FIREFOX] = webdriver.Browser.FIREFOX;
30  map[goog.userAgent.IE] = webdriver.Browser.INTERNET_EXPLORER;
31  map[goog.userAgent.product.IPAD] = webdriver.Browser.IPAD;
32  map[goog.userAgent.product.IPHONE] = webdriver.Browser.IPHONE;
33  map[goog.userAgent.OPERA] = webdriver.Browser.OPERA;
34  map[goog.userAgent.product.SAFARI] = webdriver.Browser.SAFARI;
35  return map[true] || '';
36})();

47/**
48 * Creates new WebDriver clients backed by a pure-JavaScript command executor.
49 * @param {!Window} win The target window.
50 * @param {webdriver.promise.ControlFlow=} opt_flow The flow to
51 *     schedule commands through. Defaults to the active flow object.
52 * @return {webdriver.WebDriver} The new WebDriver instance.
53 */
54webdriver.browser.createDriver = function(win, opt_flow) {
55  var capabilities = new webdriver.Capabilities().
56      set(webdriver.Capability.BROWSER_NAME, webdriver.browser.BROWSER_NAME_);
57  var session = new webdriver.Session(
58      webdriver.browser.STATIC_SESSION_ID_, capabilities);
59  var executor = new webdriver.browser.CommandExecutor(win);
60  return new webdriver.WebDriver(session, executor, opt_flow);
61};
